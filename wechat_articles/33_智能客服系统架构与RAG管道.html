# src/knowledge/ingestion.py

"""

知识库文档摄入模块

支持PDF、HTML、Markdown等多种格式的文档导入

"""

import os

import hashlib

from typing import List, Optional

from dataclasses import dataclass, field

from datetime import datetime

 

from langchain_text_splitters import RecursiveCharacterTextSplitter

from langchain_community.document_loaders import (

    PyPDFLoader, UnstructuredHTMLLoader, UnstructuredMarkdownLoader

)

 

 

@dataclass

class DocumentChunk:

    """文档切片数据结构"""

    chunk_id: str

    content: str

    metadata: dict = field(default_factory=dict)

    embedding: Optional[List[float]] = None

 

 

class KnowledgeIngestion:

    """知识库摄入引擎"""

 

    # 根据文件扩展名映射加载器

    LOADER_MAP = {

        ".pdf": PyPDFLoader,

        ".html": UnstructuredHTMLLoader,

        ".htm": UnstructuredHTMLLoader,

        ".md": UnstructuredMarkdownLoader,

    }

 

    def __init__(self, chunk_size: int = 512, chunk_overlap: int = 64):

        self.splitter = RecursiveCharacterTextSplitter(

            chunk_size=chunk_size,

            chunk_overlap=chunk_overlap,

            separators=["\n\n", "\n", "。", "！", "？", ".", "!", "?", " "],

            length_function=len,

        )

 

    def load_document(self, file_path: str) -> List[DocumentChunk]:

        """加载并切分单个文档"""

        ext = os.path.splitext(file_path)[1].lower()

        loader_cls = self.LOADER_MAP.get(ext)

        if not loader_cls:

            raise ValueError(f"不支持的文件格式: {ext}")

 

        # 加载原始文档

        loader = loader_cls(file_path)

        raw_docs = loader.load()

 

        # 递归切分

        split_docs = self.splitter.split_documents(raw_docs)

 

        # 转换为自定义数据结构

        chunks = []

        for i, doc in enumerate(split_docs):

            chunk_id = self._generate_chunk_id(file_path, i)

            chunk = DocumentChunk(

                chunk_id=chunk_id,

                content=doc.page_content,

                metadata={

                    "source": file_path,

                    "chunk_index": i,

                    "total_chunks": len(split_docs),

                    "ingestion_time": datetime.now().isoformat(),

                    **doc.metadata,

                },

            )

            chunks.append(chunk)

 

        return chunks

 

    def batch_load(self, directory: str) -> List[DocumentChunk]:

        """批量加载目录下所有支持的文档"""

        all_chunks = []

        supported_exts = set(self.LOADER_MAP.keys())

 

        for root, _, files in os.walk(directory):

            for fname in files:

                ext = os.path.splitext(fname)[1].lower()

                if ext in supported_exts:

                    fpath = os.path.join(root, fname)

                    try:

                        chunks = self.load_document(fpath)

                        all_chunks.extend(chunks)

                        print(f"[OK] {fpath} -> {len(chunks)} chunks")

                    except Exception as e:

                        print(f"[FAIL] {fpath}: {e}")

 

        return all_chunks

 

    @staticmethod

    def _generate_chunk_id(file_path: str, index: int) -> str:

        """基于文件路径和切片序号生成唯一ID"""

        raw = f"{file_path}::{index}"

        return hashlib.md5(raw.encode()).hexdigest()

"""

RAG检索模块

实现混合检索（稀疏+稠密）+ RRF融合 + 重排序

"""

from typing import List, Tuple, Optional

from dataclasses import dataclass

 

import numpy as np

 

 

@dataclass

class RetrievalResult:

    """检索结果"""

    chunk_id: str

    content: str

    score: float

    source: str

    method: str  # "sparse" | "dense" | "faq"

 

 

class HybridRetriever:

    """混合检索器：BM25稀疏检索 + 向量稠密检索 + FAQ精确匹配"""

 

    def __init__(

        self,

        vector_store,

        bm25_index,

        faq_store,

        embedding_model,

        reranker_model=None,

        top_k: int = 10,

        rerank_top_k: int = 5,

        relevance_threshold: float = 0.7,

    ):

        self.vector_store = vector_store

        self.bm25_index = bm25_index

        self.faq_store = faq_store

        self.embedding_model = embedding_model

        self.reranker_model = reranker_model

        self.top_k = top_k

        self.rerank_top_k = rerank_top_k

        self.relevance_threshold = relevance_threshold

 

    def retrieve(self, query: str) -> List[RetrievalResult]:

        """执行混合检索"""

        # 第一步：并行检索

        dense_results = self._dense_search(query)

        sparse_results = self._sparse_search(query)

        faq_results = self._faq_search(query)

 

        # 第二步：RRF融合

        merged = self._rrf_fusion(dense_results, sparse_results, faq_results)

 

        # 第三步：重排序

        if self.reranker_model:

            merged = self._rerank(query, merged)

 

        # 第四步：相关性过滤

        filtered = [r for r in merged if r.score >= self.relevance_threshold]

 

        return filtered[: self.rerank_top_k]

 

    def _dense_search(self, query: str) -> List[RetrievalResult]:

        """向量稠密检索"""

        query_embedding = self.embedding_model.encode(query)

        results = self.vector_store.search(

            query_embedding, top_k=self.top_k

        )

        return [

            RetrievalResult(

                chunk_id=r["id"],

                content=r["content"],

                score=r["score"],

                source=r["metadata"].get("source", "unknown"),

                method="dense",

            )

            for r in results

        ]

 

    def _sparse_search(self, query: str) -> List[RetrievalResult]:

        """BM25稀疏检索"""

        results = self.bm25_index.search(query, top_k=self.top_k)

        return [

            RetrievalResult(

                chunk_id=r["id"],

                content=r["content"],

                score=r["score"],

                source=r["metadata"].get("source", "unknown"),

                method="sparse",

            )

            for r in results

        ]

 

    def _faq_search(self, query: str) -> List[RetrievalResult]:

        """FAQ精确匹配"""

        results = self.faq_store.match(query, top_k=5)

        return [

            RetrievalResult(

                chunk_id=r["id"],

                content=r["answer"],

                score=r["score"],

                source="faq",

                method="faq",

            )

            for r in results

        ]

 

    @staticmethod

    def _rrf_fusion(

        *result_lists: List[RetrievalResult], k: int = 60

    ) -> List[RetrievalResult]:

        """

        RRF (Reciprocal Rank Fusion) 倒数排名融合

        公式: score(d) = sum(1 / (k + rank_i(d)))

        """

        score_map = {}  # chunk_id -> (rrf_score, result_obj)

 

        for result_list in result_lists:

            for rank, result in enumerate(result_list):

                rrf_score = 1.0 / (k + rank + 1)

                if result.chunk_id in score_map:

                    old_score, old_result = score_map[result.chunk_id]

                    score_map[result.chunk_id] = (

                        old_score + rrf_score,

                        old_result,

                    )

                else:

                    score_map[result.chunk_id] = (rrf_score, result)

 

        # 按融合分数排序

        sorted_results = sorted(

            score_map.values(), key=lambda x: x[0], reverse=True

        )

        # 更新分数为归一化后的RRF分数

        if sorted_results:

            max_score = sorted_results[0][0]

            return [

                RetrievalResult(

                    chunk_id=r.chunk_id,

                    content=r.content,

                    score=s / max_score,  # 归一化到[0, 1]

                    source=r.source,

                    method="fusion",

                )

                for s, r in sorted_results

            ]

        return []

 

    def _rerank(

        self, query: str, results: List[RetrievalResult]

    ) -> List[RetrievalResult]:

        """使用Cross-Encoder重排序"""

        pairs = [(query, r.content) for r in results]

        scores = self.reranker_model.predict(pairs)

 

        for result, score in zip(results, scores):

            result.score = float(score)

 

        results.sort(key=lambda x: x.score, reverse=True)

        return results

"""

RAG回复生成模块

基于检索结果和对话历史生成回答

"""

from typing import List, Dict, Optional

 

 

class RAGGenerator:

    """RAG回复生成器"""

 

    SYSTEM_PROMPT = """你是一个专业的电商客服助手。请根据提供的参考资料回答用户问题。

 

规则：

1. 只根据参考资料中的信息回答，不要编造内容

2. 如果参考资料不足以回答问题，诚实告知并建议转人工客服

3. 回答要简洁、友好、专业

4. 涉及金额、时间等关键信息必须准确引用原文

5. 回答末尾不要重复问用户"还有什么可以帮助您的"

"""

 

    ANSWER_TEMPLATE = """参考资料：

{context}

 

对话历史：

{history}

 

用户问题：{query}

 

请根据以上参考资料回答用户的问题。"""

 

    def __init__(self, llm_client, max_context_length: int = 3000):

        self.llm = llm_client

        self.max_context_length = max_context_length

 

    def generate(

        self,

        query: str,

        retrieved_docs: List[Dict],

        conversation_history: Optional[List[Dict]] = None,

    ) -> str:

        """生成回复"""

        # 构建上下文

        context = self._build_context(retrieved_docs)

        history = self._format_history(conversation_history or [])

 

        # 填充模板

        user_prompt = self.ANSWER_TEMPLATE.format(

            context=context, history=history, query=query

        )

 

        # 调用大模型

        messages = [

            {"role": "system", "content": self.SYSTEM_PROMPT},

            {"role": "user", "content": user_prompt},

        ]

 

        response = self.llm.chat(messages, temperature=0.3, max_tokens=500)

        answer = response["choices"][0]["message"]["content"]

 

        # 幻觉检测（简易版：检查关键实体是否来自原文）

        if self._detect_hallucination(answer, context):

            answer = (

                "抱歉，我无法确认这个信息的准确性。"

                "建议您联系人工客服获取更准确的解答。"

            )

 

        return answer

 

    def _build_context(self, docs: List[Dict]) -> str:

        """构建上下文，控制总长度"""

        context_parts = []

        total_len = 0

        for i, doc in enumerate(docs):

            text = f"[资料{i+1}] {doc['content']}"

            if total_len + len(text) > self.max_context_length:

                break

            context_parts.append(text)

            total_len += len(text)

        return "\n\n".join(context_parts)

 

    @staticmethod

    def _format_history(history: List[Dict]) -> str:

        """格式化对话历史（取最近5轮）"""

        recent = history[-10:]  # 最近5轮（user+assistant各一条）

        lines = []

        for msg in recent:

            role = "用户" if msg["role"] == "user" else "客服"

            lines.append(f"{role}：{msg['content']}")

        return "\n".join(lines) if lines else "（无历史对话）"

 

    @staticmethod

    def _detect_hallucination(answer: str, context: str) -> bool:

        """

        简易幻觉检测：检查回答中的数字是否出现在上下文中

        生产环境建议使用专门的幻觉检测模型

        """

        import re

        # 提取回答中的数字

        answer_numbers = set(re.findall(r"\d+\.?\d*", answer))

        context_numbers = set(re.findall(r"\d+\.?\d*", context))

 

        # 如果回答中出现了上下文中不存在的"敏感数字"（如金额）

        suspicious = answer_numbers - context_numbers

        # 过滤掉常见的非敏感数字

        common_numbers = {"1", "2", "3", "5", "7", "10", "24", "30", "48"}

        suspicious -= common_numbers

 

        return len(suspicious) > 3  # 超过3个可疑数字则判定为幻觉
