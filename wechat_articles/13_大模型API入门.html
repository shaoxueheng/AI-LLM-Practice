8.3 Chat Completions 接口详解

8.3.1 请求结构

几乎所有主流 LLM API 都遵循相似的 Chat Completions 接口设计：

图8-8

8.3.2 完整请求示例

# 第8章/python

import os

from openai import OpenAI

 

# 初始化客户端

client = OpenAI(

    api_key=os.environ.get("OPENAI_API_KEY"),

    base_url="https://api.openai.com/v1"  # 可替换为国产API地址

)

 

# 发送请求

response = client.chat.completions.create(

    model="gpt-4o",

    messages=[

        {"role": "system", "content": "你是一个有帮助的AI助手。"},

        {"role": "user", "content": "请用一句话解释什么是 Transformer。"}

    ],

    temperature=0.7,     # 创造性程度 (0=确定性, 2=最随机)

    max_tokens=500,      # 最大输出长度

    top_p=0.9,          # 核采样概率

    presence_penalty=0,  # 话题多样性惩罚

    frequency_penalty=0, # 重复惩罚

)

 

# 解析响应

answer = response.choices[0].message.content

print(answer)

 

# 查看 Token 使用情况

print(f"输入Token: {response.usage.prompt_tokens}")

print(f"输出Token: {response.usage.completion_tokens}")

print(f"总Token: {response.usage.total_tokens}")

由于大多数国产 API 都兼容 OpenAI 格式，只需修改 base_url 和 api_key：

# OpenAI GPT-4o

client_openai = OpenAI(

    api_key=os.environ["OPENAI_API_KEY"],

)

 

# 通义千问 Qwen

client_qwen = OpenAI(

    api_key=os.environ["DASHSCOPE_API_KEY"],

    base_url="https://dashscope.aliyuncs.com/compatible-mode/v1"

# DeepSeek

client_deepseek = OpenAI(

    api_key=os.environ["DEEPSEEK_API_KEY"],

for name, client, model in [

    resp = client.chat.completions.create(

    )

8.4 流式输出 Streaming 与异步调用

8.4.1 为什么需要流式输出

普通请求需要等模型完整生成后才返回，对于长回复可能需要等待几十秒。流式输出让模型逐 Token 返回，用户可以实时看到生成过程。

图8-10

8.4.2 流式输出代码

# 第8章/python

# 流式输出示例

stream = client.chat.completions.create(

    model="gpt-4o",

    messages=[

        {"role": "user", "content": "请写一首关于春天的五言绝句"}

    ],

    stream=True  # 开启流式输出

)

 

# 逐 Token 打印

full_response = ""

for chunk in stream:

    if chunk.choices[0].delta.content is not None:

        token = chunk.choices[0].delta.content

        print(token, end="", flush=True)  # 实时打印

        full_response += token

 

print()  # 换行

print(f"\n完整回复: {full_response}")

8.4.3 异步调用

当需要同时处理多个请求时，异步调用可以大幅提升吞吐量：

# 第8章/python

import asyncio

from openai import AsyncOpenAI

 

async_client = AsyncOpenAI(api_key=os.environ["OPENAI_API_KEY"])

 

async def ask(question: str) -> str:

    """异步发送单个请求"""

    response = await async_client.chat.completions.create(

        model="gpt-4o-mini",

        messages=[{"role": "user", "content": question}]

    )

    return response.choices[0].message.content

 

async def batch_ask(questions: list) -> list:

    """批量并发请求"""

    tasks = [ask(q) for q in questions]

    results = await asyncio.gather(*tasks)

    return results

 

# 使用

questions = [

    "什么是机器学习？",

    "什么是深度学习？",

    "什么是强化学习？",

]

results = asyncio.run(batch_ask(questions))

for q, a in zip(questions, results):

    print(f"Q: {q}\nA: {a}\n")

8.5 多轮对话管理与上下文窗口

8.5.1 多轮对话的本质

LLM API 是无状态的——每次请求都是独立的。要实现多轮对话，需要在每次请求中发送完整的对话历史。

图8-12

8.5.2 上下文窗口限制与滑窗策略

每个模型都有最大上下文长度限制（如 GPT-4o 为 128K Token）。当对话历史超过限制时，需要进行截断。

图8-13

8.5.3 滑窗策略代码

# 第8章/python

class ConversationManager:

    """多轮对话管理器"""

 

    def __init__(self, system_prompt: str, max_tokens: int = 4096):

        self.system_prompt = system_prompt

        self.max_tokens = max_tokens

        self.messages = [{"role": "system", "content": system_prompt}]

 

    def add_user_message(self, content: str):

        self.messages.append({"role": "user", "content": content})

 

    def add_assistant_message(self, content: str):

        self.messages.append({"role": "assistant", "content": content})

 

    def get_messages(self) -> list:

        """获取在上下文窗口内的消息"""

        # 简单估算 Token 数（实际应使用 tiktoken）

        total_tokens = sum(len(m["content"]) // 2 for m in self.messages)

 

        if total_tokens <= self.max_tokens:

            return self.messages

 

        # 滑窗策略：保留 system + 最近的消息

        result = [self.messages[0]]  # 始终保留 system

        remaining = self.max_tokens - len(self.messages[0]["content"]) // 2

 

        # 从最后开始添加消息

        for msg in reversed(self.messages[1:]):

            msg_tokens = len(msg["content"]) // 2

            if remaining >= msg_tokens:

                result.insert(1, msg)

                remaining -= msg_tokens

            else:

                break

 

        return result
