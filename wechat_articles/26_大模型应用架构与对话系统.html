原则 说明 类比
**流式优先** 所有LLM交互使用SSE流式传输 自来水 vs 一桶水
**模型解耦** 业务逻辑不直接依赖特定模型 插座标准化，电器随便换
**无状态服务** 后端不保存会话状态，由缓存/DB管理 银行柜员不记客户，看身份证
**优雅降级** 模型超时时返回预设回复 电梯坏了还有楼梯
**可观测性** 记录每次调用的延迟、token数、成本 汽车仪表盘

# === FastAPI 模型服务 ===

from fastapi import FastAPI, HTTPException

from fastapi.responses import StreamingResponse

from pydantic import BaseModel, Field

from openai import OpenAI

import json

import uuid

import time

 

app = FastAPI(title="AI模型服务", version="1.0.0")

 

# --- 数据模型 ---

class Message(BaseModel):

    role: str = Field(..., description="角色: system/user/assistant")

    content: str = Field(..., description="消息内容")

 

class ChatRequest(BaseModel):

    model: str = Field(default="qwen-plus", description="模型名称")

    messages: list[Message] = Field(..., description="消息列表")

    temperature: float = Field(default=0.7, ge=0, le=2)

    max_tokens: int = Field(default=2048, ge=1, le=8192)

    stream: bool = Field(default=False, description="是否流式返回")

 

class ChatResponse(BaseModel):

    id: str

    object: str = "chat.completion"

    created: int

    model: str

    choices: list[dict]

    usage: dict

 

# --- 模型客户端 ---

client = OpenAI(api_key="your-api-key", base_url="https://api.example.com/v1")

 

# --- 流式生成器 ---

async def stream_generator(request: ChatRequest):

    """流式输出SSE事件"""

    response = client.chat.completions.create(

        model=request.model,

        messages=[m.model_dump() for m in request.messages],

        temperature=request.temperature,

        max_tokens=request.max_tokens,

        stream=True

    )

 

    for chunk in response:

        delta = chunk.choices[0].delta

        if delta.content:

            data = {

                "id": f"chatcmpl-{uuid.uuid4().hex[:8]}",

                "object": "chat.completion.chunk",

                "choices": [{"index": 0, "delta": {"content": delta.content}}]

            }

            yield f"data: {json.dumps(data, ensure_ascii=False)}\n\n"

 

    yield "data: [DONE]\n\n"

 

# --- API端点 ---

@app.post("/v1/chat/completions")

async def chat_completions(request: ChatRequest):

    """对话补全接口（支持流式/非流式）"""

    if request.stream:

        return StreamingResponse(

            stream_generator(request),

            media_type="text/event-stream"

        )

 

    # 非流式

    response = client.chat.completions.create(

        model=request.model,

        messages=[m.model_dump() for m in request.messages],

        temperature=request.temperature,

        max_tokens=request.max_tokens,

    )

 

    return ChatResponse(

        id=f"chatcmpl-{uuid.uuid4().hex[:8]}",

        created=int(time.time()),

        model=request.model,

        choices=[{

            "index": 0,

            "message": {

                "role": "assistant",

                "content": response.choices[0].message.content

            },

            "finish_reason": "stop"

        }],

        usage={

            "prompt_tokens": response.usage.prompt_tokens,

            "completion_tokens": response.usage.completion_tokens,

            "total_tokens": response.usage.total_tokens,

        }

    )

 

@app.get("/health")

async def health_check():

    return {"status": "healthy", "timestamp": time.time()}

 

# 启动: uvicorn main:app --host 0.0.0.0 --port 8000

    allow_origins=["*"],

    allow_methods=["*"],

    allow_headers=["*"],

)

 

# 请求日志中间件

@app.middleware("http")

async def log_requests(request: Request, call_next):

    start_time = time.time()

    response = await call_next(request)

    duration = time.time() - start_time

 

    logging.info(

        f"{request.method} {request.url.path} "

        f"- {response.status_code} - {duration:.3f}s"

    )

    return response

 

# 全局异常处理

@app.exception_handler(Exception)

async def global_exception_handler(request: Request, exc: Exception):

    logging.error(f"Unhandled error: {exc}", exc_info=True)

    return JSONResponse(

        status_code=500,

        content={"error": {"message": "Internal server error", "type": "server_error"}}

    )

import gradio as gr

from openai import OpenAI

 

client = OpenAI(api_key="your-key", base_url="https://api.example.com/v1")

 

def chat(message: str, history: list) -> str:

    """处理对话"""

    messages = [{"role": "system", "content": "你是一个有帮助的AI助手。"}]

 

    # 加入历史对话

    for h in history:

        messages.append({"role": "user", "content": h[0]})

        if h[1]:

            messages.append({"role": "assistant", "content": h[1]})

 

    messages.append({"role": "user", "content": message})

 

    # 流式输出

    response = client.chat.completions.create(

        model="qwen-plus",

        messages=messages,

        stream=True

    )

 

    partial = ""

    for chunk in response:

        if chunk.choices[0].delta.content:

            partial += chunk.choices[0].delta.content

            yield partial

 

# 创建聊天界面

demo = gr.ChatInterface(

    fn=chat,

    title="AI对话助手",

    description="基于大语言模型的智能对话系统",

    examples=["介绍一下你自己", "帮我写一首关于春天的诗", "解释量子计算"],

    theme=gr.themes.Soft(),

    retry_btn="重试",

    undo_btn="撤销",

    clear_btn="清空",

)

 

demo.launch(server_name="0.0.0.0", server_port=7860, share=True)

# 第14章/python

from openai import OpenAI

import json

 

class IntentRecognizer:

    """基于LLM的意图识别与槽位提取"""

 

    SYSTEM_PROMPT = """你是一个意图识别引擎。分析用户消息，返回JSON格式：

{

    "intent": "意图名称",

    "confidence": 0.0-1.0,

    "slots": {

        "slot_name": "slot_value"  // 提取到的信息

    },

    "missing_slots": ["slot_name"]  // 缺失的必要信息

}

 

可识别的意图：

- greeting: 问候

- chitchat: 闲聊

- knowledge_qa: 知识问答

- book_ticket: 订票

- check_weather: 查天气

- unknown: 无法识别"""

 

    def __init__(self, client: OpenAI):

        self.client = client

 

    def recognize(self, user_message: str, context: list = None) -> dict:

        messages = [

            {"role": "system", "content": self.SYSTEM_PROMPT}

        ]

        if context:

            messages.extend(context[-4:])  # 最近2轮对话上下文

        messages.append({"role": "user", "content": user_message})

 

        response = self.client.chat.completions.create(

            model="qwen-plus",

            messages=messages,

            response_format={"type": "json_object"},

            temperature=0.1  # 低温度保证稳定性

        )

 

        return json.loads(response.choices[0].message.content)

14.4.3 多轮对话管理器

# 第14章/python

class ConversationManager:

    """多轮对话管理器"""

 

    def __init__(self, client: OpenAI, system_prompt: str = "你是一个有帮助的助手。"):

        self.client = client

        self.system_prompt = system_prompt

        self.sessions: dict[str, list] = {}  # session_id -> messages

        self.intent_recognizer = IntentRecognizer(client)

 

    def get_or_create_session(self, session_id: str) -> list:

        if session_id not in self.sessions:

            self.sessions[session_id] = [

                {"role": "system", "content": self.system_prompt}

            ]

        return self.sessions[session_id]

 

    def chat(self, session_id: str, user_message: str) -> str:

        messages = self.get_or_create_session(session_id)

 

        # 1. 意图识别

        intent_result = self.intent_recognizer.recognize(

            user_message, messages[-4:]

        )

 

        # 2. 根据意图路由处理

        intent = intent_result.get("intent", "unknown")

 

        if intent == "knowledge_qa":

            # 走RAG路径（简化示例）

            enhanced_message = f"[知识问答模式] {user_message}"

        elif intent in ("book_ticket", "check_weather"):

            # 走任务型对话路径

            missing = intent_result.get("missing_slots", [])

            if missing:

                return f"好的，我来帮您处理。请补充以下信息：{', '.join(missing)}"

            enhanced_message = user_message

        else:

            enhanced_message = user_message

 

        # 3. 添加用户消息

        messages.append({"role": "user", "content": enhanced_message})

 

        # 4. 上下文窗口管理（保留最近N轮）

        if len(messages) > 21:  # system + 10轮对话

            messages = [messages[0]] + messages[-20:]

            self.sessions[session_id] = messages

 

        # 5. 调用LLM生成回复

        response = self.client.chat.completions.create(

            model="qwen-plus",

            messages=messages,

            temperature=0.7,

        )

 

        assistant_message = response.choices[0].message.content

        messages.append({"role": "assistant", "content": assistant_message})

 

        return assistant_message
