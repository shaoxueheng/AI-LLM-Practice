# 第7章/python

"""

第7章实战: 高质量 Prompt 模板库

文件: code/ch07/prompt_library.py

"""

 

from dataclasses import dataclass, field

from typing import Dict, List, Optional

import json

 

@dataclass

class PromptTemplate:

    """单个 Prompt 模板"""

    name: str                    # 模板名称

    category: str                # 分类

    description: str             # 描述

    system_prompt: str           # 系统提示

    user_template: str           # 用户消息模板（含变量占位符）

    variables: List[str]         # 需要填充的变量列表

    examples: List[Dict] = field(default_factory=list)  # 使用示例

    tags: List[str] = field(default_factory=list)        # 标签

 

    def render(self, **kwargs) -> Dict[str, str]:

        """渲染模板，填充变量"""

        # 检查是否提供了所有必需变量

        missing = set(self.variables) - set(kwargs.keys())

        if missing:

            raise ValueError(f"缺少变量: {missing}")

 

        user_msg = self.user_template

        for var, val in kwargs.items():

            user_msg = user_msg.replace(f"{{{{{var}}}}}", str(val))

 

        return {

            "system": self.system_prompt,

            "user": user_msg

        }

 

 

class PromptLibrary:

    """Prompt 模板库管理器"""

 

    def __init__(self):

        self.templates: Dict[str, PromptTemplate] = {}

        self._init_builtin_templates()

 

    def _init_builtin_templates(self):

        """初始化内置模板"""

 

        # === 通用模板 ===

        self.add(PromptTemplate(

            name="text_summary",

            category="通用",

            description="文本摘要提取",

            system_prompt="你是一个专业的文本分析师。请准确提取核心信息，不添加原文没有的内容。",

            user_template="请将以下文本总结为{{length}}字以内的摘要：\n\n{{text}}",

            variables=["text", "length"],

            tags=["摘要", "文本处理"]

        ))

 

        self.add(PromptTemplate(

            name="translation",

            category="通用",

            description="高质量翻译",

            system_prompt="你是一位精通{{source_lang}}和{{target_lang}}的专业翻译。翻译时注意：信达雅，保持原文风格，专业术语准确。",

            user_template="请将以下{{source_lang}}文本翻译为{{target_lang}}：\n\n{{text}}",

            variables=["source_lang", "target_lang", "text"],

            tags=["翻译", "多语言"]

        ))

 

        # === 分析模板 ===

        self.add(PromptTemplate(

            name="sentiment_analysis",

            category="分析",

            description="情感分析（结构化输出）",

            system_prompt="你是一个情感分析专家。请严格按照指定的 JSON 格式输出分析结果。",

            user_template="""请分析以下文本的情感，以 JSON 格式输出：

 

格式要求：

{

  "sentiment": "正面|负面|中性",

  "confidence": 0.0到1.0之间,

  "emotions": ["情感词1", "情感词2"],

  "summary": "一句话分析"

}

 

文本: "{{text}}"

 

请直接输出 JSON。""",

            variables=["text"],

            tags=["情感分析", "NLP", "JSON"]

        ))

 

        # === 编程模板 ===

        self.add(PromptTemplate(

            name="code_review",

            category="编程",

            description="代码审查",

            system_prompt="你是一位资深软件工程师，精通{{language}}。请从代码质量、性能、安全性、可读性四个维度进行审查。",

            user_template="""请审查以下{{language}}代码：

 

{{code}}

# 第7章/text

 

请按以下格式输出审查意见：

1. **整体评价**：简要总结

2. **问题列表**：按严重程度排序

3. **改进建议**：具体的代码修改建议

4. **优点**：值得肯定的地方""",

            variables=["language", "code"],

            tags=["代码审查", "编程"]

        ))

 

        # === 思维链模板 ===

        self.add(PromptTemplate(

            name="cot_reasoning",

            category="推理",

            description="思维链推理模板",

            system_prompt="你是一个善于逐步推理的分析师。在回答问题时，请先分析问题，再逐步推理，最后给出结论。",

            user_template="""请回答以下问题。

 

问题: {{question}}

 

请按以下步骤思考：

1. 【理解问题】问题的关键是什么？

2. 【收集信息】解答需要哪些已知条件？

3. 【逐步推理】一步一步推导...

4. 【验证答案】检查推理是否正确

5. 【最终答案】给出明确的结论""",

            variables=["question"],

            tags=["推理", "思维链", "CoT"]

        ))

 

    def add(self, template: PromptTemplate):

        """添加模板"""

        self.templates[template.name] = template

 

    def get(self, name: str) -> PromptTemplate:

        """获取模板"""

        if name not in self.templates:

            raise KeyError(f"模板 '{name}' 不存在")

        return self.templates[name]

 

    def search(self, keyword: str) -> List[PromptTemplate]:

        """按关键词搜索模板"""

        results = []

        kw = keyword.lower()

        for t in self.templates.values():

            if (kw in t.name.lower() or

                kw in t.description.lower() or

                any(kw in tag.lower() for tag in t.tags)):

                results.append(t)

        return results

 

    def list_categories(self) -> Dict[str, List[str]]:

        """列出所有分类及其模板"""

        categories = {}

        for t in self.templates.values():

            if t.category not in categories:

                categories[t.category] = []

            categories[t.category].append(t.name)

        return categories

 

    def export_json(self, filepath: str):

        """导出模板库为 JSON"""

        data = {}

        for name, t in self.templates.items():

            data[name] = {

                "name": t.name,

                "category": t.category,

                "description": t.description,

                "system_prompt": t.system_prompt,

                "user_template": t.user_template,

                "variables": t.variables,

                "tags": t.tags,

            }

        with open(filepath, 'w', encoding='utf-8') as f:

            json.dump(data, f, ensure_ascii=False, indent=2)

        print(f"模板库已导出到 {filepath}")

# 使用示例

# ============================================================

 

if __name__ == "__main__":

    # 创建模板库

    library = PromptLibrary()

 

    # 列出所有分类

    print("=== 模板库分类 ===")

    for cat, names in library.list_categories().items():

        print(f"  {cat}: {', '.join(names)}")

 

    # 使用情感分析模板

    print("\n=== 情感分析模板示例 ===")

    template = library.get("sentiment_analysis")

    result = template.render(text="这家餐厅的菜品很精致，但等了很久才上菜")

    print(f"System: {result['system']}")

    print(f"User: {result['user']}")

 

    # 使用思维链模板

    print("\n=== 思维链推理模板示例 ===")

    template = library.get("cot_reasoning")

    result = template.render(question="如果一个水池有两个进水管和一个出水管，进水管每小时分别进3吨和5吨水，出水管每小时排2吨水，水池容量100吨，从空池开始多久能注满？")

    print(f"User: {result['user']}")

 

    # 搜索模板

    print("\n=== 搜索 '分析' 相关模板 ===")

    results = library.search("分析")

    for t in results:

        print(f"  {t.name}: {t.description}")

 

    # 导出模板库

    library.export_json("prompt_library.json")
