"""

Agent基础类和角色定义

每个Agent都是一个独立的"专业人士"

"""

from abc import ABC, abstractmethod

from typing import Dict, List, Any, Optional

from dataclasses import dataclass, field

import uuid

import time

 

 

@dataclass

class AgentMessage:

    """Agent间通信消息"""

    msg_id: str = field(default_factory=lambda: str(uuid.uuid4())[:8])

    sender: str = ""

    receiver: str = ""

    msg_type: str = "task"  # task | result | feedback | error

    content: Any = None

    metadata: Dict = field(default_factory=dict)

    timestamp: float = field(default_factory=time.time)

 

 

@dataclass

class AgentConfig:

    """Agent配置"""

    name: str

    role: str

    description: str

    system_prompt: str

    model: str = "qwen2.5-7b-instruct"

    temperature: float = 0.3

    max_tokens: int = 2000

    tools: List[str] = field(default_factory=list)

    max_iterations: int = 10

 

 

class BaseAgent(ABC):

    """Agent基类"""

 

    def __init__(self, config: AgentConfig, llm_client):

        self.config = config

        self.llm = llm_client

        self.agent_id = f"{config.role}_{uuid.uuid4().hex[:6]}"

        self.message_history: List[AgentMessage] = []

 

    @abstractmethod

    async def execute(

        self, task: Dict, context: Dict

    ) -> Dict:

        """执行任务——每个角色的核心逻辑不同"""

        pass

 

    def _call_llm(

        self,

        user_prompt: str,

        extra_context: str = "",

    ) -> str:

        """调用大模型"""

        system = self.config.system_prompt

        if extra_context:

            system += f"\n\n额外上下文：{extra_context}"

 

        messages = [

            {"role": "system", "content": system},

            {"role": "user", "content": user_prompt},

        ]

 

        response = self.llm.chat(

            messages,

            model=self.config.model,

            temperature=self.config.temperature,

            max_tokens=self.config.max_tokens,

        )

        return response["choices"][0]["message"]["content"]

class PlannerAgent(BaseAgent):

    """规划者Agent：将复杂任务分解为DAG"""

 

    SYSTEM_PROMPT = """你是一个任务规划专家。你的职责是：

1. 分析用户的复杂任务需求

2. 将任务分解为多个可独立执行的子任务

3. 识别子任务之间的依赖关系

4. 输出结构化的任务DAG（JSON格式）

 

输出格式要求：

{

"tasks": [

{

"task_id": "t1",

"name": "任务名称",

"description": "详细描述",

"agent_role": "执行此任务的角色",

"dependencies": ["依赖的task_id"],

"expected_output": "预期输出描述"

}

]

}

# 第19章/text

 

可用的角色有：

- data_collector: 数据采集（API调用、数据库查询、文件读取）

- data_analyst: 数据分析（统计分析、建模、可视化）

- report_writer: 报告撰写（文字生成、格式排版）

- tool_specialist: 工具操作（代码执行、文件操作）

- quality_reviewer: 质量审核（事实核查、格式检查）"""

 

    def __init__(self, llm_client):

        config = AgentConfig(

            name="Planner",

            role="planner",

            description="任务分解与规划",

            system_prompt=self.SYSTEM_PROMPT,

            temperature=0.2,

            max_tokens=3000,

        )

        super().__init__(config, llm_client)

 

    async def execute(self, task: Dict, context: Dict) -> Dict:

        """分解任务为DAG"""

        import json

 

        prompt = f"""请将以下任务分解为可执行的子任务DAG：

 

任务描述：{task.get('description', '')}

额外要求：{task.get('requirements', '')}

可用数据源：{task.get('data_sources', '')}

 

请输出JSON格式的任务DAG。"""

 

        result = self._call_llm(prompt)

 

        # 解析JSON

        try:

            # 提取JSON块

            json_start = result.find("{")

            json_end = result.rfind("}") + 1

            if json_start >= 0 and json_end > json_start:

                dag = json.loads(result[json_start:json_end])

                return {"status": "success", "dag": dag}

        except json.JSONDecodeError:

            pass

 

        return {"status": "error", "message": "Failed to parse DAG"}

 

 

class DataCollectorAgent(BaseAgent):

    """数据采集Agent"""

 

    SYSTEM_PROMPT = """你是一个数据采集专家。你的职责是：

1. 根据任务要求从指定数据源获取数据

2. 支持API调用、数据库查询、文件读取

3. 对获取的数据进行初步验证

4. 返回结构化的数据结果

 

如果需要执行代码，请输出完整可运行的Python代码。"""

 

    def __init__(self, llm_client, tool_executor=None):

        config = AgentConfig(

            name="DataCollector",

            role="data_collector",

            description="数据采集与获取",

            system_prompt=self.SYSTEM_PROMPT,

            tools=["api_call", "db_query", "file_read"],

        )

        super().__init__(config, llm_client)

        self.tool_executor = tool_executor

 

    async def execute(self, task: Dict, context: Dict) -> Dict:

        """执行数据采集"""

        prompt = f"""请完成以下数据采集任务：

 

任务：{task.get('description', '')}

数据源：{task.get('data_source', '')}

前置任务结果：{context}

 

请描述你的采集方案，并生成可执行的代码。"""

 

        plan = self._call_llm(prompt)

 

        # 如果有工具执行器，尝试执行代码

        if self.tool_executor:

            code = self._extract_code(plan)

            if code:

                result = await self.tool_executor.run_code(code)

                return {"status": "success", "data": result, "plan": plan}

 

        return {"status": "success", "plan": plan}

 

    @staticmethod

    def _extract_code(text: str) -> Optional[str]:

        """从LLM输出中提取代码块"""

        import re

        pattern = r"```python\n(.*?)```"

        match = re.search(pattern, text, re.DOTALL)

        return match.group(1).strip() if match else None

"""

工具注册与执行框架

支持工具注册、权限管理、安全执行

"""

import asyncio

import subprocess

import tempfile

import os

from typing import Dict, Any, Callable, Awaitable, Optional

from dataclasses import dataclass, field

from enum import Enum

import logging

 

logger = logging.getLogger(__name__)

 

 

class ToolPermission(Enum):

    READ_ONLY = "read_only"

    READ_WRITE = "read_write"

    EXECUTE = "execute"

    NETWORK = "network"

    ADMIN = "admin"

 

 

@dataclass

class ToolDefinition:

    """工具定义"""

    name: str

    description: str

    parameters: Dict[str, Any]

    handler: Callable[..., Awaitable[Any]]

    permissions: list = field(default_factory=list)

    timeout: int = 60

 

 

class ToolRegistry:

    """工具注册中心"""

 

    def __init__(self):

        self._tools: Dict[str, ToolDefinition] = {}

        self._execution_log: list = []

 

    def register(self, tool: ToolDefinition):

        """注册工具"""

        self._tools[tool.name] = tool

        logger.info(f"Tool registered: {tool.name}")

 

    def get_tool_schemas(self) -> list:

        """获取所有工具的Schema（供LLM使用）"""

        schemas = []

        for tool in self._tools.values():

            schemas.append({

                "type": "function",

                "function": {

                    "name": tool.name,

                    "description": tool.description,

                    "parameters": tool.parameters,

                },

            })

        return schemas

 

    async def execute_tool(

        self,

        tool_name: str,

        arguments: Dict,

        agent_permissions: list = None,

    ) -> Dict:

        """执行工具（带权限检查和超时控制）"""

        tool = self._tools.get(tool_name)

        if not tool:

            return {"error": f"Tool not found: {tool_name}"}

 

        # 权限检查

        if agent_permissions:

            required = set(tool.permissions)

            granted = set(agent_permissions)

            if not required.issubset(granted):

                missing = required - granted

                return {

                    "error": f"Permission denied. Missing: {missing}"

                }

 

        # 执行（带超时）

        try:

            result = await asyncio.wait_for(

                tool.handler(**arguments),

                timeout=tool.timeout,

            )

            self._execution_log.append({

                "tool": tool_name,

                "args": arguments,

                "status": "success",

            })

            return {"result": result}

        except asyncio.TimeoutError:

            return {"error": f"Tool {tool_name} timed out"}

        except Exception as e:

            return {"error": f"Tool {tool_name} failed: {str(e)}"}

class PythonExecutor:

    """安全的Python代码执行器"""

 

    def __init__(self, allowed_modules=None):

        self.allowed_modules = allowed_modules or [

            "pandas", "numpy", "matplotlib", "seaborn",

            "json", "csv", "math", "statistics",

        ]

 

    async def run_code(self, code: str) -> Dict:

        """在隔离环境中执行Python代码"""

        # 安全检查

        dangerous_patterns = [

            "import os", "import sys", "import subprocess",

            "__import__", "eval(", "exec(", "open(",

            "os.system", "shutil.rmtree",

        ]

        for pattern in dangerous_patterns:

            if pattern in code:

                return {

                    "error": f"Security violation: {pattern} is not allowed"

                }

 

        # 写入临时文件

        with tempfile.NamedTemporaryFile(

            mode="w", suffix=".py", delete=False

        ) as f:

            # 注入安全限制

            safe_code = (

                "import resource\n"

                "resource.setrlimit("

                "resource.RLIMIT_CPU, (30, 30))\n"

                f"{code}\n"

            )

            f.write(safe_code)

            temp_path = f.name

 

        try:

            # 在子进程中执行

            proc = await asyncio.create_subprocess_exec(

                "python3",

                temp_path,

                stdout=asyncio.subprocess.PIPE,

                stderr=asyncio.subprocess.PIPE,

            )

            stdout, stderr = await asyncio.wait_for(

                proc.communicate(), timeout=60

            )

 

            if proc.returncode == 0:

                return {

                    "status": "success",

                    "output": stdout.decode("utf-8"),

                }

            else:

                return {

                    "status": "error",

                    "error": stderr.decode("utf-8"),

                }

        finally:

            os.unlink(temp_path)

 

 

async def api_call_handler(

    url: str,

    method: str = "GET",

    headers: Optional[Dict] = None,

    body: Optional[Dict] = None,

) -> Dict:

    """HTTP API调用工具"""

    import aiohttp

 

    async with aiohttp.ClientSession() as session:

        kwargs = {"headers": headers or {}}

        if body:

            kwargs["json"] = body

 

        async with session.request(method, url, **kwargs) as resp:

            return {

                "status_code": resp.status,

                "body": await resp.json(),

            }

 

 

def build_default_registry() -> ToolRegistry:

    """构建默认工具注册中心"""

    registry = ToolRegistry()

    executor = PythonExecutor()

 

    # 注册Python执行器

    registry.register(ToolDefinition(

        name="python_exec",

        description="执行Python代码，用于数据处理和分析",

        parameters={

            "type": "object",

            "properties": {

                "code": {

                    "type": "string",

                    "description": "要执行的Python代码",

                },

            },

            "required": ["code"],

        },

        handler=executor.run_code,

        permissions=[ToolPermission.EXECUTE],

        timeout=120,

    ))

 

    # 注册API调用工具

    registry.register(ToolDefinition(

        name="api_call",

        description="调用HTTP API获取数据",

        parameters={

            "type": "object",

            "properties": {

                "url": {"type": "string"},

                "method": {"type": "string", "default": "GET"},

                "headers": {"type": "object"},

                "body": {"type": "object"},

            },

            "required": ["url"],

        },

        handler=api_call_handler,

        permissions=[ToolPermission.NETWORK],

        timeout=30,

    ))

 

    return registry
