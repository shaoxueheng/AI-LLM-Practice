8.6 Function Calling / Tool Use 工具调用

8.6.1 什么是工具调用

Function Calling 让大模型可以"调用外部工具"——当模型判断需要使用工具时，它会生成一个结构化的函数调用请求，由你的程序执行后把结果返回给模型。

8.6.2 Function Calling 完整代码

# 第8章/python

import json

from openai import OpenAI

 

client = OpenAI()

 

# 第1步：定义工具（函数签名）

tools = [

    {

        "type": "function",

        "function": {

            "name": "get_weather",

            "description": "获取指定城市的当前天气信息",

            "parameters": {

                "type": "object",

                "properties": {

                    "city": {

                        "type": "string",

                        "description": "城市名称，如'北京'、'上海'"

                    },

                    "unit": {

                        "type": "string",

                        "enum": ["celsius", "fahrenheit"],

                        "description": "温度单位"

                    }

                },

                "required": ["city"]

            }

        }

    },

    {

        "type": "function",

        "function": {

            "name": "search_web",

            "description": "搜索互联网获取最新信息",

            "parameters": {

                "type": "object",

                "properties": {

                    "query": {

                        "type": "string",

                        "description": "搜索关键词"

                    }

                },

                "required": ["query"]

            }

        }

    }

]

 

# 第2步：模拟工具实现

def get_weather(city: str, unit: str = "celsius") -> str:

    """模拟天气API"""

    weather_data = {

        "北京": {"temp": 25, "condition": "晴"},

        "上海": {"temp": 28, "condition": "多云"},

        "广州": {"temp": 32, "condition": "阴"},

    }

    data = weather_data.get(city, {"temp": 20, "condition": "未知"})

    return json.dumps({"city": city, "temperature": data["temp"], "condition": data["condition"]})

 

def search_web(query: str) -> str:

    """模拟搜索API"""

    return json.dumps({"results": [f"搜索'{query}'的结果: 这是一条模拟搜索结果"]})

 

# 工具路由表

tool_functions = {

    "get_weather": get_weather,

    "search_web": search_web,

}

 

# 第3步：发送请求

messages = [

    {"role": "system", "content": "你是一个有帮助的助手，可以查询天气和搜索信息。"},

    {"role": "user", "content": "北京和上海今天天气怎么样？"}

]

 

response = client.chat.completions.create(

    model="gpt-4o",

    messages=messages,

    tools=tools,

    tool_choice="auto"  # 让模型自动决定是否调用工具

)

 

# 第4步：处理工具调用

assistant_message = response.choices[0].message

messages.append(assistant_message)

 

if assistant_message.tool_calls:

    for tool_call in assistant_message.tool_calls:

        func_name = tool_call.function.name

        func_args = json.loads(tool_call.function.arguments)

 

        # 执行工具

        result = tool_functions[func_name](**func_args)

 

        # 将工具结果加入消息

        messages.append({

            "role": "tool",

            "tool_call_id": tool_call.id,

            "content": result

        })

 

    # 第5步：让模型根据工具结果生成最终回复

    final_response = client.chat.completions.create(

        model="gpt-4o",

        messages=messages

    )

    print(final_response.choices[0].message.content)
