步骤 3：编写调用代码

# 第1章/python

"""

第1章实战：体验大模型 API 调用

使用 OpenAI 兼容接口调用 DeepSeek 模型

"""

from openai import OpenAI

 

# 配置 API（以 DeepSeek 为例）

client = OpenAI(

    api_key="your-api-key-here",  # 替换为你的 API Key

    base_url="https://api.deepseek.com"  # DeepSeek 的 API 地址

)

 

def chat_with_model(prompt: str) -> str:

    """与大模型对话"""

    response = client.chat.completions.create(

        model="deepseek-chat",

        messages=[

            {"role": "system", "content": "你是一个友好的AI助手。"},

            {"role": "user", "content": prompt}

        ],

        temperature=0.7,

        max_tokens=1000

    )

    return response.choices[0].message.content

 

def main():

    print("=" * 60)

    print("欢迎体验大模型 API 调用！")

    print("=" * 60)

 

    # 测试1：知识问答

    print("\n【测试1：知识问答】")

    answer = chat_with_model("用一句话解释什么是大语言模型？")

    print(f"回答：{answer}")

 

    # 测试2：文本生成

    print("\n【测试2：创意写作】")

    story = chat_with_model("用三句话写一个关于AI学习编程的小故事。")

    print(f"故事：{story}")

 

    # 测试3：代码生成

    print("\n【测试3：代码生成】")

    code = chat_with_model("用Python写一个计算斐波那契数列前10项的函数。")

    print(f"代码：{code}")

 

    # 测试4：翻译

    print("\n【测试4：翻译】")

    translation = chat_with_model("将以下句子翻译成英文：人工智能正在改变世界的运行方式。")

    print(f"翻译：{translation}")

 

    # 交互式对话

    print("\n" + "=" * 60)

    print("进入交互模式（输入 'quit' 退出）：")

    print("=" * 60)

 

    while True:

        user_input = input("\n你：")

        if user_input.lower() in ['quit', 'exit', 'q']:

            print("再见！")

            break

        response = chat_with_model(user_input)

        print(f"\nAI：{response}")

 

if __name__ == "__main__":

    main()

步骤 4：运行并观察
