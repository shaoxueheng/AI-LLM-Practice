from typing import TypedDict, List, Literal, Annotated

from langgraph.graph import StateGraph, START, END

from langchain_openai import ChatOpenAI

from langchain_core.messages import HumanMessage, SystemMessage

import operator

 

# ===== 状态定义 =====

class ResearchState(TypedDict):

    topic: str

    research_data: Annotated[list, operator.add]  # 可追加的列表

    analysis: str

    report: str

    review_feedback: str

    is_approved: bool

 

llm = ChatOpenAI(model="gpt-4o-mini")

 

# ===== Agent 角色定义 =====

def researcher_agent(state: ResearchState) -> dict:

    """研究员：搜索和收集信息"""

    messages = [

        SystemMessage(content="你是研究员。请围绕给定主题搜索关键信息，列出5个核心要点。"),

        HumanMessage(content=f"研究主题：{state['topic']}"),

    ]

    result = llm.invoke(messages)

    return {"research_data": [result.content]}

 

def analyst_agent(state: ResearchState) -> dict:

    """分析师：分析和总结"""

    data = "\n".join(state["research_data"])

    messages = [

        SystemMessage(content="你是数据分析师。基于研究数据进行深度分析，识别趋势和洞察。"),

        HumanMessage(content=f"研究数据：\n{data}"),

    ]

    result = llm.invoke(messages)

    return {"analysis": result.content}

 

def writer_agent(state: ResearchState) -> dict:

    """作家：撰写报告"""

    messages = [

        SystemMessage(content="你是专业作家。基于分析结果撰写结构化报告。"),

        HumanMessage(content=f"分析结果：\n{state['analysis']}"),

    ]

    result = llm.invoke(messages)

    return {"report": result.content}

 

def reviewer_agent(state: ResearchState) -> dict:

    """审核员：审查报告质量"""

    messages = [

        SystemMessage(content="""你是审核员。评估报告质量。

如果满意，回复"APPROVED"。

如果需要修改，给出具体意见。"""),

        HumanMessage(content=f"报告：\n{state['report']}"),

    ]

    result = llm.invoke(messages)

    is_approved = "APPROVED" in result.content.upper()

    return {"review_feedback": result.content, "is_approved": is_approved}

 

def route_review(state: ResearchState) -> Literal["writer", "__end__"]:

    return "__end__" if state["is_approved"] else "writer"

 

# ===== 构建图 =====

graph = StateGraph(ResearchState)

graph.add_node("researcher", researcher_agent)

graph.add_node("analyst", analyst_agent)

graph.add_node("writer", writer_agent)

graph.add_node("reviewer", reviewer_agent)

 

graph.add_edge(START, "researcher")

graph.add_edge("researcher", "analyst")

graph.add_edge("analyst", "writer")

graph.add_edge("writer", "reviewer")

graph.add_conditional_edges("reviewer", route_review)

 

team = graph.compile()

 

# ===== 运行 =====

result = team.invoke({

    "topic": "2025年大模型技术发展趋势",

    "research_data": [],

})

print(result["report"])

"""Agent 长期记忆实现"""

import chromadb

from sentence_transformers import SentenceTransformer

from datetime import datetime

import json

 

class AgentLongTermMemory:

    """基于向量数据库的 Agent 长期记忆"""

 

    def __init__(self, db_path: str = "./agent_memory"):

        self.embedder = SentenceTransformer("BAAI/bge-large-zh-v1.5")

        self.client = chromadb.PersistentClient(path=db_path)

 

        # 两个记忆集合：经验和事实

        self.experience = self.client.get_or_create_collection("experience")

        self.facts = self.client.get_or_create_collection("facts")

 

    def save_experience(self, task: str, result: str, success: bool):

        """保存任务执行经验"""

        memory = {

            "task": task,

            "result": result,

            "success": success,

            "timestamp": datetime.now().isoformat(),

        }

        text = f"任务: {task}\n结果: {result}\n成功: {success}"

        embedding = self.embedder.encode([text]).tolist()

 

        self.experience.add(

            ids=[f"exp_{datetime.now().timestamp()}"],

            documents=[text],

            embeddings=embedding,

            metadatas=[memory],

        )

 

    def save_fact(self, fact: str, source: str):

        """保存学到的事实"""

        embedding = self.embedder.encode([fact]).tolist()

        self.facts.add(

            ids=[f"fact_{datetime.now().timestamp()}"],

            documents=[fact],

            embeddings=embedding,

            metadatas=[{"source": source, "timestamp": datetime.now().isoformat()}],

        )

 

    def recall_experience(self, current_task: str, top_k: int = 3) -> list:

        """回忆相关经验"""

        embedding = self.embedder.encode([current_task]).tolist()

        results = self.experience.query(

            query_embeddings=embedding, n_results=top_k

        )

        return results["documents"][0] if results["documents"] else []

 

    def recall_facts(self, query: str, top_k: int = 5) -> list:

        """检索相关事实"""

        embedding = self.embedder.encode([query]).tolist()

        results = self.facts.query(

            query_embeddings=embedding, n_results=top_k

        )

        return results["documents"][0] if results["documents"] else []

12.7.1 MCP 是什么

MCP（Model Context Protocol）是 Anthropic 于 2024 年底推出的开放协议标准，旨在为 AI 模型提供一种统一的方式来连接外部工具和数据源。你可以把它理解为"AI 时代的 USB 接口"——就像 USB 让各种设备都能连接电脑，MCP 让各种工具都能连接 AI 模型。

    raise ValueError(f"未知工具: {name}")

 

# ===== 启动服务器 =====

async def main():

        await server.run(read_stream, write_stream)

# 第12章/python

"""

全能 Agent：联网搜索 + 代码执行

基于 LangGraph 构建

"""

 

import os

import json

import subprocess

import tempfile

from typing import TypedDict, List, Annotated, Literal

from langchain_openai import ChatOpenAI

from langchain_core.messages import (

    HumanMessage, AIMessage, SystemMessage, ToolMessage

)

from langchain_core.tools import tool

from langgraph.graph import StateGraph, START, END

from langgraph.graph.message import add_messages

from langgraph.prebuilt import ToolNode

 

 

# ===== 1. 工具定义 =====

 

@tool

def search_web(query: str) -> str:

    """搜索互联网获取实时信息。输入为搜索关键词。"""

    try:

        import requests

        response = requests.post(

            "https://api.tavily.com/search",

            json={

                "api_key": os.getenv("TAVILY_API_KEY"),

                "query": query,

                "max_results": 5,

                "include_answer": True,

            },

            timeout=10,

        )

        data = response.json()

        if "answer" in data and data["answer"]:

            return f"简要回答: {data['answer']}\n\n详细结果:\n" + "\n".join(

                f"- {r['title']}: {r['content'][:200]}" for r in data.get("results", [])

            )

        return "\n".join(

            f"- {r['title']}: {r['content'][:200]}" for r in data.get("results", [])

        )

    except Exception as e:

        return f"搜索失败: {e}"

 

 

@tool

def run_python(code: str) -> str:

    """在安全沙箱中执行 Python 代码。支持数据分析、计算等。

    代码中可以使用 print() 输出结果。

    已预装库：numpy, pandas, matplotlib。

    """

    # 安全检查

    forbidden = ["os.system", "subprocess", "eval(", "exec(", "__import__",

                 "shutil.rmtree", "os.remove"]

    for word in forbidden:

        if word in code:

            return f"安全检查未通过：禁止使用 {word}"

 

    with tempfile.NamedTemporaryFile(mode="w", suffix=".py", delete=False) as f:

        f.write(code)

        f.flush()

        try:

            result = subprocess.run(

                ["python", f.name],

                capture_output=True,

                text=True,

                timeout=30,  # 30秒超时

            )

            output = result.stdout

            if result.stderr:

                output += f"\n[错误]: {result.stderr}"

            return output if output.strip() else "代码执行成功（无输出）"

        except subprocess.TimeoutExpired:

            return "执行超时（超过30秒）"

        finally:

            os.unlink(f.name)

 

 

@tool

def read_file(file_path: str) -> str:

    """读取文件内容。输入为文件路径。"""

    try:

        with open(file_path, "r", encoding="utf-8") as f:

            content = f.read()

        if len(content) > 5000:

            return content[:5000] + f"\n...(共 {len(content)} 字符，已截断)"

        return content

    except Exception as e:

        return f"读取失败: {e}"

 

 

@tool

def write_file(file_path: str, content: str) -> str:

    """将内容写入文件。参数：file_path 文件路径, content 写入内容。"""

    try:

        with open(file_path, "w", encoding="utf-8") as f:

            f.write(content)

        return f"已成功写入 {file_path}（{len(content)} 字符）"

    except Exception as e:

        return f"写入失败: {e}"

 

 

@tool

def calculator(expression: str) -> str:

    """计算数学表达式。如: '2**10', 'sqrt(144)', '3.14*5**2'。"""

    import math

    safe_dict = {

        "sqrt": math.sqrt, "sin": math.sin, "cos": math.cos,

        "tan": math.tan, "log": math.log, "pi": math.pi, "e": math.e,

        "abs": abs, "round": round, "pow": pow, "max": max, "min": min,

    }

    try:

        result = eval(expression, {"__builtins__": {}}, safe_dict)

        return str(result)

    except Exception as e:

        return f"计算错误: {e}"

 

 

# ===== 2. Agent 状态 =====

 

class AgentState(TypedDict):

    messages: Annotated[list, add_messages]

 

 

# ===== 3. 构建 Agent =====

 

tools = [search_web, run_python, read_file, write_file, calculator]

 

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

llm_with_tools = llm.bind_tools(tools)

 

SYSTEM_PROMPT = """你是一个全能 AI 助手，具备以下能力：

 

1. **联网搜索**: 使用 search_web 工具查找最新信息

2. **代码执行**: 使用 run_python 工具执行 Python 代码（支持 numpy, pandas, matplotlib）

3. **文件操作**: 使用 read_file/write_file 读写文件

4. **数学计算**: 使用 calculator 工具进行精确计算

 

工作原则：

- 面对复杂问题，先分解为小步骤

- 需要最新信息时主动搜索

- 需要精确计算时使用工具而非心算

- 数据分析任务优先使用 Python 代码

- 最终给出清晰、结构化的回答"""

 

 

def agent_node(state: AgentState) -> dict:

    """Agent 推理节点"""

    messages = [SystemMessage(content=SYSTEM_PROMPT)] + state["messages"]

    response = llm_with_tools.invoke(messages)

    return {"messages": [response]}

 

 

def should_continue(state: AgentState) -> Literal["tools", "__end__"]:

    """判断是否需要调用工具"""

    last_message = state["messages"][-1]

    if hasattr(last_message, "tool_calls") and last_message.tool_calls:

        return "tools"

    return "__end__"

 

 

# 工具执行节点

tool_node = ToolNode(tools)

 

# ===== 4. 构建图 =====

graph = StateGraph(AgentState)

 

graph.add_node("agent", agent_node)

graph.add_node("tools", tool_node)

 

graph.add_edge(START, "agent")

graph.add_conditional_edges("agent", should_continue)

graph.add_edge("tools", "agent")  # 工具执行后回到 Agent

 

app = graph.compile()

 

 

# ===== 5. 交互式对话 =====

def chat():

    """交互式对话入口"""

    print("=" * 50)

    print("  全能 AI Agent（搜索 + 代码 + 文件）")

    print("  输入 'quit' 退出")

    print("=" * 50)

 

    messages = []

 

    while True:

        user_input = input("\n你: ").strip()

        if user_input.lower() in ("quit", "exit", "q"):

            print("再见！")

            break

 

        messages.append(HumanMessage(content=user_input))

        result = app.invoke({"messages": messages})

        messages = result["messages"]

 

        # 打印 Agent 的最终回答

        final_msg = messages[-1]

        print(f"\nAgent: {final_msg.content}")

 

 

if __name__ == "__main__":

    chat()
