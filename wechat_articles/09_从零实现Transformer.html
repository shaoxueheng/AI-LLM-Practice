# 第5章/python

"""

第5章实战：从零实现一个简易 Transformer (GPT 风格)

文件: code/ch05/mini_transformer.py

 

这是一个字符级语言模型，使用 Transformer 解码器架构。

"""

 

import torch

import torch.nn as nn

import torch.nn.functional as F

import math

 

# ============================================================

# 超参数配置

# ============================================================

class Config:

    block_size = 128       # 上下文窗口长度

    vocab_size = 65        # 字符词表大小（训练时自动设置）

    n_layer = 4            # Transformer 层数

    n_head = 4             # 注意力头数

    n_embd = 128           # 嵌入维度

    dropout = 0.1          # Dropout 比例

    batch_size = 32        # 批大小

    learning_rate = 3e-4   # 学习率

    max_iters = 5000       # 训练迭代数

    eval_interval = 500    # 评估间隔

    device = 'cuda' if torch.cuda.is_available() else 'cpu'

 

# ============================================================

# 核心组件

# ============================================================

 

class RMSNorm(nn.Module):

    """RMSNorm 归一化层 —— 现代 Transformer 的标配"""

    def __init__(self, dim, eps=1e-6):

        super().__init__()

        self.eps = eps

        self.weight = nn.Parameter(torch.ones(dim))

 

    def forward(self, x):

        # 计算 RMS（均方根）

        rms = torch.sqrt(torch.mean(x ** 2, dim=-1, keepdim=True) + self.eps)

        return x / rms * self.weight

 

 

class RotaryPositionEmbedding(nn.Module):

    """旋转位置编码 RoPE"""

    def __init__(self, dim, max_seq_len=2048, base=10000.0):

        super().__init__()

        # 计算频率

        inv_freq = 1.0 / (base ** (torch.arange(0, dim, 2).float() / dim))

        self.register_buffer('inv_freq', inv_freq)

 

        # 预计算位置旋转矩阵

        t = torch.arange(max_seq_len).float()

        freqs = torch.outer(t, inv_freq)  # (seq_len, dim/2)

        self.register_buffer('cos_cached', freqs.cos())  # (seq_len, dim/2)

        self.register_buffer('sin_cached', freqs.sin())  # (seq_len, dim/2)

 

    def forward(self, x, seq_len):

        return (

            self.cos_cached[:seq_len].unsqueeze(0).unsqueeze(0),  # (1, 1, seq, dim/2)

            self.sin_cached[:seq_len].unsqueeze(0).unsqueeze(0),

        )

 

 

def apply_rotary_emb(q, k, cos, sin):

    """将 RoPE 应用到 Q 和 K 上"""

    # 将 q, k 拆成两半

    q1, q2 = q[..., ::2], q[..., 1::2]

    k1, k2 = k[..., ::2], k[..., 1::2]

 

    # 旋转操作

    q_rot = torch.cat([q1 * cos - q2 * sin, q1 * sin + q2 * cos], dim=-1)

    k_rot = torch.cat([k1 * cos - k2 * sin, k1 * sin + k2 * cos], dim=-1)

    return q_rot, k_rot

 

 

class CausalSelfAttention(nn.Module):

    """因果自注意力（带 RoPE 和多头）"""

    def __init__(self, config):

        super().__init__()

        assert config.n_embd % config.n_head == 0

 

        self.n_head = config.n_head

        self.head_dim = config.n_embd // config.n_head

 

        # Q, K, V 投影（合并为一个线性层提高效率）

        self.qkv_proj = nn.Linear(config.n_embd, 3 * config.n_embd, bias=False)

        self.out_proj = nn.Linear(config.n_embd, config.n_embd, bias=False)

        self.attn_dropout = nn.Dropout(config.dropout)

        self.resid_dropout = nn.Dropout(config.dropout)

 

        # RoPE

        self.rope = RotaryPositionEmbedding(self.head_dim, config.block_size)

 

    def forward(self, x):

        B, T, C = x.size()  # batch, seq_len, embed_dim

 

        # 计算 Q, K, V

        qkv = self.qkv_proj(x)

        q, k, v = qkv.split(C, dim=-1)

 

        # 重塑为多头: (B, n_head, T, head_dim)

        q = q.view(B, T, self.n_head, self.head_dim).transpose(1, 2)

        k = k.view(B, T, self.n_head, self.head_dim).transpose(1, 2)

        v = v.view(B, T, self.n_head, self.head_dim).transpose(1, 2)

 

        # 应用 RoPE

        cos, sin = self.rope(q, T)

        q, k = apply_rotary_emb(q, k, cos, sin)

 

        # 计算注意力分数

        scale = 1.0 / math.sqrt(self.head_dim)

        attn = torch.matmul(q, k.transpose(-2, -1)) * scale

 

        # 因果遮罩：不能看到未来

        causal_mask = torch.triu(

            torch.ones(T, T, device=x.device, dtype=torch.bool), diagonal=1

        )

        attn = attn.masked_fill(causal_mask.unsqueeze(0).unsqueeze(0), float('-inf'))

 

        # Softmax + Dropout

        attn = F.softmax(attn, dim=-1)

        attn = self.attn_dropout(attn)

 

        # 加权求和

        out = torch.matmul(attn, v)

 

        # 合并多头

        out = out.transpose(1, 2).contiguous().view(B, T, C)

        out = self.resid_dropout(self.out_proj(out))

        return out

 

 

class FeedForward(nn.Module):

    """SwiGLU 前馈网络"""

    def __init__(self, config):

        super().__init__()

        hidden_dim = int(config.n_embd * 8 / 3)  # SwiGLU 推荐的隐藏维度

        # 取最近的 8 的倍数，方便 GPU 计算

        hidden_dim = ((hidden_dim + 7) // 8) * 8

 

        self.w1 = nn.Linear(config.n_embd, hidden_dim, bias=False)

        self.w2 = nn.Linear(hidden_dim, config.n_embd, bias=False)

        self.w3 = nn.Linear(config.n_embd, hidden_dim, bias=False)  # 门控

        self.dropout = nn.Dropout(config.dropout)

 

    def forward(self, x):

        # SwiGLU: (x @ W1) * SiLU(x @ W3) @ W2

        return self.dropout(self.w2(F.silu(self.w3(x)) * self.w1(x)))

 

 

class TransformerBlock(nn.Module):

    """单个 Transformer 解码器块（Pre-Norm 风格）"""

    def __init__(self, config):

        super().__init__()

        self.ln1 = RMSNorm(config.n_embd)

        self.attn = CausalSelfAttention(config)

        self.ln2 = RMSNorm(config.n_embd)

        self.ffn = FeedForward(config)

 

    def forward(self, x):

        # Pre-Norm + 残差连接

        x = x + self.attn(self.ln1(x))

        x = x + self.ffn(self.ln2(x))

        return x

 

 

class MiniTransformer(nn.Module):

    """完整的 Mini Transformer 语言模型"""

    def __init__(self, config):

        super().__init__()

        self.config = config

        self.token_emb = nn.Embedding(config.vocab_size, config.n_embd)

        self.drop = nn.Dropout(config.dropout)

        self.blocks = nn.ModuleList([

            TransformerBlock(config) for _ in range(config.n_layer)

        ])

        self.ln_f = RMSNorm(config.n_embd)

        self.lm_head = nn.Linear(config.n_embd, config.vocab_size, bias=False)

 

        # 权重共享：输出层与嵌入层共享权重

        self.token_emb.weight = self.lm_head.weight

 

        # 初始化权重

        self.apply(self._init_weights)

        print(f"模型参数量: {sum(p.numel() for p in self.parameters()) / 1e6:.2f}M")

 

    def _init_weights(self, module):

        if isinstance(module, nn.Linear):

            torch.nn.init.normal_(module.weight, mean=0.0, std=0.02)

            if module.bias is not None:

                torch.nn.init.zeros_(module.bias)

        elif isinstance(module, nn.Embedding):

            torch.nn.init.normal_(module.weight, mean=0.0, std=0.02)

 

    def forward(self, idx, targets=None):

        B, T = idx.size()

        assert T <= self.config.block_size, f"序列长度 {T} 超过上下文窗口 {self.config.block_size}"

 

        # Token 嵌入（RoPE 在注意力层内部应用，无需额外加位置编码）

        x = self.drop(self.token_emb(idx))

 

        # N 层 Transformer Block

        for block in self.blocks:

            x = block(x)

 

        # 最终归一化 + 语言模型头

        x = self.ln_f(x)

        logits = self.lm_head(x)

 

        # 计算损失

        loss = None

        if targets is not None:

            loss = F.cross_entropy(logits.view(-1, logits.size(-1)), targets.view(-1))

        return logits, loss

 

    @torch.no_grad()

    def generate(self, idx, max_new_tokens, temperature=0.8, top_k=40):

        """自回归文本生成"""

        for _ in range(max_new_tokens):

            # 截断到上下文窗口

            idx_cond = idx[:, -self.config.block_size:]

            logits, _ = self(idx_cond)

            logits = logits[:, -1, :] / temperature

 

            # Top-K 采样

            if top_k is not None:

                v, _ = torch.topk(logits, min(top_k, logits.size(-1)))

                logits[logits < v[:, [-1]]] = float('-inf')

 

            probs = F.softmax(logits, dim=-1)

            idx_next = torch.multinomial(probs, num_samples=1)

            idx = torch.cat([idx, idx_next], dim=1)

        return idx

 

 

# ============================================================

# 数据加载与训练

# ============================================================

 

def load_data(filepath=None):

    """加载训练数据（默认使用内置的示例文本）"""

    if filepath and __import__('os').path.exists(filepath):

        with open(filepath, 'r', encoding='utf-8') as f:

            text = f.read()

    else:

        # 内置示例文本

        text = """To be, or not to be, that is the question:

Whether 'tis nobler in the mind to suffer

The slings and arrows of outrageous fortune,

Or to take arms against a sea of troubles,

And by opposing end them. To die, to sleep,

No more; and by a sleep to say we end

The heart-ache and the thousand natural shocks

That flesh is heir to: 'tis a consummation

Devoutly to be wish'd. To die, to sleep;

To sleep, perchance to dream - ay, there's the rub:

For in that sleep of death what dreams may come.""" * 100  # 重复以增加数据量

 

    # 构建字符级词表

    chars = sorted(list(set(text)))

    char_to_idx = {ch: i for i, ch in enumerate(chars)}

    idx_to_char = {i: ch for i, ch in enumerate(chars)}

 

    # 编码文本

    data = torch.tensor([char_to_idx[c] for c in text], dtype=torch.long)

 

    # 划分训练集和验证集

    n = int(0.9 * len(data))

    train_data = data[:n]

    val_data = data[n:]

 

    return train_data, val_data, char_to_idx, idx_to_char, len(chars)

 

 

def get_batch(data, config):

    """生成一个训练批次"""

    ix = torch.randint(len(data) - config.block_size, (config.batch_size,))

    x = torch.stack([data[i:i + config.block_size] for i in ix])

    y = torch.stack([data[i + 1:i + config.block_size + 1] for i in ix])

    return x.to(config.device), y.to(config.device)

 

 

@torch.no_grad()

def estimate_loss(model, train_data, val_data, config, eval_iters=200):

    """评估训练集和验证集的损失"""

    model.eval()

    losses = {}

    for split, data in [('train', train_data), ('val', val_data)]:

        batch_losses = []

        for _ in range(eval_iters):

            X, Y = get_batch(data, config)

            _, loss = model(X, Y)

            batch_losses.append(loss.item())

        losses[split] = sum(batch_losses) / len(batch_losses)

    model.train()

    return losses

 

 

def train():

    """主训练循环"""

    config = Config()

 

    # 加载数据

    train_data, val_data, char_to_idx, idx_to_char, vocab_size = load_data()

    config.vocab_size = vocab_size

    print(f"词表大小: {vocab_size}, 训练数据: {len(train_data)} 字符")

 

    # 创建模型

    model = MiniTransformer(config).to(config.device)

 

    # 优化器

    optimizer = torch.optim.AdamW(model.parameters(), lr=config.learning_rate)

 

    # 训练循环

    for step in range(config.max_iters):

        # 定期评估

        if step % config.eval_interval == 0:

            losses = estimate_loss(model, train_data, val_data, config)

            print(f"Step {step}: train loss = {losses['train']:.4f}, val loss = {losses['val']:.4f}")

 

        # 前向传播

        X, Y = get_batch(train_data, config)

        logits, loss = model(X, Y)

 

        # 反向传播

        optimizer.zero_grad(set_to_none=True)

        loss.backward()

        # 梯度裁剪

        torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)

        optimizer.step()

 

    # 生成示例

    print("\n" + "=" * 50)

    print("生成示例:")

    print("=" * 50)

    context = torch.zeros((1, 1), dtype=torch.long, device=config.device)

    generated = model.generate(context, max_new_tokens=300)

    print(''.join([idx_to_char[i.item()] for i in generated[0]]))

 

 

if __name__ == '__main__':

    train()
