# 第11章/python

"""LangGraph 文章生成工作流"""

from typing import TypedDict, Literal

from langgraph.graph import StateGraph, START, END

from langchain_openai import ChatOpenAI

from langchain_core.prompts import ChatPromptTemplate

 

# ===== 1. 定义状态 =====

class ArticleState(TypedDict):

    topic: str            # 文章主题

    outline: str          # 大纲

    draft: str            # 草稿

    review: str           # 审核意见

    is_approved: bool     # 是否通过

    revision_count: int   # 修改次数

 

llm = ChatOpenAI(model="gpt-4o-mini")

 

# ===== 2. 定义节点 =====

def generate_outline(state: ArticleState) -> dict:

    """生成文章大纲"""

    prompt = ChatPromptTemplate.from_template(

        "为以下主题生成文章大纲（3-5个要点）:\n{topic}"

    )

    chain = prompt | llm

    result = chain.invoke({"topic": state["topic"]})

    return {"outline": result.content}

 

def write_draft(state: ArticleState) -> dict:

    """撰写文章"""

    revision_hint = f"\n修改意见：{state['review']}" if state.get("review") else ""

    prompt = ChatPromptTemplate.from_template(

        "根据以下大纲撰写文章:\n{outline}{revision_hint}"

    )

    chain = prompt | llm

    result = chain.invoke({

        "outline": state["outline"],

        "revision_hint": revision_hint,

    })

    return {"draft": result.content}

 

def review_draft(state: ArticleState) -> dict:

    """审核文章"""

    prompt = ChatPromptTemplate.from_template(

        """审核以下文章，判断质量是否合格。

文章：{draft}

 

如果合格，回复"APPROVED"。

如果需要修改，给出具体修改意见。"""

    )

    chain = prompt | llm

    result = chain.invoke({"draft": state["draft"]})

    is_approved = "APPROVED" in result.content.upper()

    return {

        "review": result.content,

        "is_approved": is_approved,

        "revision_count": state.get("revision_count", 0) + 1,

    }

 

# ===== 3. 路由函数 =====

def route_review(state: ArticleState) -> Literal["publish", "write_draft"]:

    if state["is_approved"] or state["revision_count"] >= 3:

        return "publish"

    return "write_draft"

 

def publish(state: ArticleState) -> dict:

    print(f"文章已发布！修改次数: {state['revision_count']}")

    return state

 

# ===== 4. 构建图 =====

graph = StateGraph(ArticleState)

 

graph.add_node("generate_outline", generate_outline)

graph.add_node("write_draft", write_draft)

graph.add_node("review_draft", review_draft)

graph.add_node("publish", publish)

 

graph.add_edge(START, "generate_outline")

graph.add_edge("generate_outline", "write_draft")

graph.add_edge("write_draft", "review_draft")

graph.add_conditional_edges("review_draft", route_review)

graph.add_edge("publish", END)

 

# 编译并运行

app = graph.compile()

result = app.invoke({"topic": "人工智能在医疗领域的应用", "revision_count": 0})

# 第11章/python

"""

多步骤文档分析助手

基于 LangChain + LangGraph 构建

"""

 

from typing import TypedDict, List, Optional, Literal

from langgraph.graph import StateGraph, START, END

from langchain_openai import ChatOpenAI

from langchain_core.prompts import ChatPromptTemplate

from langchain_core.output_parsers import JsonOutputParser, StrOutputParser

from pydantic import BaseModel, Field

import json

 

 

# ===== 状态定义 =====

class AnalysisState(TypedDict):

    file_path: str

    raw_text: str

    doc_type: str

    structure: dict

    key_info: dict

    summary: str

    qa_history: list

    current_question: Optional[str]

 

 

# ===== 输出结构 =====

class DocStructure(BaseModel):

    doc_type: str = Field(description="文档类型：合同/报告/论文/其他")

    sections: List[str] = Field(description="文档章节列表")

    total_words: int = Field(description="总字数")

    language: str = Field(description="主要语言")

 

 

class KeyInfo(BaseModel):

    entities: List[str] = Field(description="关键实体：人名/机构/日期等")

    key_points: List[str] = Field(description="核心要点，3-5条")

    action_items: List[str] = Field(description="待办事项（如有）")

 

 

# ===== 初始化 =====

llm = ChatOpenAI(model="gpt-4o-mini", temperature=0)

llm_creative = ChatOpenAI(model="gpt-4o-mini", temperature=0.7)

 

 

# ===== 节点函数 =====

def parse_document(state: AnalysisState) -> dict:

    """步骤1: 解析文档"""

    from pathlib import Path

 

    file_path = state["file_path"]

    suffix = Path(file_path).suffix.lower()

 

    if suffix == ".pdf":

        import pdfplumber

        with pdfplumber.open(file_path) as pdf:

            text = "\n".join(p.extract_text() or "" for p in pdf.pages)

    elif suffix == ".docx":

        from docx import Document

        doc = Document(file_path)

        text = "\n".join(p.text for p in doc.paragraphs)

    elif suffix in [".txt", ".md"]:

        with open(file_path, "r", encoding="utf-8") as f:

            text = f.read()

    else:

        raise ValueError(f"不支持的格式: {suffix}")

 

    print(f"[解析完成] 提取文本 {len(text)} 字符")

    return {"raw_text": text}

 

 

def analyze_structure(state: AnalysisState) -> dict:

    """步骤2: 分析文档结构"""

    parser = JsonOutputParser(pydantic_object=DocStructure)

    prompt = ChatPromptTemplate.from_messages([

        ("system", "你是文档分析专家。分析以下文档的结构。{format_instructions}"),

        ("human", "文档内容（前3000字）:\n{text}"),

    ])

 

    chain = prompt | llm | parser

    # 只取前3000字用于结构分析

    text_preview = state["raw_text"][:3000]

    result = chain.invoke({

        "text": text_preview,

        "format_instructions": parser.get_format_instructions(),

    })

 

    print(f"[结构分析] 文档类型: {result['doc_type']}")

    return {"doc_type": result["doc_type"], "structure": result}

 

 

def extract_key_info(state: AnalysisState) -> dict:

    """步骤3: 提取关键信息"""

    parser = JsonOutputParser(pydantic_object=KeyInfo)

    prompt = ChatPromptTemplate.from_messages([

        ("system", "你是信息提取专家。从{doc_type}类文档中提取关键信息。{format_instructions}"),

        ("human", "文档内容:\n{text}"),

    ])

 

    chain = prompt | llm | parser

    # 取前 8000 字用于信息提取

    result = chain.invoke({

        "text": state["raw_text"][:8000],

        "doc_type": state["doc_type"],

        "format_instructions": parser.get_format_instructions(),

    })

 

    print(f"[信息提取] 发现 {len(result['key_points'])} 个要点")

    return {"key_info": result}

 

 

def generate_summary(state: AnalysisState) -> dict:

    """步骤4: 生成综合摘要"""

    prompt = ChatPromptTemplate.from_messages([

        ("system", "你是文档摘要专家。请基于分析结果生成结构化摘要。"),

        ("human", """

文档类型: {doc_type}

文档结构: {structure}

关键信息: {key_info}

 

请生成一份不超过500字的结构化摘要，包括：

1. 文档概述

2. 核心内容

3. 关键结论或建议

"""),

    ])

 

    chain = prompt | llm_creative | StrOutputParser()

    summary = chain.invoke({

        "doc_type": state["doc_type"],

        "structure": json.dumps(state["structure"], ensure_ascii=False),

        "key_info": json.dumps(state["key_info"], ensure_ascii=False),

    })

 

    print("[摘要生成] 完成")

    return {"summary": summary}

 

 

def answer_question(state: AnalysisState) -> dict:

    """步骤5: 回答用户问题"""

    prompt = ChatPromptTemplate.from_messages([

        ("system", """你是文档问答助手。基于以下文档内容回答问题。

文档摘要: {summary}

关键信息: {key_info}"""),

        ("human", "{question}"),

    ])

 

    chain = prompt | llm | StrOutputParser()

    answer = chain.invoke({

        "summary": state["summary"],

        "key_info": json.dumps(state["key_info"], ensure_ascii=False),

        "question": state["current_question"],

    })

 

    history = state.get("qa_history", [])

    history.append({"q": state["current_question"], "a": answer})

 

    return {"qa_history": history}

 

 

# ===== 构建工作流图 =====

def build_analysis_graph():

    graph = StateGraph(AnalysisState)

 

    graph.add_node("parse", parse_document)

    graph.add_node("analyze", analyze_structure)

    graph.add_node("extract", extract_key_info)

    graph.add_node("summarize", generate_summary)

    graph.add_node("qa", answer_question)

 

    graph.add_edge(START, "parse")

    graph.add_edge("parse", "analyze")

    graph.add_edge("analyze", "extract")

    graph.add_edge("extract", "summarize")

    graph.add_edge("summarize", END)

 

    return graph.compile()

 

 

# ===== 使用示例 =====

if __name__ == "__main__":

    # 分析文档

    app = build_analysis_graph()

    result = app.invoke({

        "file_path": "./sample_contract.pdf",

        "qa_history": [],

    })

 

    print("\n" + "=" * 50)

    print("文档分析报告")

    print("=" * 50)

    print(f"\n文档类型: {result['doc_type']}")

    print(f"\n关键要点:")

    for i, point in enumerate(result['key_info']['key_points'], 1):

        print(f"  {i}. {point}")

    print(f"\n摘要:\n{result['summary']}")
