"""
A/B测试框架
支持流量分割、指标收集和统计分析
"""
import hashlib
import time
from typing import Dict, Any, List, Optional
from dataclasses import dataclass, field
from collections import defaultdict
import math
 
 
@dataclass
class ExperimentConfig:
    """实验配置"""
    name: str
    description: str
    traffic_split: float = 0.5  # B组流量比例
    variants: Dict[str, Dict] = field(default_factory=dict)
 
 
@dataclass
class MetricRecord:
    """指标记录"""
    user_id: str
    variant: str  # "A" or "B"
    latency_ms: float
    satisfaction: Optional[int] = None  # 1-5
    resolved: Optional[bool] = None
    escalated: bool = False
    timestamp: float = 0.0
 
    def __post_init__(self):
        if self.timestamp == 0.0:
            self.timestamp = time.time()
 
 
class ABTestFramework:
    """A/B测试引擎"""
 
    def __init__(self):
        self.experiments: Dict[str, ExperimentConfig] = {}
        self.metrics: Dict[str, List[MetricRecord]] = defaultdict(list)
 
    def create_experiment(self, config: ExperimentConfig):
        """创建实验"""
        self.experiments[config.name] = config
 
    def assign_variant(self, experiment_name: str, user_id: str) -> str:
        """为用户分配实验组"""
        config = self.experiments.get(experiment_name)
        if not config:
            return "A"
 
        # 基于user_id的确定性Hash分流
        hash_val = int(
            hashlib.md5(
                f"{experiment_name}:{user_id}".encode()
            ).hexdigest(),
            16,
        )
        bucket = hash_val % 100
 
        return "B" if bucket < config.traffic_split * 100 else "A"
 
    def record_metric(self, experiment_name: str, record: MetricRecord):
        """记录指标"""
        self.metrics[experiment_name].append(record)
 
    def analyze(self, experiment_name: str) -> Dict[str, Any]:
        """统计分析实验结果"""
        records = self.metrics.get(experiment_name, [])
        if not records:
            return {"error": "No data"}
 
        group_a = [r for r in records if r.variant == "A"]
        group_b = [r for r in records if r.variant == "B"]
 
        result = {
            "sample_size_a": len(group_a),
            "sample_size_b": len(group_b),
        }
 
        # 延迟对比
        if group_a and group_b:
            lat_a = [r.latency_ms for r in group_a]
            lat_b = [r.latency_ms for r in group_b]
            result["avg_latency_a"] = sum(lat_a) / len(lat_a)
            result["avg_latency_b"] = sum(lat_b) / len(lat_b)
 
            # 简易t检验
            result["latency_t_stat"], result["latency_p_value"] = (
                self._welch_t_test(lat_a, lat_b)
            )
 
        # 解决率对比
        resolved_a = [r for r in group_a if r.resolved is not None]
        resolved_b = [r for r in group_b if r.resolved is not None]
        if resolved_a:
            result["resolve_rate_a"] = (
                sum(1 for r in resolved_a if r.resolved) / len(resolved_a)
            )
        if resolved_b:
            result["resolve_rate_b"] = (
                sum(1 for r in resolved_b if r.resolved) / len(resolved_b)
            )
 
        # 人工升级率
        if group_a:
            result["escalation_rate_a"] = (
                sum(1 for r in group_a if r.escalated) / len(group_a)
            )
        if group_b:
            result["escalation_rate_b"] = (
                sum(1 for r in group_b if r.escalated) / len(group_b)
            )
 
        return result
 
    @staticmethod
    def _welch_t_test(
        sample_a: List[float], sample_b: List[float]
    ) -> tuple:
        """Welch's t-test（不假设等方差）"""
        n_a, n_b = len(sample_a), len(sample_b)
        if n_a < 2 or n_b < 2:
            return 0.0, 1.0
 
        mean_a = sum(sample_a) / n_a
        mean_b = sum(sample_b) / n_b
        var_a = sum((x - mean_a) ** 2 for x in sample_a) / (n_a - 1)
        var_b = sum((x - mean_b) ** 2 for x in sample_b) / (n_b - 1)
 
        se = math.sqrt(var_a / n_a + var_b / n_b)
        if se == 0:
            return 0.0, 1.0
 
        t_stat = (mean_a - mean_b) / se
 
        # 简化的p值估算（双尾检验）
        df = min(n_a, n_b) - 1
        # 使用近似公式
        p_value = 2.0 * math.exp(
            -0.717 * abs(t_stat) - 0.416 * t_stat**2
        )
        p_value = min(max(p_value, 0.0), 1.0)
 
        return t_stat, p_value

# 第17章/python
# src/gateway/app.py
"""
智能客服系统 - API网关
提供HTTP和WebSocket接口
"""
from fastapi import FastAPI, WebSocket, WebSocketDisconnect, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from typing import Optional, Dict, List
import time
import uuid
 
from prometheus_client import (
    Counter,
    Histogram,
    generate_latest,
    CONTENT_TYPE_LATEST,
)
from starlette.responses import Response
 
 
# ========== Prometheus指标定义 ==========
REQUEST_COUNT = Counter(
    "cs_request_total", "Total requests", ["endpoint", "status"]
)
REQUEST_LATENCY = Histogram(
    "cs_request_latency_seconds",
    "Request latency",
    ["endpoint"],
    buckets=[0.1, 0.5, 1.0, 2.0, 5.0],
)
INTENT_COUNT = Counter(
    "cs_intent_total", "Intent distribution", ["intent"]
)
ESCALATION_COUNT = Counter(
    "cs_escalation_total", "Escalation to human"
)
 
 
# ========== 数据模型 ==========
class ChatRequest(BaseModel):
    session_id: Optional[str] = None
    user_id: str
    message: str
 
 
class ChatResponse(BaseModel):
    session_id: str
    reply: str
    intent: Optional[str] = None
    confidence: Optional[float] = None
    escalated: bool = False
 
 
# ========== FastAPI应用 ==========
app = FastAPI(title="智能客服系统", version="1.0.0")
 
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)
 
# 会话存储（生产环境用Redis）
sessions: Dict[str, dict] = {}
 
 
@app.post("/api/chat", response_model=ChatResponse)
async def chat(request: ChatRequest):
    """同步聊天接口"""
    start_time = time.time()
 
    try:
        session_id = request.session_id or str(uuid.uuid4())
 
        # 获取或创建会话上下文
        if session_id not in sessions:
            sessions[session_id] = {
                "user_id": request.user_id,
                "history": [],
                "state": "greeting",
            }
 
        ctx = sessions[session_id]
 
        # TODO: 这里集成DialogManager处理逻辑
        # reply, updated_ctx = dialog_manager.process(request.message, ctx)
        reply = f"[模拟回复] 收到您的消息：{request.message}"
        intent = "order.track"
        confidence = 0.92
 
        # 记录指标
        INTENT_COUNT.labels(intent=intent).inc()
        REQUEST_COUNT.labels(endpoint="/api/chat", status="200").inc()
 
        latency = time.time() - start_time
        REQUEST_LATENCY.labels(endpoint="/api/chat").observe(latency)
 
        return ChatResponse(
            session_id=session_id,
            reply=reply,
            intent=intent,
            confidence=confidence,
            escalated=False,
        )
 
    except Exception as e:
        REQUEST_COUNT.labels(endpoint="/api/chat", status="500").inc()
        raise HTTPException(status_code=500, detail=str(e))
 
 
@app.websocket("/ws/chat/{session_id}")
async def websocket_chat(websocket: WebSocket, session_id: str):
    """WebSocket实时聊天接口"""
    await websocket.accept()
 
    try:
        while True:
            data = await websocket.receive_json()
            message = data.get("message", "")
 
            # TODO: 集成DialogManager
            reply = f"[WS回复] {message}"
 
            await websocket.send_json({
                "session_id": session_id,
                "reply": reply,
                "timestamp": time.time(),
            })
 
    except WebSocketDisconnect:
        # 清理会话
        sessions.pop(session_id, None)
 
 
@app.get("/metrics")
async def metrics():
    """Prometheus指标端点"""
    return Response(
        content=generate_latest(), media_type=CONTENT_TYPE_LATEST
    )
 
 
@app.get("/health")
async def health():
    """健康检查端点"""
    return {"status": "healthy", "timestamp": time.time()}

17.7.3 关键经验总结

1. 先做好知识库，再做花哨的功能——RAG的效果80%取决于知识库质量

2. 规则和AI互补——简单场景用规则快速响应，复杂场景用LLM

3. 永远留一条人工通路——AI客服的目标不是100%替代人工，而是让人工只处理真正有价值的问题

4. 监控先行——没有指标就没有优化方向

5. 渐进式上线——从10%流量开始，逐步放量，不要一步到位

本章小结：我们从零搭建了一套完整的智能客服系统，涵盖了知识库构建、RAG管道、意图识别、多轮对话、人机协作、A/B测试和运维监控。这不是一个"demo项目"，而是一个可以真正上线的生产级系统架构。在下一章中，我们将把视角从"应用层"下沉到"模型层"，动手训练一个行业垂直大模型。
