10.2.3 代码示例：多格式解析

# 第10章/python
"""统一文档解析器"""
from pathlib import Path
 
class DocumentParser:
    """支持多种格式的文档解析器"""
 
    def parse(self, file_path: str) -> str:
        """根据文件类型自动选择解析方法"""
        suffix = Path(file_path).suffix.lower()
        parser_map = {
            ".pdf": self._parse_pdf,
            ".docx": self._parse_docx,
            ".html": self._parse_html,
            ".md": self._parse_markdown,
            ".txt": self._parse_text,
        }
        parser = parser_map.get(suffix)
        if not parser:
            raise ValueError(f"不支持的文件格式: {suffix}")
        return parser(file_path)
 
    def _parse_pdf(self, file_path: str) -> str:
        """使用 pdfplumber 解析 PDF"""
        import pdfplumber
        texts = []
        with pdfplumber.open(file_path) as pdf:
            for page in pdf.pages:
                text = page.extract_text()
                if text:
                    texts.append(text)
                # 提取表格并转为文本
                for table in page.extract_tables():
                    header = table[0]
                    for row in table[1:]:
                        row_text = " | ".join(
                            f"{h}: {v}" for h, v in zip(header, row) if v
                        )
                        texts.append(row_text)
        return "\n\n".join(texts)
 
    def _parse_docx(self, file_path: str) -> str:
        """解析 Word 文档"""
        from docx import Document
        doc = Document(file_path)
        texts = [para.text for para in doc.paragraphs if para.text.strip()]
        return "\n\n".join(texts)
 
    def _parse_html(self, file_path: str) -> str:
        """解析 HTML 网页"""
        from bs4 import BeautifulSoup
        with open(file_path, "r", encoding="utf-8") as f:
            soup = BeautifulSoup(f.read(), "html.parser")
        # 移除脚本和样式
        for tag in soup(["script", "style", "nav", "footer"]):
            tag.decompose()
        return soup.get_text(separator="\n", strip=True)
 
    def _parse_markdown(self, file_path: str) -> str:
        """Markdown 直接读取"""
        with open(file_path, "r", encoding="utf-8") as f:
            return f.read()
 
    def _parse_text(self, file_path: str) -> str:
        """纯文本直接读取"""
        with open(file_path, "r", encoding="utf-8") as f:
            return f.read()

10.3 文本切分策略：固定长度、语义切分、递归切分

10.3.1 为什么需要切分

大模型的上下文窗口有限（通常 4K-128K tokens），而一篇文档可能有数万字。我们需要将文档切分成小块（Chunk），每块既要包含足够的上下文，又不能太长以至于稀释关键信息。

切分是 RAG 中最容易被忽视却最影响效果的环节。

10.3.5 代码实现：递归文本切分器

# 第10章/python
"""递归文本切分器"""
from typing import List
 
class RecursiveTextSplitter:
    """递归文本切分器：优先在自然边界切分"""
 
    def __init__(
        self,
        chunk_size: int = 500,
        chunk_overlap: int = 100,
        separators: List[str] = None,
    ):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.separators = separators or ["\n\n", "\n", "。", "！", "？", ".", " "]
 
    def split(self, text: str) -> List[str]:
        """递归切分文本"""
        return self._split_recursive(text, self.separators)
 
    def _split_recursive(self, text: str, separators: List[str]) -> List[str]:
        """尝试用当前分隔符切分，如果单块太大则用下一级分隔符"""
        if len(text) <= self.chunk_size:
            return [text.strip()] if text.strip() else []
 
        # 找到可用的分隔符
        separator = separators[0] if separators else ""
        remaining_separators = separators[1:] if len(separators) > 1 else []
 
        # 按分隔符切分
        if separator:
            parts = text.split(separator)
        else:
            # 最后手段：按字符数强制切分
            return self._force_split(text)
 
        # 合并小块，确保每块不超过 chunk_size
        chunks = []
        current_chunk = ""
 
        for part in parts:
            candidate = current_chunk + separator + part if current_chunk else part
            if len(candidate) <= self.chunk_size:
                current_chunk = candidate
            else:
                if current_chunk:
                    chunks.append(current_chunk.strip())
                if len(part) > self.chunk_size:
                    # 单个部分太大，递归用更细的分隔符
                    sub_chunks = self._split_recursive(part, remaining_separators)
                    chunks.extend(sub_chunks)
                    current_chunk = ""
                else:
                    current_chunk = part
 
        if current_chunk.strip():
            chunks.append(current_chunk.strip())
 
        # 添加重叠
        return self._add_overlap(chunks)
 
    def _force_split(self, text: str) -> List[str]:
        """按字符数强制切分"""
        chunks = []
        for i in range(0, len(text), self.chunk_size - self.chunk_overlap):
            chunk = text[i:i + self.chunk_size]
            if chunk.strip():
                chunks.append(chunk.strip())
        return chunks
 
    def _add_overlap(self, chunks: List[str]) -> List[str]:
        """在块之间添加重叠"""
        if len(chunks) <= 1 or self.chunk_overlap <= 0:
            return chunks
 
        overlapped = [chunks[0]]
        for i in range(1, len(chunks)):
            prev_tail = chunks[i - 1][-self.chunk_overlap:]
            overlapped.append(prev_tail + chunks[i])
        return overlapped

10.4 向量嵌入 Embedding 模型选型与使用

10.4.1 什么是向量嵌入

向量嵌入（Embedding）是将文本转换为高维数值向量的过程。语义相近的文本，其向量在空间中的距离也相近。

10.4.2 向量空间可视化

想象所有文本都被映射到一个高维空间中（降维后可以可视化为二维平面）：

  -------+-------------------------> 维度1 (主题)

关键洞察：好的 Embedding 模型能够将语义相近的文本映射到空间中相近的位置，让"检索"变成"找最近的邻居"。

10.4.4 代码示例：使用 Embedding 模型

# 第10章/python
"""向量嵌入模型使用示例"""
from sentence_transformers import SentenceTransformer
import numpy as np
 
# 加载模型（首次下载约 1.2GB）
model = SentenceTransformer("BAAI/bge-large-zh-v1.5")
 
# 文本转向量
texts = [
    "如何申请退款？",
    "我想退货，怎么操作？",
    "Python 列表的用法",
    "退款要多久到账？",
]
 
embeddings = model.encode(texts, normalize_embeddings=True)
print(f"向量维度: {embeddings.shape}")  # (4, 1024)
 
# 计算相似度矩阵
similarity = np.dot(embeddings, embeddings.T)
print("\n相似度矩阵:")
for i, t1 in enumerate(texts):
    for j, t2 in enumerate(texts):
        if j > i:
            print(f"  '{t1}' <-> '{t2}': {similarity[i][j]:.3f}")
