9.6.2 BLEU 指标详解

BLEU（Bilingual Evaluation Understudy）通过计算生成文本与参考文本之间的 n-gram 精确率来评估质量。

# 第9章/text
BLEU 计算示例：
 
参考文本：今天 天气 真 不错 适合 出去 散步
生成文本：今天 天气 很好 适合 出去 走走
 
1-gram 匹配：今天、天气、适合、出去 → 4/6 = 0.667
2-gram 匹配：今天天气、适合出去 → 2/5 = 0.4
3-gram 匹配：无完全匹配 → 0/4 = 0
4-gram 匹配：无完全匹配 → 0/3 = 0
 
BLEU-1 = 0.667
BLEU-2 = sqrt(0.667 × 0.4) = 0.516

9.6.4 人工评测的实施方法

自动指标有其局限性——BLEU/ROUGE 高不代表实际效果好。人工评测虽然成本高，但更能反映真实用户体验。

图9-18

推荐做法：

1. 准备 100-200 条测试用例

2. 找 3 位以上评估员独立评分

3. 计算评估员间一致性（Kappa 系数）

4. 使用 ELO 评分系统进行模型间对比

9.7.3 完整训练代码

以下是完整可运行的微调脚本（详见 code/ch09/qlora_finetune.py）：

# 第9章/python
"""
QLoRA 微调 Qwen2.5-7B 电商客服模型
要求：至少 12GB 显存（推荐 RTX 4090）
"""
 
import json
import torch
from datasets import Dataset
from transformers import (
    AutoTokenizer,
    AutoModelForCausalLM,
    BitsAndBytesConfig,
    TrainingArguments,
)
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training
from trl import SFTTrainer
 
# ============ 1. 配置 ============
MODEL_NAME = "Qwen/Qwen2.5-7B-Instruct"
OUTPUT_DIR = "./output/qwen-ecommerce-cs"
DATA_PATH = "./data/ecommerce_cs_train.json"
 
# ============ 2. 量化配置 ============
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,                     # 4-bit 量化
    bnb_4bit_quant_type="nf4",             # NormalFloat4 量化类型
    bnb_4bit_compute_dtype=torch.bfloat16, # 计算时用 BF16
    bnb_4bit_use_double_quant=True,        # 双重量化
)
 
# ============ 3. 加载模型和分词器 ============
print("正在加载模型...")
tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME, trust_remote_code=True)
tokenizer.pad_token = tokenizer.eos_token
 
model = AutoModelForCausalLM.from_pretrained(
    MODEL_NAME,
    quantization_config=bnb_config,
    device_map="auto",
    trust_remote_code=True,
)
model = prepare_model_for_kbit_training(model)
 
# ============ 4. LoRA 配置 ============
lora_config = LoraConfig(
    r=16,                          # 秩，平衡效果和效率
    lora_alpha=32,                 # 缩放因子，通常 = 2 * r
    lora_dropout=0.05,             # Dropout 防过拟合
    bias="none",                   # 不训练偏置
    task_type="CAUSAL_LM",
    target_modules=[               # 目标模块
        "q_proj", "k_proj", "v_proj", "o_proj",
        "gate_proj", "up_proj", "down_proj",
    ],
)
 
model = get_peft_model(model, lora_config)
model.print_trainable_parameters()
# 输出示例：trainable params: 20,971,520 || all params: 7,636,013,056 || trainable%: 0.2747
 
# ============ 5. 加载训练数据 ============
def load_data(data_path: str) -> Dataset:
    """加载 ShareGPT 格式的训练数据"""
    with open(data_path, "r", encoding="utf-8") as f:
        raw_data = json.load(f)
 
    formatted = []
    for item in raw_data:
        messages = []
        for turn in item["conversations"]:
            role_map = {"system": "system", "human": "user", "gpt": "assistant"}
            messages.append({
                "role": role_map[turn["from"]],
                "content": turn["value"],
            })
        text = tokenizer.apply_chat_template(
            messages, tokenize=False, add_generation_prompt=False
        )
        formatted.append({"text": text})
 
    return Dataset.from_list(formatted)
 
dataset = load_data(DATA_PATH)
print(f"训练数据量：{len(dataset)} 条")
 
# ============ 6. 训练参数 ============
training_args = TrainingArguments(
    output_dir=OUTPUT_DIR,
    num_train_epochs=3,
    per_device_train_batch_size=2,
    gradient_accumulation_steps=8,      # 有效批量 = 2 * 8 = 16
    learning_rate=2e-4,
    lr_scheduler_type="cosine",
    warmup_ratio=0.05,
    weight_decay=0.01,
    logging_steps=10,
    save_steps=100,
    save_total_limit=3,
    bf16=True,                          # 使用 BF16 训练
    gradient_checkpointing=True,        # 节省显存
    max_grad_norm=1.0,                  # 梯度裁剪
    report_to="wandb",                  # 训练监控
    run_name="qwen-ecommerce-cs",
)
 
# ============ 7. 开始训练 ============
trainer = SFTTrainer(
    model=model,
    args=training_args,
    train_dataset=dataset,
    tokenizer=tokenizer,
    max_seq_length=2048,
    dataset_text_field="text",
)
 
print("开始训练...")
trainer.train()
 
# ============ 8. 保存模型 ============
trainer.save_model(OUTPUT_DIR)
tokenizer.save_pretrained(OUTPUT_DIR)
print(f"模型已保存至：{OUTPUT_DIR}")

9.7.5 模型推理测试

# 第9章/python
"""微调后模型的推理测试"""
from transformers import AutoTokenizer, AutoModelForCausalLM
from peft import PeftModel
import torch
 
# 加载基础模型和 LoRA 适配器
base_model = AutoModelForCausalLM.from_pretrained(
    "Qwen/Qwen2.5-7B-Instruct",
    torch_dtype=torch.bfloat16,
    device_map="auto",
)
model = PeftModel.from_pretrained(base_model, "./output/qwen-ecommerce-cs")
tokenizer = AutoTokenizer.from_pretrained("./output/qwen-ecommerce-cs")
 
# 构造测试对话
messages = [
    {"role": "system", "content": "你是【星辰商城】的智能客服小星。"},
    {"role": "user", "content": "我想退货，买了两天的耳机不想要了"},
]
 
text = tokenizer.apply_chat_template(messages, tokenize=False, add_generation_prompt=True)
inputs = tokenizer(text, return_tensors="pt").to(model.device)
 
outputs = model.generate(
    **inputs,
    max_new_tokens=256,
    temperature=0.7,
    top_p=0.9,
    do_sample=True,
)
response = tokenizer.decode(outputs[0][inputs["input_ids"].shape[-1]:], skip_special_tokens=True)
print(f"客服回复：{response}")

9.7.6 模型合并与导出

# 第9章/python
"""将 LoRA 适配器合并回基础模型"""
from transformers import AutoModelForCausalLM, AutoTokenizer
from peft import PeftModel
import torch
 
# 加载基础模型（全精度）
base_model = AutoModelForCausalLM.from_pretrained(
    "Qwen/Qwen2.5-7B-Instruct",
    torch_dtype=torch.bfloat16,
    device_map="cpu",  # CPU 上合并，避免显存问题
)
 
# 加载并合并 LoRA
model = PeftModel.from_pretrained(base_model, "./output/qwen-ecommerce-cs")
merged_model = model.merge_and_unload()
 
# 保存合并后的模型
merged_model.save_pretrained("./output/qwen-ecommerce-cs-merged")
tokenizer = AutoTokenizer.from_pretrained("./output/qwen-ecommerce-cs")
tokenizer.save_pretrained("./output/qwen-ecommerce-cs-merged")
 
print("合并完成！可以直接部署 ./output/qwen-ecommerce-cs-merged")
