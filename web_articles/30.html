第16章 大模型评测与安全

核心隐喻：大模型的"质检流水线"与"安全防护网"
如果说训练大模型是"生产汽车"，那么评测就是出厂前的"质检" -- 确保这辆车跑得够快、转弯够灵、刹车够稳。而安全则是"碰撞测试和安全气囊" -- 在极端情况下保护驾驶员和路人。没有经过严格质检和安全测试的汽车，绝不能上路。同样，没有经过评测与安全审计的大模型，绝不能上线。

16.2 构建领域专属评测数据集

16.2.1 为什么通用基准不够用

通用评测基准就像高考 -- 它能评估学生的综合素质，但无法评估一个医生是否胜任临床诊断、一个律师是否精通合同法。当大模型应用于特定领域时，需要构建领域专属评测集。

import json
from dataclasses import dataclass, asdict
from enum import Enum
 
class Difficulty(Enum):
    EASY = "easy"
    MEDIUM = "medium"
    HARD = "hard"
 
class QuestionType(Enum):
    MULTIPLE_CHOICE = "multiple_choice"  # 选择题
    SHORT_ANSWER = "short_answer"        # 简答题
    OPEN_ENDED = "open_ended"            # 开放题
    CODE = "code"                        # 代码题
 
@dataclass
class EvalQuestion:
    """评测题目模板"""
    id: str                          # 唯一标识
    domain: str                      # 领域（如"medical", "legal"）
    subdomain: str                   # 子领域（如"cardiology", "contract"）
    question_type: str               # 题型
    difficulty: str                  # 难度
    question: str                    # 题目内容
    choices: list[str] | None        # 选项（选择题专用）
    reference_answer: str            # 参考答案
    scoring_criteria: str            # 评分标准
    metadata: dict                   # 附加信息
 
# 医疗领域评测数据集示例

# 保存为标准格式
def save_dataset(questions: list, filepath: str):
    data = {
        "version": "1.0",
        "domain": questions[0].domain if questions else "",
        "total_questions": len(questions),
        "questions": [asdict(q) for q in questions]
    }
    with open(filepath, "w", encoding="utf-8") as f:
        json.dump(data, f, ensure_ascii=False, indent=2)
 
save_dataset(medical_eval_dataset, "medical_eval_v1.json")

# eval_pipeline.py - 自动化评测流水线
import json
import asyncio
import time
from openai import AsyncOpenAI
from dataclasses import dataclass
 
@dataclass
class EvalResult:
    question_id: str
    question: str
    model_answer: str
    reference_answer: str
    score: float
    max_score: float
    details: str
 
class EvalPipeline:
    """自动化评测流水线"""
 
    def __init__(self, model_client: AsyncOpenAI, model_name: str):
        self.client = model_client
        self.model_name = model_name
        self.results: list[EvalResult] = []
 
    async def load_dataset(self, filepath: str) -> list[dict]:
        """加载评测数据集"""
        with open(filepath, "r", encoding="utf-8") as f:
            data = json.load(f)
        print(f"加载 {data['total_questions']} 道题目 (领域: {data['domain']})")
        return data["questions"]
 
    async def run_inference(self, question: dict) -> str:
        """对单个问题进行推理"""
        prompt = question["question"]
        if question.get("choices"):
            prompt += "\n" + "\n".join(question["choices"])
            prompt += "\n请直接回答选项字母。"
 
        response = await self.client.chat.completions.create(
            model=self.model_name,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.1,  # 评测时用低温度保证确定性
            max_tokens=1024
        )
        return response.choices[0].message.content
 
    def score_multiple_choice(self, model_answer: str, reference: str) -> tuple[float, str]:
        """选择题评分"""
        # 提取模型回答中的选项字母
        import re
        match = re.search(r'[A-D]', model_answer.upper())
        predicted = match.group(0) if match else ""
        correct = predicted == reference.upper()
        return (1.0 if correct else 0.0, f"预测:{predicted} 正确:{reference}")
 
    async def score_open_ended(self, question: str, model_answer: str,
                                reference: str, criteria: str) -> tuple[float, str]:
        """开放题LLM评分"""
        judge_prompt = f"""请作为评分专家，对以下回答进行评分。
 
问题：{question}
 
参考答案：{reference}
 
待评分回答：{model_answer}
 
评分标准：{criteria}
 
请严格按照评分标准打分，返回JSON格式：
{{"score": 分数, "max_score": 满分, "reason": "评分理由"}}"""
 
        response = await self.client.chat.completions.create(
            model="gpt-4o",  # 用强模型做评判
            messages=[{"role": "user", "content": judge_prompt}],
            response_format={"type": "json_object"},
            temperature=0.1
        )
 
        result = json.loads(response.choices[0].message.content)
        return (result["score"], result["reason"])
 
    async def evaluate(self, dataset_path: str) -> dict:
        """执行完整评测"""
        questions = await self.load_dataset(dataset_path)
 
        print(f"开始评测模型: {self.model_name}")
        start_time = time.time()
 
        for i, q in enumerate(questions):
            print(f"  [{i+1}/{len(questions)}] {q['id']}", end=" ")
 
            # 推理
            model_answer = await self.run_inference(q)
 
            # 评分
            if q["question_type"] == "multiple_choice":
                score, detail = self.score_multiple_choice(
                    model_answer, q["reference_answer"]
                )
                max_score = 1.0
            elif q["question_type"] == "open_ended":
                score, detail = await self.score_open_ended(
                    q["question"], model_answer,
                    q["reference_answer"], q["scoring_criteria"]
                )
                max_score = q.get("metadata", {}).get("max_score", 10)
            else:
                score, detail, max_score = 0, "未知题型", 1
 
            self.results.append(EvalResult(
                question_id=q["id"],
                question=q["question"][:50] + "...",
                model_answer=model_answer[:100],
                reference_answer=q["reference_answer"][:100],
                score=score,
                max_score=max_score,
                details=detail
            ))
            print(f"得分: {score}/{max_score}")
 
        elapsed = time.time() - start_time
        return self.generate_report(elapsed)
 
    def generate_report(self, elapsed_time: float) -> dict:
        """生成评测报告"""
        total_score = sum(r.score for r in self.results)
        total_max = sum(r.max_score for r in self.results)
 
        # 按子领域统计
        domain_scores = {}
        for r in self.results:
            domain = r.question_id.split("-")[1]
            if domain not in domain_scores:
                domain_scores[domain] = {"score": 0, "max": 0, "count": 0}
            domain_scores[domain]["score"] += r.score
            domain_scores[domain]["max"] += r.max_score
            domain_scores[domain]["count"] += 1
 
        report = {
            "model": self.model_name,
            "timestamp": time.strftime("%Y-%m-%d %H:%M:%S"),
            "total_questions": len(self.results),
            "total_score": total_score,
            "total_max_score": total_max,
            "accuracy": total_score / total_max if total_max > 0 else 0,
            "elapsed_seconds": elapsed_time,
            "domain_breakdown": {
                domain: {
                    "accuracy": d["score"] / d["max"],
                    "questions": d["count"]
                }
                for domain, d in domain_scores.items()
            }
        }
 
        print(f"\n====== 评测报告 ======")
        print(f"模型: {report['model']}")
        print(f"总分: {total_score:.1f}/{total_max:.1f} ({report['accuracy']:.1%})")
        print(f"用时: {elapsed_time:.1f}秒")
        for domain, stats in report["domain_breakdown"].items():
            print(f"  {domain}: {stats['accuracy']:.1%} ({stats['questions']}题)")
 
        return report
 
# 使用示例
async def main():
    client = AsyncOpenAI(api_key="your-key", base_url="https://api.example.com/v1")
    pipeline = EvalPipeline(client, model_name="qwen-plus")
    report = await pipeline.evaluate("medical_eval_v1.json")
 
    # 保存报告
    with open("eval_report.json", "w", encoding="utf-8") as f:
        json.dump(report, f, ensure_ascii=False, indent=2)
 
if __name__ == "__main__":
    asyncio.run(main())
