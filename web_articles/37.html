# src/training/sft_lora.py
"""
领域SFT微调（LoRA方案）
使用PEFT库进行参数高效微调
"""
import os
import json
import logging
from typing import Dict, List
from dataclasses import dataclass
 
import torch
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    TrainingArguments,
)
from peft import LoraConfig, get_peft_model, TaskType
from trl import SFTTrainer, SFTConfig
from datasets import Dataset
 
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)
 
 
@dataclass
class SFTData:
    """SFT数据项"""
    instruction: str
    input: str
    output: str
 
 
def load_sft_data(file_path: str) -> Dataset:
    """加载SFT格式数据"""
    with open(file_path, "r", encoding="utf-8") as f:
        raw_data = json.load(f)
 
    # 转换为ChatML格式
    formatted = []
    for item in raw_data:
        instruction = item.get("instruction", "")
        inp = item.get("input", "")
        output = item.get("output", "")
 
        if inp:
            user_content = f"{instruction}\n\n{inp}"
        else:
            user_content = instruction
 
        text = (
            f"<|im_start|>system\n"
            f"你是一位专业的医学助手，请根据你的医学知识准确回答问题。"
            f"<|im_end|>\n"
            f"<|im_start|>user\n{user_content}<|im_end|>\n"
            f"<|im_start|>assistant\n{output}<|im_end|>"
        )
        formatted.append({"text": text})
 
    return Dataset.from_list(formatted)
 
 
def train_lora(
    base_model_path: str,
    data_path: str,
    output_dir: str,
    lora_rank: int = 64,
    lora_alpha: int = 128,
    learning_rate: float = 2e-4,
    num_epochs: int = 3,
    batch_size: int = 4,
    max_seq_length: int = 2048,
):
    """执行LoRA微调"""
    # 1. 加载模型和分词器
    logger.info(f"Loading model: {base_model_path}")
    tokenizer = AutoTokenizer.from_pretrained(
        base_model_path, trust_remote_code=True
    )
    if tokenizer.pad_token is None:
        tokenizer.pad_token = tokenizer.eos_token
 
    model = AutoModelForCausalLM.from_pretrained(
        base_model_path,
        torch_dtype=torch.bfloat16,
        device_map="auto",
        trust_remote_code=True,
    )
 
    # 2. 配置LoRA
    lora_config = LoraConfig(
        task_type=TaskType.CAUSAL_LM,
        r=lora_rank,
        lora_alpha=lora_alpha,
        lora_dropout=0.05,
        target_modules=[
            "q_proj", "k_proj", "v_proj", "o_proj",
            "gate_proj", "up_proj", "down_proj",
        ],
        bias="none",
    )
 
    model = get_peft_model(model, lora_config)
    trainable_params = sum(
        p.numel() for p in model.parameters() if p.requires_grad
    )
    total_params = sum(p.numel() for p in model.parameters())
    logger.info(
        f"Trainable: {trainable_params / 1e6:.1f}M / "
        f"Total: {total_params / 1e9:.2f}B "
        f"({trainable_params / total_params:.2%})"
    )
 
    # 3. 加载数据
    dataset = load_sft_data(data_path)
    logger.info(f"Dataset size: {len(dataset)}")
 
    # 4. 训练配置
    sft_config = SFTConfig(
        output_dir=output_dir,
        num_train_epochs=num_epochs,
        per_device_train_batch_size=batch_size,
        gradient_accumulation_steps=4,
        learning_rate=learning_rate,
        lr_scheduler_type="cosine",
        warmup_ratio=0.1,
        bf16=True,
        logging_steps=10,
        save_steps=200,
        eval_strategy="no",
        save_total_limit=3,
        max_seq_length=max_seq_length,
        dataset_text_field="text",
        report_to="tensorboard",
    )
 
    # 5. 创建Trainer
    trainer = SFTTrainer(
        model=model,
        args=sft_config,
        train_dataset=dataset,
        tokenizer=tokenizer,
    )
 
    # 6. 开始训练
    logger.info("Starting LoRA SFT training...")
    result = trainer.train()
 
    # 7. 保存LoRA权重
    model.save_pretrained(os.path.join(output_dir, "lora_weights"))
    tokenizer.save_pretrained(os.path.join(output_dir, "lora_weights"))
 
    logger.info(f"Training done. Metrics: {result.metrics}")
    return result
 
 
if __name__ == "__main__":
    train_lora(
        base_model_path="output/cpt_medical_qwen7b/final",
        data_path="data/medical_sft.json",
        output_dir="output/sft_medical_lora",
        lora_rank=64,
        num_epochs=3,
    )

# src/evaluation/domain_eval.py
"""
领域模型评估框架
支持多维度自动化评估
"""
import json
import time
from typing import Dict, List, Optional
from dataclasses import dataclass, field
 
 
@dataclass
class EvalQuestion:
    """评估题目"""
    question_id: str
    category: str  # 类别：knowledge/reasoning/format/safety
    question: str
    reference_answer: str
    choices: Optional[List[str]] = None  # 选择题选项
    answer_key: Optional[str] = None    # 选择题正确选项
 
 
@dataclass
class EvalResult:
    """评估结果"""
    question_id: str
    model_answer: str
    is_correct: bool
    latency_ms: float
    category: str
 
 
class DomainEvaluator:
    """领域模型评估器"""
 
    def __init__(self, model_client, judge_client=None):
        self.model_client = model_client
        self.judge_client = judge_client  # 用于主观评估的裁判模型
 
    def evaluate_dataset(
        self, eval_file: str
    ) -> Dict[str, float]:
        """评估整个数据集"""
        with open(eval_file, "r", encoding="utf-8") as f:
            questions = [EvalQuestion(**q) for q in json.load(f)]
 
        results = []
        for q in questions:
            result = self._evaluate_single(q)
            results.append(result)
 
        return self._compute_metrics(results)
 
    def _evaluate_single(self, question: EvalQuestion) -> EvalResult:
        """评估单个问题"""
        start = time.time()
        model_answer = self.model_client.generate(question.question)
        latency = (time.time() - start) * 1000
 
        # 判定正确性
        if question.choices:
            # 选择题：精确匹配
            is_correct = self._check_choice(
                model_answer, question.answer_key
            )
        elif self.judge_client:
            # 开放题：使用裁判模型评估
            is_correct = self._judge_answer(
                question.question,
                question.reference_answer,
                model_answer,
            )
        else:
            # 简单关键词匹配
            is_correct = self._keyword_match(
                model_answer, question.reference_answer
            )
 
        return EvalResult(
            question_id=question.question_id,
            model_answer=model_answer,
            is_correct=is_correct,
            latency_ms=latency,
            category=question.category,
        )
 
    @staticmethod
    def _check_choice(answer: str, correct_key: str) -> bool:
        """检查选择题答案"""
        answer_clean = answer.strip().upper()
        return correct_key.upper() in answer_clean[:5]
 
    @staticmethod
    def _keyword_match(answer: str, reference: str) -> bool:
        """关键词匹配（简化版）"""
        # 提取参考答案中的关键词
        import re
        keywords = re.findall(r"[\u4e00-\u9fff]{2,}", reference)
        if not keywords:
            return False
        hit = sum(1 for kw in keywords if kw in answer)
        return hit / len(keywords) > 0.5
 
    def _judge_answer(
        self, question: str, reference: str, answer: str
    ) -> bool:
        """使用裁判模型评估答案质量"""
        prompt = f"""请判断以下模型回答是否正确。
 
问题：{question}
参考答案：{reference}
模型回答：{answer}
 
请只回答"正确"或"错误"。"""
 
        result = self.judge_client.generate(prompt)
        return "正确" in result
 
    @staticmethod
    def _compute_metrics(results: List[EvalResult]) -> Dict[str, float]:
        """计算评估指标"""
        if not results:
            return {}
 
        # 总体准确率
        total_correct = sum(1 for r in results if r.is_correct)
        overall_acc = total_correct / len(results)
 
        # 分类别准确率
        category_stats = {}
        for r in results:
            if r.category not in category_stats:
                category_stats[r.category] = {"correct": 0, "total": 0}
            category_stats[r.category]["total"] += 1
            if r.is_correct:
                category_stats[r.category]["correct"] += 1
 
        metrics = {
            "overall_accuracy": overall_acc,
            "total_questions": len(results),
            "avg_latency_ms": sum(r.latency_ms for r in results) / len(results),
        }
 
        for cat, stats in category_stats.items():
            metrics[f"accuracy_{cat}"] = (
                stats["correct"] / stats["total"]
            )
 
        return metrics

# 第18章/python
# src/training/version_manager.py
"""
模型版本管理
追踪每次迭代的数据、配置和评估结果
"""
import json
import os
from datetime import datetime
from typing import Dict, List, Optional
from dataclasses import dataclass, field, asdict
 
 
@dataclass
class ModelVersion:
    """模型版本信息"""
    version: str
    base_model: str
    training_type: str  # "cpt" | "sft" | "dpo" | "incremental"
    data_stats: Dict = field(default_factory=dict)
    training_config: Dict = field(default_factory=dict)
    eval_metrics: Dict = field(default_factory=dict)
    created_at: str = ""
    model_path: str = ""
    status: str = "training"  # training | evaluating | deployed | retired
    notes: str = ""
 
    def __post_init__(self):
        if not self.created_at:
            self.created_at = datetime.now().isoformat()
 
 
class ModelVersionManager:
    """模型版本管理器"""
 
    def __init__(self, registry_path: str = "model_registry.json"):
        self.registry_path = registry_path
        self.versions: List[ModelVersion] = []
        self._load()
 
    def register(self, version: ModelVersion):
        """注册新版本"""
        self.versions.append(version)
        self._save()
 
    def get_deployed(self) -> Optional[ModelVersion]:
        """获取当前部署的版本"""
        for v in reversed(self.versions):
            if v.status == "deployed":
                return v
        return None
 
    def compare_versions(
        self, v1: str, v2: str
    ) -> Dict[str, Dict]:
        """对比两个版本的评估指标"""
        ver1 = self._find(v1)
        ver2 = self._find(v2)
        if not ver1 or not ver2:
            return {"error": "Version not found"}
 
        comparison = {}
        all_keys = set(ver1.eval_metrics.keys()) | set(
            ver2.eval_metrics.keys()
        )
        for key in all_keys:
            val1 = ver1.eval_metrics.get(key, "N/A")
            val2 = ver2.eval_metrics.get(key, "N/A")
            if isinstance(val1, (int, float)) and isinstance(
                val2, (int, float)
            ):
                delta = val2 - val1
                comparison[key] = {
                    v1: val1,
                    v2: val2,
                    "delta": delta,
                    "improved": delta > 0,
                }
            else:
                comparison[key] = {v1: val1, v2: val2}
        return comparison
 
    def _find(self, version_str: str) -> Optional[ModelVersion]:
        for v in self.versions:
            if v.version == version_str:
                return v
        return None
 
    def _save(self):
        with open(self.registry_path, "w", encoding="utf-8") as f:
            json.dump(
                [asdict(v) for v in self.versions],
                f,
                ensure_ascii=False,
                indent=2,
            )
 
    def _load(self):
        if os.path.exists(self.registry_path):
            with open(self.registry_path, "r", encoding="utf-8") as f:
                data = json.load(f)
                self.versions = [ModelVersion(**d) for d in data]
