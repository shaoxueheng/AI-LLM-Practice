# 第8章/python
"""
第8章实战: 多模型切换的智能助手
文件: code/ch08/multi_model_assistant.py
 
支持功能:
- 多模型切换 (OpenAI / Qwen / DeepSeek)
- 流式输出
- 多轮对话管理
- 工具调用 (天气/计算器)
"""
 
import os
import json
from openai import OpenAI
from typing import Optional, Dict, List, Generator
 
# ============================================================
# 模型配置
# ============================================================
 
MODEL_CONFIGS = {
    "gpt-4o": {
        "name": "GPT-4o",
        "api_key_env": "OPENAI_API_KEY",
        "base_url": "https://api.openai.com/v1",
        "model": "gpt-4o",
    },
    "gpt-4o-mini": {
        "name": "GPT-4o Mini",
        "api_key_env": "OPENAI_API_KEY",
        "base_url": "https://api.openai.com/v1",
        "model": "gpt-4o-mini",
    },
    "qwen-max": {
        "name": "通义千问 Max",
        "api_key_env": "DASHSCOPE_API_KEY",
        "base_url": "https://dashscope.aliyuncs.com/compatible-mode/v1",
        "model": "qwen-max",
    },
    "deepseek": {
        "name": "DeepSeek V3",
        "api_key_env": "DEEPSEEK_API_KEY",
        "base_url": "https://api.deepseek.com",
        "model": "deepseek-chat",
    },
}
 
# ============================================================
# 工具定义
# ============================================================
 
TOOLS = [
    {
        "type": "function",
        "function": {
            "name": "calculator",
            "description": "执行数学计算",
            "parameters": {
                "type": "object",
                "properties": {
                    "expression": {
                        "type": "string",
                        "description": "数学表达式，如 '2 + 3 * 4'"
                    }
                },
                "required": ["expression"]
            }
        }
    },
    {
        "type": "function",
        "function": {
            "name": "get_current_time",
            "description": "获取当前日期和时间",
            "parameters": {
                "type": "object",
                "properties": {}
            }
        }
    }
]
 
def calculator(expression: str) -> str:
    """安全的数学计算器"""
    try:
        # 只允许数学运算
        allowed_chars = set('0123456789+-*/().% ')
        if not all(c in allowed_chars for c in expression):
            return json.dumps({"error": "不支持的表达式"})
        result = eval(expression)
        return json.dumps({"expression": expression, "result": str(result)})
    except Exception as e:
        return json.dumps({"error": str(e)})
 
def get_current_time() -> str:
    """获取当前时间"""
    from datetime import datetime
    now = datetime.now()
    return json.dumps({
        "date": now.strftime("%Y-%m-%d"),
        "time": now.strftime("%H:%M:%S"),
        "weekday": ["周一","周二","周三","周四","周五","周六","周日"][now.weekday()]
    })
 
TOOL_FUNCTIONS = {
    "calculator": calculator,
    "get_current_time": get_current_time,
}
 
# ============================================================
# 核心助手类
# ============================================================
 
class MultiModelAssistant:
    """多模型切换的智能助手"""
 
    def __init__(self, default_model: str = "deepseek"):
        self.current_model = default_model
        self.clients: Dict[str, OpenAI] = {}
        self.system_prompt = (
            "你是一个智能助手，可以回答问题、进行数学计算、查询时间等。\n"
            "请用简洁友好的方式回答用户的问题。"
        )
        self.messages: List[Dict] = [
            {"role": "system", "content": self.system_prompt}
        ]
        self._init_clients()
 
    def _init_clients(self):
        """初始化所有可用的 API 客户端"""
        for key, config in MODEL_CONFIGS.items():
            api_key = os.environ.get(config["api_key_env"])
            if api_key:
                self.clients[key] = OpenAI(
                    api_key=api_key,
                    base_url=config["base_url"]
                )
                print(f"  [OK] {config['name']} 已就绪")
            else:
                print(f"  [--] {config['name']} 未配置 (缺少 {config['api_key_env']})")
 
    def switch_model(self, model_key: str) -> str:
        """切换模型"""
        if model_key not in MODEL_CONFIGS:
            return f"未知模型: {model_key}。可选: {', '.join(MODEL_CONFIGS.keys())}"
        if model_key not in self.clients:
            return f"{MODEL_CONFIGS[model_key]['name']} 未配置 API Key"
        self.current_model = model_key
        return f"已切换到 {MODEL_CONFIGS[model_key]['name']}"
 
    def chat(self, user_input: str) -> str:
        """发送消息并获取回复"""
        if self.current_model not in self.clients:
            return f"当前模型 {self.current_model} 不可用，请切换模型"
 
        self.messages.append({"role": "user", "content": user_input})
 
        client = self.clients[self.current_model]
        model = MODEL_CONFIGS[self.current_model]["model"]
 
        try:
            # 发送请求（带工具）
            response = client.chat.completions.create(
                model=model,
                messages=self.messages,
                tools=TOOLS,
                tool_choice="auto",
                stream=False
            )
 
            assistant_msg = response.choices[0].message
 
            # 处理工具调用
            if assistant_msg.tool_calls:
                self.messages.append(assistant_msg)
                for tool_call in assistant_msg.tool_calls:
                    func_name = tool_call.function.name
                    func_args = json.loads(tool_call.function.arguments)
                    result = TOOL_FUNCTIONS[func_name](**func_args)
                    self.messages.append({
                        "role": "tool",
                        "tool_call_id": tool_call.id,
                        "content": result
                    })
 
                # 获取最终回复
                response = client.chat.completions.create(
                    model=model,
                    messages=self.messages,
                    stream=False
                )
                assistant_msg = response.choices[0].message
 
            reply = assistant_msg.content or ""
            self.messages.append({"role": "assistant", "content": reply})
            return reply
 
        except Exception as e:
            error_msg = f"API 调用错误: {str(e)}"
            return error_msg
 
    def chat_stream(self, user_input: str) -> Generator[str, None, None]:
        """流式输出版本"""
        if self.current_model not in self.clients:
            yield f"当前模型 {self.current_model} 不可用"
            return
 
        self.messages.append({"role": "user", "content": user_input})
 
        client = self.clients[self.current_model]
        model = MODEL_CONFIGS[self.current_model]["model"]
 
        try:
            stream = client.chat.completions.create(
                model=model,
                messages=self.messages,
                stream=True
            )
 
            full_response = ""
            for chunk in stream:
                if chunk.choices[0].delta.content:
                    token = chunk.choices[0].delta.content
                    full_response += token
                    yield token
 
            self.messages.append({"role": "assistant", "content": full_response})
 
        except Exception as e:
            yield f"\n[错误] {str(e)}"
 
    def clear_history(self):
        """清除对话历史"""
        self.messages = [{"role": "system", "content": self.system_prompt}]
 
    def get_status(self) -> str:
        """获取当前状态"""
        model_name = MODEL_CONFIGS[self.current_model]["name"]
        n_messages = len(self.messages) - 1
        available = [MODEL_CONFIGS[k]["name"] for k in self.clients]
        return (
            f"当前模型: {model_name}\n"
            f"对话轮数: {n_messages // 2}\n"
            f"可用模型: {', '.join(available)}"
        )
 
 
# ============================================================
# 命令行界面
# ============================================================
 
def main():
    """命令行交互界面"""
    print("=" * 60)
    print("  多模型智能助手 v1.0")
    print("=" * 60)
    print("命令: /switch <模型> | /status | /clear | /stream | /quit")
    print("-" * 60)
 
    assistant = MultiModelAssistant(default_model="deepseek")
    use_stream = True
 
    while True:
        try:
            user_input = input("\n你: ").strip()
        except (KeyboardInterrupt, EOFError):
            print("\n再见!")
            break
 
        if not user_input:
            continue
 
        # 处理命令
        if user_input.startswith("/"):
            parts = user_input.split()
            cmd = parts[0].lower()
 
            if cmd == "/quit":
                print("再见!")
                break
            elif cmd == "/switch" and len(parts) > 1:
                print(assistant.switch_model(parts[1]))
            elif cmd == "/status":
                print(assistant.get_status())
            elif cmd == "/clear":
                assistant.clear_history()
                print("对话历史已清除")
            elif cmd == "/stream":
                use_stream = not use_stream
                print(f"流式输出: {'开启' if use_stream else '关闭'}")
            else:
                print("未知命令。可用: /switch /status /clear /stream /quit")
            continue
 
        # 正常对话
        model_name = MODEL_CONFIGS[assistant.current_model]["name"]
        print(f"\n{model_name}: ", end="", flush=True)
 
        if use_stream:
            for token in assistant.chat_stream(user_input):
                print(token, end="", flush=True)
            print()
        else:
            reply = assistant.chat(user_input)
            print(reply)
 
 
if __name__ == "__main__":
    main()

关键收获：

1. 大多数国产 API 兼容 OpenAI 格式，切换模型只需改 base_url

2. 流式输出大幅提升用户体验，是生产环境的必选项

3. 多轮对话需要手动维护消息历史，注意上下文窗口限制

4. Function Calling 是连接 LLM 与外部世界的桥梁

5. 合理使用异步和缓存可以显著降低 API 调用成本

1. 如果你的应用需要支持 100 个并发用户，如何设计 API 调用架构？

2. 多轮对话中，消息历史超过上下文窗口怎么处理最优？

3. Function Calling 和 RAG 各自适合什么场景？能否结合使用？
