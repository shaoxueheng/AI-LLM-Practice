# src/workflow/dag_engine.py
"""
DAG工作流引擎
支持任务依赖管理、并行执行、错误处理
"""
import asyncio
import time
import uuid
import logging
from enum import Enum
from typing import (
    Dict, List, Set, Optional, Callable, Any, Awaitable,
)
from dataclasses import dataclass, field
from collections import defaultdict
 
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)
 
 
class TaskStatus(Enum):
    PENDING = "pending"
    READY = "ready"
    RUNNING = "running"
    COMPLETED = "completed"
    FAILED = "failed"
    SKIPPED = "skipped"
    RETRYING = "retrying"
 
 
@dataclass
class TaskNode:
    """DAG中的任务节点"""
    task_id: str
    name: str
    agent_role: str
    handler: Optional[Callable[..., Awaitable[Any]]] = None
    dependencies: List[str] = field(default_factory=list)
    status: TaskStatus = TaskStatus.PENDING
    result: Any = None
    error: Optional[str] = None
    retry_count: int = 0
    max_retries: int = 2
    timeout_seconds: int = 300
    start_time: Optional[float] = None
    end_time: Optional[float] = None
    metadata: Dict = field(default_factory=dict)
 
 
@dataclass
class DAGWorkflow:
    """DAG工作流定义"""
    workflow_id: str = field(
        default_factory=lambda: str(uuid.uuid4())[:8]
    )
    name: str = ""
    tasks: Dict[str, TaskNode] = field(default_factory=dict)
    status: str = "created"
    created_at: float = field(default_factory=time.time)
 
 
class DAGEngine:
    """DAG工作流执行引擎"""
 
    def __init__(self, max_parallel: int = 5):
        self.max_parallel = max_parallel
        self.semaphore = asyncio.Semaphore(max_parallel)
        self.task_results: Dict[str, Any] = {}
 
    def build_workflow(
        self, name: str, task_definitions: List[Dict]
    ) -> DAGWorkflow:
        """从定义构建工作流"""
        workflow = DAGWorkflow(name=name)
 
        for td in task_definitions:
            node = TaskNode(
                task_id=td["task_id"],
                name=td["name"],
                agent_role=td["agent_role"],
                handler=td.get("handler"),
                dependencies=td.get("dependencies", []),
                max_retries=td.get("max_retries", 2),
                timeout_seconds=td.get("timeout_seconds", 300),
            )
            workflow.tasks[node.task_id] = node
 
        # 验证DAG无环
        if self._has_cycle(workflow):
            raise ValueError("工作流存在循环依赖！")
 
        return workflow
 
    async def execute(self, workflow: DAGWorkflow) -> Dict[str, Any]:
        """执行整个工作流"""
        workflow.status = "running"
        logger.info(
            f"Starting workflow: {workflow.name} [{workflow.workflow_id}]"
        )
 
        # 初始化所有任务状态
        for task in workflow.tasks.values():
            task.status = TaskStatus.PENDING
 
        # 持续调度直到所有任务完成
        while not self._all_done(workflow):
            # 找出可执行的任务（依赖已满足）
            ready_tasks = self._get_ready_tasks(workflow)
 
            if not ready_tasks and not self._any_running(workflow):
                # 没有可执行的任务也没有运行中的任务 = 死锁或全部失败
                logger.error("Workflow stuck: no ready or running tasks")
                break
 
            # 并行启动就绪任务
            if ready_tasks:
                await asyncio.gather(
                    *[
                        self._execute_task(task, workflow)
                        for task in ready_tasks
                    ]
                )
            else:
                # 等待运行中的任务完成
                await asyncio.sleep(0.1)
 
        # 汇总结果
        workflow.status = "completed"
        return {
            tid: {
                "status": task.status.value,
                "result": task.result,
                "duration": (
                    (task.end_time - task.start_time)
                    if task.end_time and task.start_time
                    else None
                ),
            }
            for tid, task in workflow.tasks.items()
        }
 
    async def _execute_task(
        self, task: TaskNode, workflow: DAGWorkflow
    ):
        """执行单个任务（带重试和超时）"""
        async with self.semaphore:
            task.status = TaskStatus.RUNNING
            task.start_time = time.time()
            logger.info(f"  [START] {task.name} (agent: {task.agent_role})")
 
            for attempt in range(task.max_retries + 1):
                try:
                    # 收集依赖任务的结果作为输入
                    dep_results = {
                        dep_id: workflow.tasks[dep_id].result
                        for dep_id in task.dependencies
                        if workflow.tasks[dep_id].status
                        == TaskStatus.COMPLETED
                    }
 
                    # 执行任务（带超时）
                    if task.handler:
                        result = await asyncio.wait_for(
                            task.handler(dep_results, task.metadata),
                            timeout=task.timeout_seconds,
                        )
                    else:
                        result = {"message": f"Mock result for {task.name}"}
 
                    task.result = result
                    task.status = TaskStatus.COMPLETED
                    task.end_time = time.time()
                    self.task_results[task.task_id] = result
                    duration = task.end_time - task.start_time
                    logger.info(
                        f"  [DONE] {task.name} ({duration:.1f}s)"
                    )
                    return
 
                except asyncio.TimeoutError:
                    task.error = f"Timeout after {task.timeout_seconds}s"
                    logger.warning(
                        f"  [TIMEOUT] {task.name} (attempt {attempt + 1})"
                    )
                except Exception as e:
                    task.error = str(e)
                    logger.warning(
                        f"  [ERROR] {task.name}: {e} "
                        f"(attempt {attempt + 1})"
                    )
 
                if attempt < task.max_retries:
                    task.status = TaskStatus.RETRYING
                    task.retry_count += 1
                    await asyncio.sleep(2 ** attempt)  # 指数退避
 
            # 所有重试都失败
            task.status = TaskStatus.FAILED
            task.end_time = time.time()
            logger.error(f"  [FAILED] {task.name}: {task.error}")
 
    def _get_ready_tasks(self, workflow: DAGWorkflow) -> List[TaskNode]:
        """获取所有依赖已满足且未执行的任务"""
        ready = []
        for task in workflow.tasks.values():
            if task.status != TaskStatus.PENDING:
                continue
            # 检查所有依赖是否已完成
            deps_met = all(
                workflow.tasks[dep_id].status == TaskStatus.COMPLETED
                for dep_id in task.dependencies
            )
            # 如果有依赖失败了，跳过此任务
            deps_failed = any(
                workflow.tasks[dep_id].status == TaskStatus.FAILED
                for dep_id in task.dependencies
            )
            if deps_failed:
                task.status = TaskStatus.SKIPPED
                task.error = "Skipped due to dependency failure"
                continue
            if deps_met:
                task.status = TaskStatus.READY
                ready.append(task)
        return ready
 
    def _all_done(self, workflow: DAGWorkflow) -> bool:
        """检查是否所有任务都已完成/失败/跳过"""
        terminal_states = {
            TaskStatus.COMPLETED,
            TaskStatus.FAILED,
            TaskStatus.SKIPPED,
        }
        return all(
            t.status in terminal_states
            for t in workflow.tasks.values()
        )
 
    def _any_running(self, workflow: DAGWorkflow) -> bool:
        """是否有正在运行的任务"""
        running_states = {TaskStatus.RUNNING, TaskStatus.RETRYING}
        return any(
            t.status in running_states
            for t in workflow.tasks.values()
        )
 
    @staticmethod
    def _has_cycle(workflow: DAGWorkflow) -> bool:
        """拓扑排序检测环"""
        in_degree = defaultdict(int)
        graph = defaultdict(list)
 
        for tid, task in workflow.tasks.items():
            for dep in task.dependencies:
                graph[dep].append(tid)
                in_degree[tid] += 1
 
        queue = [
            tid
            for tid in workflow.tasks
            if in_degree[tid] == 0
        ]
        visited = 0
 
        while queue:
            node = queue.pop(0)
            visited += 1
            for neighbor in graph[node]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
 
        return visited != len(workflow.tasks)
