语音处理模块 (audio_module.py)：

# 第13章/python
import whisper
import numpy as np
 
class AudioProcessor:
    """语音处理模块：ASR语音识别"""
 
    def __init__(self, model_size: str = "base"):
        self.model = whisper.load_model(model_size)
 
    def transcribe(self, audio_path: str, language: str = "zh") -> dict:
        """将语音转为文字"""
        result = self.model.transcribe(
            audio_path,
            language=language,
            word_timestamps=True
        )
        return {
            "text": result["text"],
            "segments": [
                {
                    "start": seg["start"],
                    "end": seg["end"],
                    "text": seg["text"]
                }
                for seg in result["segments"]
            ],
            "language": result.get("language", language)
        }

视觉理解模块 (vision_module.py)：

# 第13章/python
from transformers import pipeline
from PIL import Image
 
class VisionProcessor:
    """视觉处理模块：图像理解和描述"""
 
    def __init__(self, model_name: str = "Salesforce/blip2-opt-2.7b"):
        self.captioner = pipeline(
            "image-to-text",
            model=model_name,
            device_map="auto"
        )
 
    def describe_image(self, image_path: str) -> str:
        """生成图像描述"""
        image = Image.open(image_path).convert("RGB")
        result = self.captioner(image, max_new_tokens=200)
        return result[0]["generated_text"]
 
    def answer_question(self, image_path: str, question: str) -> str:
        """基于图像回答问题"""
        image = Image.open(image_path).convert("RGB")
        result = self.captioner(image, prompt=question, max_new_tokens=200)
        return result[0]["generated_text"]

综合分析模块 (analyzer.py)：

# 第13章/python
from openai import OpenAI
 
class MultimodalAnalyzer:
    """多模态内容综合分析"""
 
    def __init__(self, api_key: str, base_url: str = None):
        self.client = OpenAI(api_key=api_key, base_url=base_url)
 
    def analyze(self, contents: dict) -> dict:
        """综合分析多模态内容"""
        # 构建综合提示
        prompt_parts = ["请对以下多模态内容进行综合分析：\n"]
 
        if contents.get("text"):
            prompt_parts.append(f"【文本内容】\n{contents['text']}\n")
        if contents.get("image_description"):
            prompt_parts.append(f"【图片描述】\n{contents['image_description']}\n")
        if contents.get("audio_transcript"):
            prompt_parts.append(f"【语音转写】\n{contents['audio_transcript']}\n")
 
        prompt_parts.append("""
请返回JSON格式的分析结果，包含：
1. summary: 内容摘要（100字以内）
2. sentiment: 情感倾向（正面/中性/负面）及置信度
3. tags: 关键标签列表（3-5个）
4. key_points: 核心要点列表
""")
 
        response = self.client.chat.completions.create(
            model="qwen-plus",
            messages=[
                {"role": "system", "content": "你是一个多模态内容分析专家。"},
                {"role": "user", "content": "\n".join(prompt_parts)}
            ],
            response_format={"type": "json_object"}
        )
 
        import json
        return json.loads(response.choices[0].message.content)

13.6.4 Gradio 界面搭建

# 第13章/python
import gradio as gr
from audio_module import AudioProcessor
from vision_module import VisionProcessor
from analyzer import MultimodalAnalyzer
 
# 初始化模块
audio_proc = AudioProcessor(model_size="base")
vision_proc = VisionProcessor()
analyzer = MultimodalAnalyzer(api_key="your-api-key")
 
def process_multimodal(image, audio, text, question):
    """处理多模态输入"""
    contents = {}
 
    # 处理图片
    if image is not None:
        contents["image_description"] = vision_proc.describe_image(image)
 
    # 处理语音
    if audio is not None:
        result = audio_proc.transcribe(audio)
        contents["audio_transcript"] = result["text"]
 
    # 处理文本
    if text:
        contents["text"] = text
 
    # 综合分析
    analysis = analyzer.analyze(contents)
 
    return (
        analysis.get("summary", ""),
        analysis.get("sentiment", ""),
        ", ".join(analysis.get("tags", [])),
        "\n".join(f"- {p}" for p in analysis.get("key_points", []))
    )
 
# 构建界面
with gr.Blocks(title="多模态内容分析平台", theme=gr.themes.Soft()) as demo:
    gr.Markdown("# 多模态内容分析平台")
    gr.Markdown("上传图片、语音或输入文字，AI将进行综合分析")
 
    with gr.Row():
        with gr.Column():
            image_input = gr.Image(type="filepath", label="上传图片")
            audio_input = gr.Audio(type="filepath", label="上传语音")
            text_input = gr.Textbox(label="输入文字", lines=3)
            question_input = gr.Textbox(label="提问（可选）", lines=2)
            submit_btn = gr.Button("开始分析", variant="primary")
 
        with gr.Column():
            summary_output = gr.Textbox(label="内容摘要")
            sentiment_output = gr.Textbox(label="情感分析")
            tags_output = gr.Textbox(label="关键标签")
            points_output = gr.Textbox(label="核心要点")
 
    submit_btn.click(
        fn=process_multimodal,
        inputs=[image_input, audio_input, text_input, question_input],
        outputs=[summary_output, sentiment_output, tags_output, points_output]
    )
 
if __name__ == "__main__":
    demo.launch(server_name="0.0.0.0", server_port=7860)
