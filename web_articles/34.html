# src/intent/classifier.py
"""
意图识别模块
双层架构：规则快速路由 + LLM兜底分类
"""
import re
from typing import Tuple, Dict, List, Optional
from dataclasses import dataclass
 
 
@dataclass
class IntentResult:
    """意图识别结果"""
    intent: str
    confidence: float
    slots: Dict[str, str]
    method: str  # "rule" | "llm"
 
 
class IntentClassifier:
    """双层意图识别器"""
 
    # 规则层：关键词到意图的映射
    KEYWORD_RULES = {
        "order.track": {
            "keywords": ["物流", "快递", "到哪了", "发货了吗", "运单"],
            "patterns": [r"订单.*在哪", r"什么时候.*到"],
        },
        "order.refund": {
            "keywords": ["退款", "退货", "退钱", "不想要了"],
            "patterns": [r"怎么退", r"申请.*退"],
        },
        "order.modify": {
            "keywords": ["改地址", "修改订单", "换个地址"],
            "patterns": [r"修改.*地址", r"更改.*收货"],
        },
        "product.price": {
            "keywords": ["多少钱", "价格", "优惠", "打折"],
            "patterns": [r".*多少钱", r"有.*优惠"],
        },
        "product.spec": {
            "keywords": ["参数", "规格", "尺寸", "材质", "功能"],
            "patterns": [r"什么材质", r"多大尺寸"],
        },
        "account.password": {
            "keywords": ["密码", "忘记密码", "登不上", "找回密码"],
            "patterns": [r"密码.*忘了", r"登录不了"],
        },
        "complaint.quality": {
            "keywords": ["质量问题", "坏了", "破损", "假货"],
            "patterns": [r"收到.*坏", r"质量.*差"],
        },
    }
 
    # 槽位提取规则
    SLOT_PATTERNS = {
        "order_id": r"(?:订单号?|单号)[：:\s]*([A-Za-z0-9]{10,20})",
        "phone": r"(?:手机号?|电话)[：:\s]*(1[3-9]\d{9})",
        "product_name": r"(?:商品|产品)[：:\s]*([^\s,，。]+)",
    }
 
    # LLM Few-shot模板
    LLM_PROMPT = """你是一个意图分类器。请将用户的输入分类到以下意图之一：
 
可选意图列表：
- order.track: 查询物流/快递状态
- order.refund: 申请退款/退货
- order.modify: 修改订单信息
- product.price: 咨询价格/优惠
- product.spec: 咨询商品规格参数
- account.password: 账户密码问题
- complaint.quality: 商品质量投诉
- chitchat: 闲聊/无关问题
- unknown: 无法判断
 
示例：
用户：我的快递到哪了 -> order.track
用户：这个东西怎么退 -> order.refund
用户：有什么优惠活动吗 -> product.price
 
用户：{query}
请只输出意图标签，不要解释。"""
 
    def __init__(self, llm_client=None):
        self.llm_client = llm_client
 
    def classify(self, query: str) -> IntentResult:
        """执行意图识别"""
        # 第一层：规则匹配
        rule_result = self._rule_match(query)
        if rule_result and rule_result.confidence >= 0.9:
            # 补充槽位提取
            rule_result.slots = self._extract_slots(query)
            return rule_result
 
        # 第二层：LLM分类
        llm_result = self._llm_classify(query)
        llm_result.slots = self._extract_slots(query)
        return llm_result
 
    def _rule_match(self, query: str) -> Optional[IntentResult]:
        """基于规则的意图匹配"""
        best_intent = None
        best_score = 0.0
 
        for intent, rules in self.KEYWORD_RULES.items():
            score = 0.0
            # 关键词匹配
            keyword_hits = sum(
                1 for kw in rules["keywords"] if kw in query
            )
            if keyword_hits > 0:
                score = min(0.7 + keyword_hits * 0.1, 1.0)
 
            # 正则模式匹配（加权更高）
            pattern_hits = sum(
                1
                for pat in rules.get("patterns", [])
                if re.search(pat, query)
            )
            if pattern_hits > 0:
                score = max(score, min(0.85 + pattern_hits * 0.05, 1.0))
 
            if score > best_score:
                best_score = score
                best_intent = intent
 
        if best_intent:
            return IntentResult(
                intent=best_intent,
                confidence=best_score,
                slots={},
                method="rule",
            )
        return None
 
    def _llm_classify(self, query: str) -> IntentResult:
        """基于LLM的意图分类"""
        if not self.llm_client:
            return IntentResult(
                intent="unknown", confidence=0.0, slots={}, method="llm"
            )
 
        prompt = self.LLM_PROMPT.format(query=query)
        response = self.llm_client.chat(
            [{"role": "user", "content": prompt}],
            temperature=0.0,
            max_tokens=20,
        )
        predicted = response["choices"][0]["message"]["content"].strip()
 
        # 验证是否为合法意图
        valid_intents = set(self.KEYWORD_RULES.keys()) | {
            "chitchat",
            "unknown",
        }
        if predicted in valid_intents:
            return IntentResult(
                intent=predicted, confidence=0.8, slots={}, method="llm"
            )
        return IntentResult(
            intent="unknown", confidence=0.3, slots={}, method="llm"
        )
 
    def _extract_slots(self, query: str) -> Dict[str, str]:
        """槽位提取"""
        slots = {}
        for slot_name, pattern in self.SLOT_PATTERNS.items():
            match = re.search(pattern, query)
            if match:
                slots[slot_name] = match.group(1)
        return slots

# 第17章/python
# src/dialog/manager.py
"""
对话管理模块
基于有限状态机的多轮对话管理
"""
from enum import Enum
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field
from datetime import datetime
import uuid
 
 
class DialogState(Enum):
    """对话状态枚举"""
    GREETING = "greeting"
    INTENT_RECOGNITION = "intent_recognition"
    SLOT_FILLING = "slot_filling"
    KNOWLEDGE_QA = "knowledge_qa"
    ACTION_EXECUTION = "action_execution"
    ANSWER_DELIVERY = "answer_delivery"
    CLARIFICATION = "clarification"
    SATISFACTION = "satisfaction"
    HUMAN_ESCALATION = "human_escalation"
    ERROR_HANDLING = "error_handling"
    CLOSING = "closing"
 
 
@dataclass
class ConversationContext:
    """会话上下文"""
    session_id: str = field(default_factory=lambda: str(uuid.uuid4()))
    user_id: str = ""
    state: DialogState = DialogState.GREETING
    current_intent: Optional[str] = None
    slots: Dict[str, str] = field(default_factory=dict)
    required_slots: List[str] = field(default_factory=list)
    history: List[Dict[str, str]] = field(default_factory=list)
    retry_count: int = 0
    slot_ask_count: int = 0
    created_at: datetime = field(default_factory=datetime.now)
    metadata: Dict[str, Any] = field(default_factory=dict)
 
 
class DialogManager:
    """对话管理器——有限状态机核心引擎"""
 
    # 意图 -> 所需槽位 的映射
    INTENT_SLOTS = {
        "order.track": ["order_id"],
        "order.refund": ["order_id"],
        "order.modify": ["order_id"],
        "account.password": ["phone"],
    }
 
    # 槽位 -> 追问话术
    SLOT_PROMPTS = {
        "order_id": "请提供您的订单号，订单号可以在"我的订单"中查看。",
        "phone": "请提供您的注册手机号，用于验证身份。",
        "product_name": "请告诉我您咨询的是哪个商品？",
    }
 
    MAX_SLOT_ASK = 3  # 最多追问3次
    MAX_RETRY = 2     # 最多重试2次
 
    def __init__(
        self,
        intent_classifier,
        rag_service,
        action_executor,
        ticket_service,
    ):
        self.intent_classifier = intent_classifier
        self.rag_service = rag_service
        self.action_executor = action_executor
        self.ticket_service = ticket_service
 
    def process(
        self, user_input: str, context: ConversationContext
    ) -> tuple:
        """
        处理用户输入，返回(回复文本, 更新后的上下文)
        这是状态机的核心驱动方法
        """
        # 记录用户输入
        context.history.append({"role": "user", "content": user_input})
 
        # 根据当前状态分发处理
        handler = {
            DialogState.GREETING: self._handle_greeting,
            DialogState.INTENT_RECOGNITION: self._handle_intent,
            DialogState.SLOT_FILLING: self._handle_slot_filling,
            DialogState.KNOWLEDGE_QA: self._handle_knowledge_qa,
            DialogState.ACTION_EXECUTION: self._handle_action,
            DialogState.CLARIFICATION: self._handle_clarification,
            DialogState.SATISFACTION: self._handle_satisfaction,
            DialogState.ERROR_HANDLING: self._handle_error,
        }.get(context.state, self._handle_intent)
 
        reply = handler(user_input, context)
 
        # 记录回复
        context.history.append({"role": "assistant", "content": reply})
        return reply, context
 
    def _handle_greeting(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """处理首次问候"""
        context.state = DialogState.INTENT_RECOGNITION
        return self._handle_intent(user_input, context)
 
    def _handle_intent(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """意图识别状态"""
        result = self.intent_classifier.classify(user_input)
        context.current_intent = result.intent
        context.slots.update(result.slots)
 
        if result.confidence < 0.5:
            context.state = DialogState.CLARIFICATION
            return "抱歉，我不太理解您的意思。能否换个方式描述一下您的问题？"
 
        # 检查是否需要槽位
        required = self.INTENT_SLOTS.get(result.intent, [])
        context.required_slots = required
        missing = [s for s in required if s not in context.slots]
 
        if missing:
            context.state = DialogState.SLOT_FILLING
            return self.SLOT_PROMPTS.get(
                missing[0], "请提供更多信息。"
            )
 
        # 知识类意图走RAG
        if result.intent.startswith("product.") or result.intent in (
            "chitchat",
        ):
            context.state = DialogState.KNOWLEDGE_QA
            return self._handle_knowledge_qa(user_input, context)
 
        # 操作类意图执行动作
        context.state = DialogState.ACTION_EXECUTION
        return self._handle_action(user_input, context)
 
    def _handle_slot_filling(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """槽位填充状态"""
        context.slot_ask_count += 1
 
        # 尝试从用户回复中提取槽位
        new_slots = self.intent_classifier._extract_slots(user_input)
        context.slots.update(new_slots)
 
        # 检查是否还缺槽位
        missing = [
            s for s in context.required_slots if s not in context.slots
        ]
 
        if not missing:
            context.state = DialogState.ACTION_EXECUTION
            context.slot_ask_count = 0
            return self._handle_action(user_input, context)
 
        if context.slot_ask_count >= self.MAX_SLOT_ASK:
            context.state = DialogState.HUMAN_ESCALATION
            return self._escalate_to_human(context)
 
        return self.SLOT_PROMPTS.get(missing[0], "请提供更多信息。")
 
    def _handle_knowledge_qa(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """知识问答状态"""
        answer = self.rag_service.query(user_input, context.history)
        if answer:
            context.state = DialogState.SATISFACTION
            return answer
        context.state = DialogState.HUMAN_ESCALATION
        return self._escalate_to_human(context)
 
    def _handle_action(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """执行业务动作"""
        try:
            result = self.action_executor.execute(
                context.current_intent, context.slots
            )
            context.state = DialogState.SATISFACTION
            return result
        except Exception as e:
            context.retry_count += 1
            if context.retry_count > self.MAX_RETRY:
                context.state = DialogState.HUMAN_ESCALATION
                return self._escalate_to_human(context)
            context.state = DialogState.ERROR_HANDLING
            return f"操作遇到问题，正在为您重试...（{e}）"
 
    def _handle_clarification(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """澄清状态——重新走意图识别"""
        context.state = DialogState.INTENT_RECOGNITION
        return self._handle_intent(user_input, context)
 
    def _handle_satisfaction(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """满意度确认"""
        positive_words = ["解决了", "好的", "谢谢", "可以了", "没问题"]
        negative_words = ["没解决", "不对", "不行", "还是不行", "不满意"]
 
        if any(w in user_input for w in positive_words):
            context.state = DialogState.CLOSING
            return "很高兴能帮到您！如有其他问题随时联系我们。"
 
        if any(w in user_input for w in negative_words):
            context.state = DialogState.HUMAN_ESCALATION
            return self._escalate_to_human(context)
 
        # 继续提问
        context.state = DialogState.INTENT_RECOGNITION
        return self._handle_intent(user_input, context)
 
    def _handle_error(
        self, user_input: str, context: ConversationContext
    ) -> str:
        """错误处理"""
        context.state = DialogState.ACTION_EXECUTION
        return self._handle_action(user_input, context)
 
    def _escalate_to_human(self, context: ConversationContext) -> str:
        """转接人工"""
        ticket_id = self.ticket_service.create_ticket(
            user_id=context.user_id,
            intent=context.current_intent,
            slots=context.slots,
            history=context.history,
        )
        return (
            f"正在为您转接人工客服，工单号：{ticket_id}。"
            f"请稍候，预计等待时间1-3分钟。"
        )

# src/ticket/router.py
"""
工单路由模块
根据用户属性、问题类型、紧急程度智能分配工单
"""
from typing import Dict, List, Optional
from dataclasses import dataclass
from enum import Enum
from datetime import datetime
import uuid
 
 
class Priority(Enum):
    P1_URGENT = 1
    P2_HIGH = 2
    P3_NORMAL = 3
    P4_LOW = 4
 
 
class SkillGroup(Enum):
    ORDER = "order"
    TECH = "tech"
    COMPLAINT = "complaint"
    GENERAL = "general"
 
 
@dataclass
class Ticket:
    ticket_id: str
    user_id: str
    priority: Priority
    skill_group: SkillGroup
    intent: str
    summary: str
    context: List[Dict]
    slots: Dict[str, str]
    assigned_to: Optional[str] = None
    status: str = "open"
    created_at: datetime = None
 
    def __post_init__(self):
        if not self.ticket_id:
            self.ticket_id = f"TK-{uuid.uuid4().hex[:8].upper()}"
        if not self.created_at:
            self.created_at = datetime.now()
 
 
class TicketRouter:
    """工单路由引擎"""
 
    # 意图 -> 技能组 映射
    INTENT_SKILL_MAP = {
        "order.track": SkillGroup.ORDER,
        "order.refund": SkillGroup.ORDER,
        "order.modify": SkillGroup.ORDER,
        "complaint.quality": SkillGroup.COMPLAINT,
        "complaint.service": SkillGroup.COMPLAINT,
        "complaint.delivery": SkillGroup.COMPLAINT,
        "account.password": SkillGroup.TECH,
    }
 
    def __init__(self, user_service, agent_pool):
        self.user_service = user_service
        self.agent_pool = agent_pool
 
    def create_and_route(
        self,
        user_id: str,
        intent: str,
        slots: Dict[str, str],
        history: List[Dict],
        sentiment_score: float = 0.0,
    ) -> Ticket:
        """创建工单并智能路由"""
        # 1. 评估优先级
        priority = self._evaluate_priority(
            user_id, intent, sentiment_score
        )
 
        # 2. 确定技能组
        skill_group = self._determine_skill_group(intent)
 
        # 3. 生成摘要（供人工快速了解上下文）
        summary = self._generate_summary(history)
 
        # 4. 创建工单
        ticket = Ticket(
            ticket_id="",
            user_id=user_id,
            priority=priority,
            skill_group=skill_group,
            intent=intent,
            summary=summary,
            context=history[-10:],  # 最近5轮对话
            slots=slots,
        )
 
        # 5. 分配客服
        agent = self.agent_pool.find_available(
            skill_group=skill_group, priority=priority
        )
        if agent:
            ticket.assigned_to = agent.agent_id
            ticket.status = "assigned"
 
        return ticket
 
    def _evaluate_priority(
        self, user_id: str, intent: str, sentiment_score: float
    ) -> Priority:
        """评估工单优先级"""
        user_info = self.user_service.get_user(user_id)
 
        # VIP用户 -> P1
        if user_info and user_info.get("is_vip"):
            return Priority.P1_URGENT
 
        # 投诉 + 强烈负面情感 -> P1
        if intent.startswith("complaint.") and sentiment_score < -0.8:
            return Priority.P1_URGENT
 
        # 投诉 -> P2
        if intent.startswith("complaint."):
            return Priority.P2_HIGH
 
        # 默认 -> P3
        return Priority.P3_NORMAL
 
    def _determine_skill_group(self, intent: str) -> SkillGroup:
        """根据意图确定技能组"""
        return self.INTENT_SKILL_MAP.get(intent, SkillGroup.GENERAL)
 
    @staticmethod
    def _generate_summary(history: List[Dict]) -> str:
        """从对话历史生成简要摘要"""
        user_msgs = [
            m["content"] for m in history if m["role"] == "user"
        ]
        if not user_msgs:
            return "无用户消息"
        # 简易摘要：取第一条和最后一条用户消息
        if len(user_msgs) == 1:
            return f"用户问题：{user_msgs[0][:100]}"
        return (
            f"用户首次提问：{user_msgs[0][:80]}... "
            f"最后一条：{user_msgs[-1][:80]}"
        )
