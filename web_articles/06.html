# 第4章/python
import jieba
 
# jieba 分词
text = "我今天学习了自然语言处理技术"
 
# 精确模式（最常用）
words = jieba.lcut(text)
print(words)  # ['我', '今天', '学习', '了', '自然语言', '处理', '技术']
 
# 全模式（尽可能多地切分）
words_full = jieba.lcut(text, cut_all=True)
print(words_full)  # ['我', '今天', '学习', '了', '自然', '自然语言', '语言', '处理', '技术']
 
# 搜索引擎模式
words_search = jieba.lcut_for_search(text)
print(words_search)  # ['我', '今天', '学习', '了', '自然', '语言', '自然语言', '处理', '技术']

4.1.4 TF-IDF：衡量词的重要性

TF-IDF（词频-逆文档频率）是一种统计方法，用来衡量一个词对一篇文档的重要程度。

图4-5

通俗理解：TF-IDF 的逻辑是——一个词如果在某篇文章中经常出现（TF 高），但在其他文章中很少出现（IDF 高），那它就是这篇文章的"关键词"。比如"Transformer"在一篇 AI 论文中频繁出现，但在大多数文章中不会出现，所以它的 TF-IDF 值就很高。
# 第4章/python
from sklearn.feature_extraction.text import TfidfVectorizer
 
# 示例文档
documents = [
    "大模型是人工智能的重要方向",
    "人工智能正在改变世界",
    "深度学习是大模型的基础技术",
    "Python 是学习人工智能的首选语言"
]
 
# 中文需要先分词
import jieba
docs_segmented = [" ".join(jieba.lcut(doc)) for doc in documents]
 
# 计算 TF-IDF
vectorizer = TfidfVectorizer()
tfidf_matrix = vectorizer.fit_transform(docs_segmented)
 
# 查看特征词
feature_names = vectorizer.get_feature_names_out()
print("特征词:", feature_names)
 
# 查看每篇文档的 TF-IDF 值
for i, doc in enumerate(documents):
    print(f"\n文档{i+1}: {doc}")
    scores = tfidf_matrix[i].toarray()[0]
    top_indices = scores.argsort()[-3:][::-1]  # Top 3
    for idx in top_indices:
        if scores[idx] > 0:
            print(f"  {feature_names[idx]}: {scores[idx]:.3f}")

4.2.4 GloVe：全局统计 + 局部上下文

GloVe（Global Vectors）是斯坦福在 2014 年提出的方法，结合了全局词共现统计和局部上下文窗口：

图4-10

Word2Vec vs GloVe 怎么选：实际使用中两者差别不大。现在更常见的做法是直接使用预训练大模型的 Embedding（如 BERT 的上下文化词向量），它们效果远超 Word2Vec 和 GloVe，因为能根据上下文动态调整词义。例如"苹果公司"和"吃苹果"中的"苹果"会得到不同的向量。
