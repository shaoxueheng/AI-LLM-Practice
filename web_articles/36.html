# src/data/cleaner.py
"""
领域数据清洗模块
实现五层过滤漏斗
"""
import re
import hashlib
from typing import List, Dict, Set, Optional, Tuple
from dataclasses import dataclass
from collections import defaultdict
 
 
@dataclass
class CleaningStats:
    """清洗统计信息"""
    total_input: int = 0
    after_dedup: int = 0
    after_language: int = 0
    after_quality: int = 0
    after_toxicity: int = 0
    after_relevance: int = 0
 
 
class DataCleaner:
    """五层漏斗数据清洗器"""
 
    def __init__(
        self,
        language_detector=None,
        quality_scorer=None,
        toxicity_detector=None,
        domain_classifier=None,
    ):
        self.language_detector = language_detector
        self.quality_scorer = quality_scorer
        self.toxicity_detector = toxicity_detector
        self.domain_classifier = domain_classifier
        self.seen_hashes: Set[str] = set()
        self.stats = CleaningStats()
 
    def clean_batch(self, documents: List[Dict]) -> List[Dict]:
        """对一批文档执行完整清洗流程"""
        self.stats.total_input = len(documents)
 
        # 第一层：去重
        docs = self._dedup(documents)
        self.stats.after_dedup = len(docs)
 
        # 第二层：语言过滤
        docs = self._filter_language(docs)
        self.stats.after_language = len(docs)
 
        # 第三层：质量过滤
        docs = self._filter_quality(docs)
        self.stats.after_quality = len(docs)
 
        # 第四层：毒性过滤
        docs = self._filter_toxicity(docs)
        self.stats.after_toxicity = len(docs)
 
        # 第五层：领域相关性过滤
        docs = self._filter_relevance(docs)
        self.stats.after_relevance = len(docs)
 
        return docs
 
    def _dedup(self, docs: List[Dict]) -> List[Dict]:
        """基于SimHash的近似去重"""
        unique_docs = []
        for doc in docs:
            text = doc.get("text", "")
            # 使用简化的n-gram指纹
            doc_hash = self._compute_fingerprint(text)
            if doc_hash not in self.seen_hashes:
                self.seen_hashes.add(doc_hash)
                unique_docs.append(doc)
        return unique_docs
 
    def _filter_language(self, docs: List[Dict]) -> List[Dict]:
        """保留中文和英文内容"""
        result = []
        for doc in docs:
            text = doc.get("text", "")
            cn_ratio = len(re.findall(r"[\u4e00-\u9fff]", text)) / max(
                len(text), 1
            )
            en_ratio = len(re.findall(r"[a-zA-Z]", text)) / max(
                len(text), 1
            )
            if cn_ratio > 0.3 or en_ratio > 0.5:
                doc["language"] = "zh" if cn_ratio > en_ratio else "en"
                result.append(doc)
        return result
 
    def _filter_quality(self, docs: List[Dict]) -> List[Dict]:
        """质量过滤：长度、完整性、信息密度"""
        result = []
        for doc in docs:
            text = doc.get("text", "")
            # 长度过滤
            if len(text) < 50 or len(text) > 100000:
                continue
            # 重复率检测（同一个词重复太多次 = 低质量）
            words = text.split()
            if len(words) > 10:
                unique_ratio = len(set(words)) / len(words)
                if unique_ratio < 0.3:
                    continue
            # 特殊字符比例
            special_ratio = len(
                re.findall(r"[^\u4e00-\u9fffa-zA-Z0-9\s。，！？、]", text)
            ) / max(len(text), 1)
            if special_ratio > 0.3:
                continue
            result.append(doc)
        return result
 
    def _filter_toxicity(self, docs: List[Dict]) -> List[Dict]:
        """毒性内容过滤"""
        # 简化版本：基于关键词黑名单
        # 生产环境建议使用专门的毒性检测模型
        result = []
        for doc in docs:
            text = doc.get("text", "")
            # 检查广告模式
            ad_patterns = [
                r"加微信", r"扫码领取", r"限时优惠",
                r"点击链接", r"免费领取",
            ]
            is_ad = any(re.search(p, text) for p in ad_patterns)
            if not is_ad:
                result.append(doc)
        return result
 
    def _filter_relevance(self, docs: List[Dict]) -> List[Dict]:
        """领域相关性过滤"""
        if not self.domain_classifier:
            return docs  # 无分类器则跳过
        result = []
        for doc in docs:
            text = doc.get("text", "")
            score = self.domain_classifier.predict(text)
            if score > 0.5:
                doc["relevance_score"] = score
                result.append(doc)
        return result
 
    @staticmethod
    def _compute_fingerprint(text: str, ngram: int = 5) -> str:
        """计算文本指纹（简化版SimHash）"""
        # 提取n-gram集合
        text_clean = re.sub(r"\s+", "", text)
        ngrams = set()
        for i in range(len(text_clean) - ngram + 1):
            ngrams.add(text_clean[i : i + ngram])
        # 对n-gram集合做哈希
        combined = "".join(sorted(ngrams))
        return hashlib.md5(combined.encode()).hexdigest()
 
    def get_stats(self) -> Dict:
        """获取清洗统计"""
        s = self.stats
        return {
            "total_input": s.total_input,
            "after_dedup": s.after_dedup,
            "after_language": s.after_language,
            "after_quality": s.after_quality,
            "after_toxicity": s.after_toxicity,
            "after_relevance": s.after_relevance,
            "retention_rate": (
                f"{s.after_relevance / max(s.total_input, 1):.1%}"
            ),
        }

# 第18章/python
# src/training/cpt_trainer.py
"""
继续预训练（Continue Pre-training）训练脚本
基于Hugging Face Transformers + DeepSpeed
"""
import os
import math
import logging
from dataclasses import dataclass, field
from typing import Optional, List
 
import torch
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    TrainingArguments,
    Trainer,
    DataCollatorForLanguageModeling,
)
from datasets import load_dataset, interleave_datasets
 
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)
 
 
@dataclass
class CPTConfig:
    """继续预训练配置"""
    # 基座模型
    base_model: str = "Qwen/Qwen2.5-7B"
    # 数据路径
    domain_data_paths: List[str] = field(default_factory=lambda: [
        "data/medical_textbooks",
        "data/clinical_cases",
        "data/medical_papers",
    ])
    general_data_path: str = "data/general_corpus"
    # 数据混合比例
    domain_ratio: float = 0.7
    general_ratio: float = 0.3
    # 训练参数
    max_seq_length: int = 4096
    learning_rate: float = 5e-6
    num_train_epochs: int = 2
    per_device_batch_size: int = 4
    gradient_accumulation_steps: int = 8
    warmup_ratio: float = 0.05
    weight_decay: float = 0.1
    lr_scheduler_type: str = "cosine"
    # 保存与评估
    output_dir: str = "output/cpt_medical"
    eval_steps: int = 500
    save_steps: int = 500
    logging_steps: int = 50
    # DeepSpeed配置
    deepspeed_config: Optional[str] = "config/ds_config_zero2.json"
 
 
class ContinuePreTrainer:
    """继续预训练执行器"""
 
    def __init__(self, config: CPTConfig):
        self.config = config
        self.tokenizer = None
        self.model = None
 
    def prepare_model(self):
        """加载基座模型"""
        logger.info(f"Loading base model: {self.config.base_model}")
        self.tokenizer = AutoTokenizer.from_pretrained(
            self.config.base_model, trust_remote_code=True
        )
        if self.tokenizer.pad_token is None:
            self.tokenizer.pad_token = self.tokenizer.eos_token
 
        self.model = AutoModelForCausalLM.from_pretrained(
            self.config.base_model,
            torch_dtype=torch.bfloat16,
            trust_remote_code=True,
        )
        # 启用梯度检查点节省显存
        self.model.gradient_checkpointing_enable()
 
        total_params = sum(p.numel() for p in self.model.parameters())
        logger.info(f"Model loaded. Total params: {total_params / 1e9:.2f}B")
 
    def prepare_data(self):
        """准备混合数据集"""
        logger.info("Preparing datasets...")
 
        # 加载领域数据
        domain_datasets = []
        for path in self.config.domain_data_paths:
            if os.path.exists(path):
                ds = load_dataset("text", data_dir=path, split="train")
                domain_datasets.append(ds)
                logger.info(f"  Domain data from {path}: {len(ds)} samples")
 
        # 加载通用数据
        general_ds = None
        if os.path.exists(self.config.general_data_path):
            general_ds = load_dataset(
                "text", data_dir=self.config.general_data_path, split="train"
            )
            logger.info(
                f"  General data: {len(general_ds)} samples"
            )
 
        # 混合数据集
        all_datasets = []
        probabilities = []
 
        if domain_datasets:
            # 领域数据均匀分配给定比例
            domain_prob_each = self.config.domain_ratio / len(domain_datasets)
            for ds in domain_datasets:
                all_datasets.append(ds)
                probabilities.append(domain_prob_each)
 
        if general_ds:
            all_datasets.append(general_ds)
            probabilities.append(self.config.general_ratio)
 
        # 归一化概率
        total = sum(probabilities)
        probabilities = [p / total for p in probabilities]
 
        mixed_dataset = interleave_datasets(
            all_datasets,
            probabilities=probabilities,
            seed=42,
            stopping_strategy="all_exhausted",
        )
 
        # 分词
        def tokenize_function(examples):
            return self.tokenizer(
                examples["text"],
                truncation=True,
                max_length=self.config.max_seq_length,
                padding=False,
            )
 
        tokenized = mixed_dataset.map(
            tokenize_function,
            batched=True,
            remove_columns=["text"],
            num_proc=8,
        )
 
        return tokenized
 
    def train(self):
        """执行训练"""
        self.prepare_model()
        dataset = self.prepare_data()
 
        # 拆分训练集和验证集（99:1）
        split = dataset.train_test_split(test_size=0.01, seed=42)
 
        # Data Collator：动态Padding + 随机Mask
        data_collator = DataCollatorForLanguageModeling(
            tokenizer=self.tokenizer,
            mlm=False,  # CLM，不是MLM
        )
 
        # 训练参数
        training_args = TrainingArguments(
            output_dir=self.config.output_dir,
            num_train_epochs=self.config.num_train_epochs,
            per_device_train_batch_size=self.config.per_device_batch_size,
            gradient_accumulation_steps=self.config.gradient_accumulation_steps,
            learning_rate=self.config.learning_rate,
            warmup_ratio=self.config.warmup_ratio,
            weight_decay=self.config.weight_decay,
            lr_scheduler_type=self.config.lr_scheduler_type,
            logging_steps=self.config.logging_steps,
            eval_steps=self.config.eval_steps,
            save_steps=self.config.save_steps,
            eval_strategy="steps",
            save_strategy="steps",
            save_total_limit=5,
            bf16=True,
            gradient_checkpointing=True,
            deepspeed=self.config.deepspeed_config,
            report_to="tensorboard",
            dataloader_num_workers=4,
            remove_unused_columns=False,
        )
 
        trainer = Trainer(
            model=self.model,
            args=training_args,
            train_dataset=split["train"],
            eval_dataset=split["test"],
            data_collator=data_collator,
        )
 
        logger.info("Starting Continue Pre-training...")
        train_result = trainer.train()
 
        # 保存最终模型
        trainer.save_model(os.path.join(self.config.output_dir, "final"))
        self.tokenizer.save_pretrained(
            os.path.join(self.config.output_dir, "final")
        )
 
        logger.info(f"Training complete. Metrics: {train_result.metrics}")
        return train_result
 
 
def main():
    """入口函数"""
    config = CPTConfig(
        base_model="Qwen/Qwen2.5-7B",
        domain_data_paths=[
            "data/medical_textbooks",
            "data/clinical_cases",
            "data/medical_papers",
        ],
        general_data_path="data/general_corpus",
        domain_ratio=0.7,
        general_ratio=0.3,
        output_dir="output/cpt_medical_qwen7b",
    )
 
    trainer = ContinuePreTrainer(config)
    trainer.train()
 
 
if __name__ == "__main__":
    main()
