# 第15章/python
def estimate_cost(
    model_size_b: float,      # 模型参数量（十亿）
    daily_requests: int,      # 日均请求数
    avg_tokens_per_req: int,  # 平均每请求输出Token数
    target_latency_ms: int,   # 目标延迟(毫秒)
):
    """简化的成本估算"""
    # 1. 估算单GPU吞吐量（vLLM, INT4量化）
    # 经验值：7B约1500 tokens/s, 72B约200 tokens/s
    throughput_per_gpu = 1500 / (model_size_b / 7)  # tokens/s
 
    # 2. 计算所需并发GPU数
    tokens_per_second_needed = (daily_requests * avg_tokens_per_req) / 86400
    # 加上延迟约束
    latency_factor = max(1, 1000 / target_latency_ms)
    gpus_needed = max(1, int(tokens_per_second_needed / throughput_per_gpu * latency_factor) + 1)
 
    # 3. 选择GPU类型
    vram_needed_gb = model_size_b * 2 * 0.5  # INT4大约每B参数0.5GB + KV Cache
    if vram_needed_gb <= 8:
        gpu_type, cost_per_hour = "RTX 4060", 0  # 自有
    elif vram_needed_gb <= 24:
        gpu_type, cost_per_hour = "RTX 4090", 0  # 自有
    elif vram_needed_gb <= 48:
        gpu_type, cost_per_hour = "L40S", 15.0
    else:
        gpu_type, cost_per_hour = "A100-80GB", 25.0
 
    # 4. 月成本
    monthly_gpu_cost = gpus_needed * cost_per_hour * 24 * 30
    monthly_api_cost = daily_requests * avg_tokens_per_req * 30 * 0.00002  # 假设API价格
 
    return {
        "gpus_needed": gpus_needed,
        "gpu_type": gpu_type,
        "monthly_self_host": f"¥{monthly_gpu_cost:,.0f}",
        "monthly_api_call": f"¥{monthly_api_cost:,.0f}",
        "recommendation": "自建" if monthly_gpu_cost < monthly_api_cost else "API调用"
    }
 
# 示例
result = estimate_cost(
    model_size_b=7,
    daily_requests=10000,
    avg_tokens_per_req=500,
    target_latency_ms=2000
)
print(result)

#!/bin/bash
# deploy_vllm.sh - vLLM 部署脚本
 
MODEL_NAME="Qwen/Qwen2.5-7B-Instruct-AWQ"
PORT=8000
GPU_MEMORY=0.9
MAX_MODEL_LEN=8192
 
echo "========================================="
echo "  vLLM 模型推理服务部署"
echo "========================================="
 
# 1. 安装依赖
pip install vllm>=0.6.0
 
# 2. 启动 vLLM 服务
python -m vllm.entrypoints.openai.api_server \
    --model $MODEL_NAME \
    --port $PORT \
    --host 0.0.0.0 \
    --dtype auto \
    --quantization awq \
    --gpu-memory-utilization $GPU_MEMORY \
    --max-model-len $MAX_MODEL_LEN \
    --trust-remote-code \
    --api-key "your-secret-key" \
    --served-model-name "qwen2.5-7b" \
    --enable-auto-tool-choice
 
echo "服务已启动: http://0.0.0.0:$PORT"
echo "API文档: http://0.0.0.0:$PORT/docs"

15.7.3 性能测试脚本

# 第15章/python
# benchmark.py - vLLM 性能基准测试
import asyncio
import time
import statistics
from openai import AsyncOpenAI
 
async def benchmark(
    base_url: str = "http://localhost:8000/v1",
    model: str = "qwen2.5-7b",
    num_requests: int = 100,
    concurrency: int = 10,
    max_tokens: int = 256,
):
    """并发性能测试"""
    client = AsyncOpenAI(base_url=base_url, api_key="your-secret-key")
 
    prompts = [
        "用100字介绍人工智能的发展历史。",
        "写一个Python快速排序函数。",
        "解释什么是机器学习中的过拟合。",
        "翻译成英文：大模型正在改变世界。",
    ]
 
    latencies = []
    token_counts = []
    errors = 0
    semaphore = asyncio.Semaphore(concurrency)
 
    async def single_request(prompt):
        nonlocal errors
        async with semaphore:
            start = time.time()
            try:
                response = await client.chat.completions.create(
                    model=model,
                    messages=[{"role": "user", "content": prompt}],
                    max_tokens=max_tokens,
                )
                elapsed = time.time() - start
                latencies.append(elapsed)
                token_counts.append(response.usage.completion_tokens)
            except Exception as e:
                errors += 1
                print(f"Error: {e}")
 
    # 执行并发测试
    print(f"开始测试: {num_requests}请求, 并发{concurrency}")
    start_time = time.time()
 
    tasks = [
        single_request(prompts[i % len(prompts)])
        for i in range(num_requests)
    ]
    await asyncio.gather(*tasks)
 
    total_time = time.time() - start_time
 
    # 计算统计指标
    total_tokens = sum(token_counts)
    print("\n====== 测试结果 ======")
    print(f"总请求数: {num_requests}")
    print(f"成功请求: {len(latencies)}")
    print(f"失败请求: {errors}")
    print(f"总耗时: {total_time:.2f}s")
    print(f"QPS: {len(latencies) / total_time:.2f}")
    print(f"总生成Token: {total_tokens}")
    print(f"吞吐量: {total_tokens / total_time:.0f} tokens/s")
    print(f"平均延迟: {statistics.mean(latencies):.3f}s")
    print(f"P50延迟: {statistics.median(latencies):.3f}s")
    print(f"P95延迟: {sorted(latencies)[int(len(latencies)*0.95)]:.3f}s")
    print(f"P99延迟: {sorted(latencies)[int(len(latencies)*0.99)]:.3f}s")
 
if __name__ == "__main__":
    asyncio.run(benchmark())

15.7.4 Nginx 负载均衡配置

# 第15章/nginx
# nginx.conf
upstream vllm_backend {
    least_conn;  # 最少连接数策略
    server 127.0.0.1:8000;
    server 127.0.0.1:8001;
}
 
server {
    listen 80;
    server_name ai-api.example.com;
 
    # 超时设置（LLM推理可能较慢）
    proxy_read_timeout 300s;
    proxy_send_timeout 300s;
 
    location /v1/ {
        proxy_pass http://vllm_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
 
        # SSE流式支持
        proxy_buffering off;
        proxy_cache off;
        chunked_transfer_encoding on;
    }
 
    location /health {
        proxy_pass http://vllm_backend;
    }
}
