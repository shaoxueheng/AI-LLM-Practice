# src/workflow/error_handler.py
"""
工作流异常处理中间件
实现重试、降级、人工介入的完整机制
"""
import asyncio
import logging
import time
from typing import Dict, Any, Optional, Callable
from dataclasses import dataclass, field
from enum import Enum
 
logger = logging.getLogger(__name__)
 
 
class ErrorSeverity(Enum):
    LOW = "low"           # 可忽略，记录日志
    MEDIUM = "medium"     # 需要重试或降级
    HIGH = "high"         # 需要人工介入
    CRITICAL = "critical"  # 需要终止工作流
 
 
@dataclass
class ErrorContext:
    """错误上下文"""
    task_id: str
    agent_role: str
    error_type: str
    error_message: str
    severity: ErrorSeverity
    retry_count: int = 0
    max_retries: int = 3
    timestamp: float = field(default_factory=time.time)
    metadata: Dict = field(default_factory=dict)
 
 
class ErrorHandler:
    """统一异常处理器"""
 
    def __init__(self, notification_service=None):
        self.notification_service = notification_service
        self.error_log: list = []
        self.human_intervention_queue: list = []
 
    async def handle(
        self,
        error: Exception,
        context: ErrorContext,
        retry_func: Optional[Callable] = None,
    ) -> Dict[str, Any]:
        """统一异常处理入口"""
        self.error_log.append(context)
 
        # 根据错误类型分派处理策略
        strategy = self._classify_error(error, context)
 
        if strategy == "retry":
            return await self._handle_retry(context, retry_func)
        elif strategy == "degrade":
            return self._handle_degrade(context)
        elif strategy == "human":
            return await self._handle_human_intervention(context)
        elif strategy == "abort":
            return self._handle_abort(context)
        else:
            return self._handle_skip(context)
 
    def _classify_error(
        self, error: Exception, context: ErrorContext
    ) -> str:
        """错误分类 -> 处理策略"""
        error_type = type(error).__name__
 
        # 超时 -> 重试
        if isinstance(error, (asyncio.TimeoutError, TimeoutError)):
            if context.retry_count < context.max_retries:
                return "retry"
            return "degrade"
 
        # 连接错误 -> 重试
        if isinstance(error, (ConnectionError, OSError)):
            if context.retry_count < context.max_retries:
                return "retry"
            return "human"
 
        # 数据验证错误 -> 降级
        if isinstance(error, (ValueError, KeyError)):
            return "degrade"
 
        # 权限错误 -> 人工
        if isinstance(error, PermissionError):
            return "human"
 
        # 严重错误 -> 终止
        if context.severity == ErrorSeverity.CRITICAL:
            return "abort"
 
        # 默认 -> 跳过
        return "skip"
 
    async def _handle_retry(
        self,
        context: ErrorContext,
        retry_func: Optional[Callable],
    ) -> Dict:
        """重试策略：指数退避"""
        wait_time = 2 ** context.retry_count
        logger.info(
            f"Retrying {context.task_id} in {wait_time}s "
            f"(attempt {context.retry_count + 1})"
        )
        await asyncio.sleep(wait_time)
 
        if retry_func:
            try:
                result = await retry_func()
                return {"action": "retried", "result": result}
            except Exception as e:
                context.retry_count += 1
                return await self.handle(e, context, retry_func)
 
        return {"action": "retry_failed"}
 
    def _handle_degrade(self, context: ErrorContext) -> Dict:
        """降级策略：使用备用方案或部分结果"""
        logger.warning(
            f"Degrading task {context.task_id}: {context.error_message}"
        )
        return {
            "action": "degraded",
            "message": (
                f"任务 {context.task_id} 已降级处理，"
                f"部分功能可能不可用"
            ),
        }
 
    async def _handle_human_intervention(
        self, context: ErrorContext
    ) -> Dict:
        """人工介入：发送通知，等待人工处理"""
        logger.warning(
            f"Human intervention required for {context.task_id}"
        )
 
        intervention = {
            "task_id": context.task_id,
            "error": context.error_message,
            "agent": context.agent_role,
            "timestamp": time.time(),
            "status": "pending",
        }
        self.human_intervention_queue.append(intervention)
 
        # 发送通知
        if self.notification_service:
            await self.notification_service.notify(
                title=f"工作流需要人工介入: {context.task_id}",
                message=(
                    f"Agent: {context.agent_role}\n"
                    f"错误: {context.error_message}"
                ),
                severity="high",
            )
 
        return {
            "action": "human_intervention",
            "intervention_id": context.task_id,
            "message": "已通知人工处理，工作流暂停等待",
        }
 
    def _handle_abort(self, context: ErrorContext) -> Dict:
        """终止策略"""
        logger.error(f"Aborting workflow due to {context.task_id}")
        return {
            "action": "aborted",
            "message": f"工作流因严重错误终止: {context.error_message}",
        }
 
    def _handle_skip(self, context: ErrorContext) -> Dict:
        """跳过策略"""
        logger.info(f"Skipping task {context.task_id}")
        return {
            "action": "skipped",
            "message": f"任务已跳过: {context.error_message}",
        }

"""
多Agent编排器
协调所有Agent完成端到端的数据分析报告生成任务
"""
import asyncio
import json
import logging
import time
from typing import Dict, Any, Optional
from dataclasses import dataclass, field
 
logger = logging.getLogger(__name__)
 
 
@dataclass
class WorkflowResult:
    """工作流执行结果"""
    workflow_id: str
    status: str  # "success" | "partial" | "failed"
    report: Optional[str] = None
    review: Optional[Dict] = None
    execution_log: list = field(default_factory=list)
    total_time_seconds: float = 0.0
    total_llm_calls: int = 0
 
 
class MultiAgentOrchestrator:
    """多Agent编排器"""
 
    def __init__(
        self,
        planner,
        data_collector,
        data_analyst,
        report_writer,
        quality_reviewer,
        error_handler,
        max_review_rounds: int = 3,
    ):
        self.planner = planner
        self.data_collector = data_collector
        self.data_analyst = data_analyst
        self.report_writer = report_writer
        self.quality_reviewer = quality_reviewer
        self.error_handler = error_handler
        self.max_review_rounds = max_review_rounds
 
        # Agent路由表
        self.agent_map = {
            "data_collector": self.data_collector,
            "data_analyst": self.data_analyst,
            "report_writer": self.report_writer,
            "quality_reviewer": self.quality_reviewer,
        }
 
    async def run(self, task_description: str) -> WorkflowResult:
        """执行完整的多Agent工作流"""
        start_time = time.time()
        result = WorkflowResult(
            workflow_id=f"wf_{int(time.time())}"
        )
 
        try:
            # ===== 阶段1: 任务规划 =====
            logger.info("[Phase 1] Task Planning")
            plan_result = await self.planner.execute(
                {"description": task_description}, {}
            )
 
            if plan_result["status"] != "success":
                result.status = "failed"
                result.execution_log.append(
                    {"phase": "planning", "error": plan_result}
                )
                return result
 
            dag = plan_result["dag"]
            logger.info(
                f"  Plan created: {len(dag['tasks'])} tasks"
            )
            result.execution_log.append(
                {"phase": "planning", "dag": dag}
            )
 
            # ===== 阶段2: 按DAG执行任务 =====
            logger.info("[Phase 2] Task Execution")
            context = {}  # 累积的上下文
 
            # 按拓扑排序执行
            for task_def in self._topological_sort(dag["tasks"]):
                agent_role = task_def.get("agent_role", "")
                agent = self.agent_map.get(agent_role)
 
                if not agent:
                    logger.warning(
                        f"  No agent for role: {agent_role}, skipping"
                    )
                    continue
 
                logger.info(
                    f"  Executing: {task_def['name']} "
                    f"(agent: {agent_role})"
                )
 
                task_result = await agent.execute(task_def, context)
 
                # 将结果加入上下文
                context[task_def["task_id"]] = task_result
                result.execution_log.append({
                    "phase": "execution",
                    "task": task_def["name"],
                    "result_status": task_result.get("status"),
                })
 
            # ===== 阶段3: 报告撰写 =====
            logger.info("[Phase 3] Report Writing")
            report_result = await self.report_writer.execute(
                {"requirements": "专业数据分析报告"}, context
            )
            report = report_result.get("report", "")
 
            # ===== 阶段4: 质量审核循环 =====
            logger.info("[Phase 4] Quality Review")
            for round_num in range(self.max_review_rounds):
                review_result = await self.quality_reviewer.execute(
                    {},
                    {
                        "report": report,
                        "data_summary": json.dumps(
                            context, ensure_ascii=False, default=str
                        )[:2000],
                    },
                )
 
                if review_result.get("passed"):
                    logger.info(
                        f"  Review passed (round {round_num + 1})"
                    )
                    result.review = review_result
                    break
 
                logger.info(
                    f"  Review rejected (round {round_num + 1}), "
                    f"revising..."
                )
 
                # 根据审核意见修改报告
                revision_result = await self.report_writer.execute(
                    {
                        "requirements": (
                            f"请根据以下审核意见修改报告：\n"
                            f"{review_result.get('review', '')}"
                        ),
                    },
                    {**context, "previous_report": report},
                )
                report = revision_result.get("report", report)
 
            result.report = report
            result.status = "success"
 
        except Exception as e:
            logger.error(f"Workflow failed: {e}")
            result.status = "failed"
            result.execution_log.append(
                {"phase": "error", "message": str(e)}
            )
 
        result.total_time_seconds = time.time() - start_time
        return result
 
    @staticmethod
    def _topological_sort(tasks: list) -> list:
        """对任务列表按依赖关系拓扑排序"""
        task_map = {t["task_id"]: t for t in tasks}
        in_degree = {t["task_id"]: 0 for t in tasks}
        graph = {t["task_id"]: [] for t in tasks}
 
        for t in tasks:
            for dep in t.get("dependencies", []):
                if dep in graph:
                    graph[dep].append(t["task_id"])
                    in_degree[t["task_id"]] += 1
 
        queue = [
            tid for tid, deg in in_degree.items() if deg == 0
        ]
        sorted_tasks = []
 
        while queue:
            tid = queue.pop(0)
            sorted_tasks.append(task_map[tid])
            for neighbor in graph[tid]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
 
        return sorted_tasks

序号经验详细说明
1先单后多先用单Agent跑通全流程，再拆分为多Agent
2通信要标准化定义清晰的消息协议，不要让Agent之间自由对话
3审核不可少每个关键输出都要有审核Agent把关
4工具权限最小化每个Agent只能调用它需要的工具
5超时和重试是必需品任何外部调用都可能失败，必须有应对措施
6日志要详细多Agent调试的复杂度是指数级的，日志是唯一的救命稻草
7成本要量化多Agent = 多次LLM调用 = 更高成本，必须做成本控制
8人工兜底永远不要假设AI能处理所有情况
