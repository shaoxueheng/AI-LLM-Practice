class ContractReviewer:
    """合同审核模块"""
 
    REVIEW_PROMPT = """你是一位资深法律顾问，请对以下合同条款进行专业审核。
 
审核维度：
1. 关键条款完整性：是否包含标的、价款、期限、违约责任、争议解决
2. 风险条款识别：单方面解约权、无限连带责任、不可抗力范围过窄
3. 金额与日期一致性
4. 合规性检查
 
请以JSON格式返回审核结果：
{
    "risk_level": "高/中/低",
    "missing_clauses": ["缺失的关键条款"],
    "risk_items": [
        {
            "clause": "原文引用",
            "risk_type": "风险类型",
            "risk_description": "风险说明",
            "suggestion": "修改建议"
        }
    ],
    "summary": "总体评价"
}"""
 
    def __init__(self, client: OpenAI):
        self.client = client
 
    def review(self, contract_text: str) -> dict:
        # 如果合同过长，分段审核后汇总
        if len(contract_text) > 6000:
            return self._review_long_contract(contract_text)
 
        response = self.client.chat.completions.create(
            model="qwen-plus",
            messages=[
                {"role": "system", "content": self.REVIEW_PROMPT},
                {"role": "user", "content": f"合同内容：\n{contract_text}"}
            ],
            response_format={"type": "json_object"},
            temperature=0.2  # 低温度保证严谨性
        )
 
        return json.loads(response.choices[0].message.content)
 
    def _review_long_contract(self, text: str) -> dict:
        """长合同分段审核"""
        # 按章节切分
        sections = self._split_by_sections(text)
        all_risks = []
 
        for section in sections:
            result = self.review(section)
            all_risks.extend(result.get("risk_items", []))
 
        # 汇总评估
        risk_level = "高" if len(all_risks) > 5 else "中" if len(all_risks) > 2 else "低"
        return {
            "risk_level": risk_level,
            "risk_items": all_risks,
            "summary": f"共发现 {len(all_risks)} 个风险点"
        }
 
    def _split_by_sections(self, text: str, max_len: int = 5000) -> list[str]:
        """按章节切分合同"""
        import re
        sections = re.split(r'\n(?=第[一二三四五六七八九十\d]+[条章节])', text)
        # 合并过短的节
        merged = []
        current = ""
        for s in sections:
            if len(current) + len(s) < max_len:
                current += "\n" + s
            else:
                if current:
                    merged.append(current)
                current = s
        if current:
            merged.append(current)
        return merged

class CodeAssistant:
    """代码助手：生成、审查、优化"""
 
    CODE_GEN_PROMPT = """你是一位资深软件工程师。请根据用户需求生成高质量代码。
 
要求：
1. 代码必须可直接运行，包含必要的import
2. 添加详细的中文文档字符串（docstring）
3. 使用类型提示（Type Hints）
4. 包含错误处理
5. 如适当，附带使用示例
6. 遵循PEP 8规范"""
 
    CODE_REVIEW_PROMPT = """你是一位代码审查专家。请对以下代码进行全面审查。
 
审查维度：
1. 功能正确性：逻辑是否正确
2. 代码质量：可读性、命名规范、代码结构
3. 性能：是否有性能隐患
4. 安全性：是否有安全漏洞（SQL注入、XSS等）
5. 最佳实践：是否遵循语言惯用写法
 
请以清晰格式给出审查意见，标注严重等级（严重/警告/建议）。"""
 
    def __init__(self, client: OpenAI, model: str = "qwen-plus"):
        self.client = client
        self.model = model
 
    def generate(self, description: str, language: str = "python",
                 context: str = "") -> str:
        """根据自然语言描述生成代码"""
        user_msg = f"请用 {language} 实现以下功能：\n{description}"
        if context:
            user_msg += f"\n\n相关上下文代码：\n```{language}\n{context}\n```"
 
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": self.CODE_GEN_PROMPT},
                {"role": "user", "content": user_msg}
            ],
            temperature=0.3
        )
        return response.choices[0].message.content
 
    def review(self, code: str, language: str = "python") -> str:
        """代码审查"""
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": self.CODE_REVIEW_PROMPT},
                {"role": "user", "content": f"```{language}\n{code}\n```"}
            ],
            temperature=0.2
        )
        return response.choices[0].message.content
 
    def optimize(self, code: str, focus: str = "性能") -> str:
        """代码优化"""
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "你是一位代码优化专家。"},
                {"role": "user", "content": (
                    f"请从 {focus} 角度优化以下代码，"
                    f"解释每处优化的原因：\n\n{code}"
                )}
            ],
            temperature=0.3
        )
        return response.choices[0].message.content
 
    def debug(self, code: str, error_msg: str) -> str:
        """错误调试"""
        response = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {"role": "system", "content": "你是一位调试专家。"},
                {"role": "user", "content": (
                    f"以下代码运行报错，请分析原因并给出修复方案：\n\n"
                    f"代码：\n{code}\n\n报错信息：\n{error_msg}"
                )}
            ],
            temperature=0.2
        )
        return response.choices[0].message.content

# 第14章/python
# writing_service.py - 写作服务核心
from fastapi import FastAPI, HTTPException
from fastapi.responses import StreamingResponse
from pydantic import BaseModel
from openai import OpenAI
import json
 
app = FastAPI(title="智能写作助手API")
client = OpenAI(api_key="your-key", base_url="https://api.example.com/v1")
 
class WriteRequest(BaseModel):
    action: str  # outline/expand/condense/rewrite/polish/translate/continue
    content: str
    style: str = "通俗易懂"
    target_length: int = 0  # 0表示自动
    language: str = "zh"
 
PROMPTS = {
    "outline": "请根据以下标题/主题，生成一个详细的文章大纲（使用Markdown标题格式）：\n\n{content}",
    "expand": "请将以下内容扩写，丰富细节和论述，保持{style}风格：\n\n{content}",
    "condense": "请将以下内容精简到原文的1/3左右，保留核心信息：\n\n{content}",
    "rewrite": "请用{style}的风格改写以下内容，保持核心意思不变：\n\n{content}",
    "polish": "请润色以下内容，提升文采和表达质量，不改变原意：\n\n{content}",
    "translate": "请将以下内容翻译为英文，保持专业性和流畅度：\n\n{content}",
    "continue": "请根据以下已有内容，自然续写下去（约300字）：\n\n{content}",
}
 
@app.post("/api/write")
async def write(request: WriteRequest):
    """写作处理接口"""
    if request.action not in PROMPTS:
        raise HTTPException(status_code=400, detail=f"不支持的操作: {request.action}")
 
    prompt = PROMPTS[request.action].format(
        content=request.content,
        style=request.style
    )
 
    async def generate():
        stream = client.chat.completions.create(
            model="qwen-plus",
            messages=[
                {"role": "system", "content": "你是一位专业写作助手，擅长各种文体的写作与编辑。"},
                {"role": "user", "content": prompt}
            ],
            stream=True,
            temperature=0.7
        )
        for chunk in stream:
            if chunk.choices[0].delta.content:
                yield f"data: {json.dumps({'content': chunk.choices[0].delta.content}, ensure_ascii=False)}\n\n"
        yield "data: [DONE]\n\n"
 
    return StreamingResponse(generate(), media_type="text/event-stream")
 
@app.post("/api/outline")
async def generate_outline(title: str, num_sections: int = 5):
    """根据标题生成文章大纲"""
    response = client.chat.completions.create(
        model="qwen-plus",
        messages=[
            {"role": "system", "content": "你是一位文章架构师，擅长设计清晰的文章结构。"},
            {"role": "user", "content": (
                f"请为标题「{title}」设计一个{num_sections}节的文章大纲。\n"
                f"要求：每节有标题和2-3个要点，返回Markdown格式。"
            )}
        ],
        temperature=0.8
    )
    return {"outline": response.choices[0].message.content}

# 第14章/python
# writing_app.py - Gradio 前端界面
import gradio as gr
import requests
import json
 
API_BASE = "http://localhost:8000"
 
def stream_write(action, content, style):
    """流式调用写作API"""
    response = requests.post(
        f"{API_BASE}/api/write",
        json={"action": action, "content": content, "style": style},
        stream=True
    )
 
    result = ""
    for line in response.iter_lines():
        if line:
            line = line.decode("utf-8")
            if line.startswith("data: ") and line != "data: [DONE]":
                data = json.loads(line[6:])
                result += data["content"]
                yield result
 
def generate_outline(title, num_sections):
    """生成大纲"""
    response = requests.post(
        f"{API_BASE}/api/outline",
        params={"title": title, "num_sections": num_sections}
    )
    return response.json()["outline"]

with gr.Blocks(title="智能写作助手", theme=gr.themes.Soft()) as demo:
    gr.Markdown("# 智能写作助手")
 
    with gr.Tabs():
        # Tab 1: 写作工作台
        with gr.TabItem("写作工作台"):
            with gr.Row():
                with gr.Column(scale=3):
                    editor = gr.Textbox(
                        label="编辑区",
                        lines=20,
                        placeholder="在此输入或粘贴你的文章内容..."
                    )
                with gr.Column(scale=1):
                    style = gr.Dropdown(
                        ["通俗易懂", "学术严谨", "生动活泼", "简洁商务"],
                        value="通俗易懂",
                        label="写作风格"
                    )
                    with gr.Group():
                        expand_btn = gr.Button("扩写", size="sm")
                        condense_btn = gr.Button("缩写", size="sm")
                        rewrite_btn = gr.Button("改写", size="sm")
                        polish_btn = gr.Button("润色", size="sm")
                        continue_btn = gr.Button("续写", size="sm")
                        translate_btn = gr.Button("翻译", size="sm")
 
                    result = gr.Textbox(label="AI输出", lines=15)
 
            # 绑定事件
            for btn, action in [
                (expand_btn, "expand"), (condense_btn, "condense"),
                (rewrite_btn, "rewrite"), (polish_btn, "polish"),
                (continue_btn, "continue"), (translate_btn, "translate")
            ]:
                btn.click(
                    fn=lambda c, s, a=action: stream_write(a, c, s),
                    inputs=[editor, style],
                    outputs=[result]
                )
 
        # Tab 2: 大纲生成
        with gr.TabItem("大纲生成"):
            title_input = gr.Textbox(label="文章标题", placeholder="输入标题...")
            sections_slider = gr.Slider(3, 10, value=5, step=1, label="章节数量")
            gen_outline_btn = gr.Button("生成大纲", variant="primary")
            outline_output = gr.Markdown(label="生成的大纲")
 
            gen_outline_btn.click(
                fn=generate_outline,
                inputs=[title_input, sections_slider],
                outputs=[outline_output]
            )
 
    demo.launch(server_name="0.0.0.0", server_port=7860)
